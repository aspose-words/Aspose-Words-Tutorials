// Inisialisasikan suatu objek Dokumen baru
Document doc = new Document();
// Buat Dokumen Builder untuk manipulasi dokumen yang mudah
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Sisipkan bagan kolom pada ukuran yang dinyatakan (tinggi lebar x)
Shape shape = builder.InsertChart(ChartType.Column, 432, 252);

//ExStep:2-
// Mengakses objek bagan dari bentuk yang disisipkan
Chart chart = shape.Chart;
// Tambahkan data seri dengan kategori dan nilai
chart.Series.Add("Aspose Series 1", new string[] { "Category 1", "Category 2" }, new double[] { 1, 2 });

//ExStep:3-
// Simpan dokumen dengan bagan kolom yang disisipkan
doc.Save("WorkingWithCharts.InsertColumnChart.docx");
