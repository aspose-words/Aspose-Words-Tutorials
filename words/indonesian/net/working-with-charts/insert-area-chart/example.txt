//ExSummary: Kode ini menunjukkan bagaimana memasukkan bagan area ke dalam dokumen Word menggunakan Aspose.Words.
//ExStepSummary:0: Kode berikut menginisialisasi dokumen baru dan pembuat dokumen.
//ExStepSummary:1: Kode berikut menyisipkan bagan area ke dalam dokumen.
//ExStepSummary:2: Kode berikut menambahkan seri ke bagan dengan titik data yang ditentukan.
//ExStepSummary:3: Kode berikut menyimpan dokumen dengan bagan area yang dimasukkan.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Membuat dokumen baru
Document doc = new Document();
// Inisialisasi pembangun dokumen
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Sisipkan suatu bagan area dengan dimensi yang dinyatakan
Shape shape = builder.InsertChart(ChartType.Area, 432, 252);

//ExStep:2-
// Mengakses objek bagan
Chart chart = shape.Chart;
// Tambahkan seri baru ke bagan
chart.Series.Add("Aspose Series 1", new [] 
{
	// Data point 1
    new DateTime(2002, 05, 01),
	// Data point 2
    new DateTime(2002, 06, 01),
	// Data point 3
    new DateTime(2002, 07, 01),
	// Data point 4
    new DateTime(2002, 08, 01),
	// Data point 5
    new DateTime(2002, 09, 01)
}, 
// Nilai yang sama untuk titik data
new double[] { 32, 32, 28, 12, 15 });

//ExStep:3-
// Simpan dokumen dengan bagan yang disisipkan
doc.Save("WorkingWithCharts.InsertAreaChart.docx");
//ExEnd
