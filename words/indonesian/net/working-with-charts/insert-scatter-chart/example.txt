//ExSummary:Kode ini menunjukkan bagaimana membuat dokumen Word dengan bagan sebar menggunakan Aspose.Words. Ini menginisialisasi sebuah dokumen, membangun isi, menyisipkan sebuah bagan tersebar, menambahkan data ke dalamnya, dan menyimpan dokumen.
//ExStepSummary:0: Inisialisasi objek Dokumen dan Dokumen Builder untuk mulai bekerja pada dokumen.
//ExStepSummary:1: Masukkan bagan penyebar ke dalam dokumen.
//ExStepSummary:2: Menambah serangkaian titik data ke bagan tersebar.
//ExStepSummary:3: Menyimpan dokumen dengan bagan tersebar ke berkas .docx.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Inisialisasi dokumen baru
Document doc = new Document();
// Buat pembangun untuk menyisipkan isi
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Sisipkan bagan penyebar ke dalam dokumen
Shape shape = builder.InsertChart(ChartType.Scatter, 432, 252); 

//ExStep:2-
// Dapatkan objek bagan dari bentuk yang disisipkan
Chart chart = shape.Chart;
// Tambahkan data seri ke bagan
chart.Series.Add("Aspose Series 1", new double[] { 0.7, 1.8, 2.6 }, new double[] { 2.7, 3.2, 0.8 });

//ExStep:3-
// Simpan dokumen dengan bagan tersebar
doc.Save("WorkingWithCharts.InsertScatterChart.docx");
//ExEnd
