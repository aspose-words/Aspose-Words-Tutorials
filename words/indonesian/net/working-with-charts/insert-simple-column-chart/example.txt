// Buat contoh dokumen Word baru.
Document doc = new Document();

//ExStep:1-
// Buat sebuah Docuter Builder untuk dengan mudah menyisipkan elemen ke dalam dokumen.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:2-
// Sisipkan suatu bentuk bagan kolom dengan lebar yang dinyatakan (432) dan tinggi (252).
Shape shape = builder.InsertChart(ChartType.Column, 432, 252);

//ExStep:3-
// Ambil objek bagan dari bentuk untuk manipulasi lebih lanjut.
Chart chart = shape.Chart;

//ExStep:4-
// Akses koleksi seri bagan tersebut.
ChartSeriesCollection seriesColl = chart.Series;
// Cetak jumlah seri dalam bagan ke konsol.
Console.WriteLine(seriesColl.Count);

//ExStep:5-
// Hapus setiap default seri yang secara otomatis dihasilkan ketika bagan dibuat.
seriesColl.Clear();

//ExStep:6-
// Definisikan sebuah array dari nama kategori untuk data bagan.
string[] categories = new string[] { "Category 1", "Category 2" };

//ExStep:7-
// Tambahkan beberapa seri ke bagan dengan nama kategori dan nilai data.
seriesColl.Add("Aspose Series 1", categories, new double[] { 1, 2 });
seriesColl.Add("Aspose Series 2", categories, new double[] { 3, 4 });
seriesColl.Add("Aspose Series 3", categories, new double[] { 5, 6 });
seriesColl.Add("Aspose Series 4", categories, new double[] { 7, 8 });
seriesColl.Add("Aspose Series 5", categories, new double[] { 9, 10 });

//ExStep:8-
// Simpan dokumen dengan bagan yang dimasukkan ke lokasi berkas yang ditentukan.
doc.Save("WorkingWithCharts.InsertSimpleColumnChart.docx");
