//ExStep:0-
// Buat instansi Dokumen baru
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Tambah teks untuk halaman pertama
builder.Writeln("This is page 1.");

//ExStep:2-
// Sisipkan istirahat halaman untuk memulai halaman baru
builder.InsertBreak(BreakType.PageBreak);

//ExStep:3-
// Tambah teks bagi halaman kedua
builder.Writeln("This is page 2.");

//ExStep:4-
// Masukkan halaman istirahat lain untuk halaman baru
builder.InsertBreak(BreakType.PageBreak);

//ExStep:5-
// Tambah teks untuk halaman ketiga
builder.Writeln("This is page 3.");

//ExStep:6-
// Menyimpan dokumen pada lokasi yang ditentukan
doc.Save("AddContentUsingDocumentBuilder.InsertBreak.docx");
