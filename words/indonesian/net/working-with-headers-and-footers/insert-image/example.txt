//ExSummary: Kode ini menunjukkan bagaimana memasukkan gambar ke dalam header dokumen Word menggunakan Aspose.Words untuk .NET.
//ExStepSummary:0: Buat dokumen Word baru menggunakan kelas Dokumen Aspose.Words.
//ExStepSummary:1: Pindah ke tajuk utama dokumen menggunakan DokumenBuilder.
//ExStepSummary:2: Masukkan suatu gambar ke dalam header dengan pilihan penyelarasan, dimensi, dan pembungkus yang spesifik.
//ExStepSummary:3: Simpan dokumen ke lokasi berkas yang dinyatakan.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Buat dokumen Word baru
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Pindahkan kursor ke tajuk utama dokumen
builder.MoveToHeaderFooter(HeaderFooterType.HeaderPrimary);

//ExStep:2-
// Sisipkan gambar ke header dengan alignmen dan dimensi spesifik
builder.InsertImage(
    ImagesDir + "Logo.jpg", RelativeHorizontalPosition.RightMargin, 10,
    RelativeVerticalPosition.Page, 10, 50, 50, 
    WrapType.Through
);

//ExStep:3-
// Menyimpan dokumen pada lokasi yang ditentukan
doc.Save("WorkingWithHeadersAndFooters.InsertImage.docx");
//ExEnd
