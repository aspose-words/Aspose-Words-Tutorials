//ExStart
//ExSummary: Kode ini menunjukkan bagaimana menerapkan batas dan shading ke sebuah paragraf dalam dokumen Word menggunakan Aspose.Words.
//ExStepSummary:0: Kode berikut menginisialisasi suatu instansi Dokumen dan Dokumen Builder baru.
//ExStepSummary:1: Kode berikut mengatur batas paragraf dengan jarak dari teks.
//ExStepSummary:2: Kode berikut menerapkan tekstur silang diagonal yang bermuara ke paragraf dengan warna - warna tertentu.
//ExStepSummary:3: Kode berikut menulis paragraf keterangan dengan pengaturan yang diterapkan.
//ExStepSummary:4: Kode berikut menyimpan dokumen ke lokasi keluaran yang ditentukan.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0-
// Inisialisasi dokumen baru
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Akses batas paragraf
BorderCollection borders = builder.ParagraphFormat.Borders;
borders.DistanceFromText = 20;
borders[BorderType.Left].LineStyle = LineStyle.Double;
borders[BorderType.Right].LineStyle = LineStyle.Double;
borders[BorderType.Top].LineStyle = LineStyle.Double;
borders[BorderType.Bottom].LineStyle = LineStyle.Double;

//ExStep:2-
// Akses properti shading paragraf
Shading shading = builder.ParagraphFormat.Shading;
shading.Texture = TextureIndex.TextureDiagonalCross;
shading.BackgroundPatternColor = System.Drawing.Color.LightCoral; 
shading.ForegroundPatternColor = System.Drawing.Color.LightSalmon;

//ExStep:3-
// Tulis isi paragraf terformat
builder.Write("I'm a formatted paragraph with double border and nice shading.");

//ExStep:4-
// Menyimpan dokumen pada lokasi yang ditentukan
doc.Save(ArtifactsDir + "DocumentFormatting.ApplyBordersAndShadingToParagraph.doc");
//ExEnd
