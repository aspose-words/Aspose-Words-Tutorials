//ExSummary: Kode ini menunjukkan bagaimana format paragraf dalam dokumen Word menggunakan Aspose.Words. Ini menunjukkan bagaimana mengatur pilihan format paragraf seperti penyelarasan, indentasi, dan spasi setelah paragraf.
//ExStepSummary:0: Langkah ini menginisialisasi Dokumen baru dan Dokumen Builder untuk membuat dokumen.
//ExStepSummary:1: Langkah ini menetapkan opsi pemformatan paragraf, termasuk penyelarasan, indentasi kiri, indentasi kanan, dan ruang setelah paragraf.
//ExStepSummary:2: Langkah ini menulis paragraf yang dibentuk dengan baik dengan opsi pemformatan yang ditentukan pada dokumen.
//ExStepSummary:3: Langkah ini menulis paragraf keterangan lain yang memperlihatkan pengaruh ruang setelah paragraf.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0- 
// Buat instansi Dokumen baru.
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1- 
// Atur opsi pemformatan paragraf.
ParagraphFormat paragraphFormat = builder.ParagraphFormat;
paragraphFormat.Alignment = ParagraphAlignment.Center;
paragraphFormat.LeftIndent = 50;
paragraphFormat.RightIndent = 50;
paragraphFormat.SpaceAfter = 25;

//ExStep:2- 
// Tulis paragraf pertama.
builder.Writeln(
	"I'm a very nice formatted paragraph. I'm intended to demonstrate how the left and right indents affect word wrapping.");

//ExStep:3- 
// Tulis paragraf kedua.
builder.Writeln(
	"I'm another nice formatted paragraph. I'm intended to demonstrate how the space after paragraph looks like.");

//ExStep:4- 
// Simpan dokumen ke lokasi yang ditentukan.
doc.Save(ArtifactsDir + "DocumentFormatting.ParagraphFormatting.docx");
//ExEnd
