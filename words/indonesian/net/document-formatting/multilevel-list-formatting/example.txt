//ExStepSummary:2: Inden kode berikut untuk membuat sub-item di bawah item tingkat atas kedua.
//ExStepSummary:3: Kode berikut lebih lanjut indentasi untuk membuat sub-sub-item dibawah sub- item kedua.
//ExStepSummary:4: Outdents kode berikut kembali ke tingkat sebelumnya untuk menambahkan item lain pada tingkat kedua.
//ExStepSummary:5: Kode berikut outdents kembali ke tingkat atas untuk menambahkan item akhir dalam daftar.
//ExStepSummary:6: Kode berikut menghapus nomor dari daftar sebelum menyimpan.

//ExStep:0-
// Inisialisasi dokumen baru
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Terapkan format bilangan baku
builder.ListFormat.ApplyNumberDefault();
builder.Writeln("Item 1");
builder.Writeln("Item 2");

//ExStep:2-
// Indentasi untuk sub-item
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.1");
builder.Writeln("Item 2.2");

//ExStep:3-
// Inden untuk sub-sub-item
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.2.1");
builder.Writeln("Item 2.2.2");

//ExStep:4-
// Outdent kembali ke tingkat kedua
builder.ListFormat.ListOutdent();
builder.Writeln("Item 2.3");

//ExStep:5-
// Outdent kembali ke level paling atas
builder.ListFormat.ListOutdent();
builder.Writeln("Item 3");

//ExStep:6-
// Hapus bilangan dari daftar
builder.ListFormat.RemoveNumbers();

//ExStep:7-
// Simpan dokumen
doc.Save("DocumentFormatting.MultilevelListFormatting.docx");
//ExEnd
