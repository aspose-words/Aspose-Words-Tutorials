//ExSummary: Kode ini menunjukkan bagaimana membuat dokumen Word menggunakan Aspose.Words, menerapkan gaya paragraf, dan menyimpan dokumen. Ini menunjukkan bagaimana menata gaya judul untuk paragraf dan menyimpan dokumen dalam format .dox.
//ExStepSummary:0: Bagian ini menginisialisasi suatu instansi Dokumen dan Dokumen Builder baru.
//ExStepSummary:1: Bagian ini menerapkan gaya paragraf judul menggunakan Dokumen Builder.
//ExStepSummary:2: Bagian ini menulis "Hello" ke dokumen.
//ExStepSummary:3: Bagian ini menyimpan dokumen ke lokasi berkas yang ditentukan.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Inisialisasi suatu instansi Dokumen baru
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Tata gaya paragraf ke Judul
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Title;

//ExStep:2-
// Tulis "Hello" menggunakan pembangun, menerapkan gaya yang ditentukan
builder.Write("Hello");

//ExStep:3-
// Simpan dokumen ke berkas .docx
doc.Save("DocumentFormatting.ApplyParagraphStyle.docx");
