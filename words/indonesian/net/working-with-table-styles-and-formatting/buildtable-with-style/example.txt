//ExStepSummary:0: Inisialisasi Dokumen dan Dokumen Builder baru untuk mengelola struktur dokumen.
//ExStepSummary:1: Mulai tabel baru dalam dokumen menggunakan Dokumen Builder.
//ExStepSummary:2: Masukkan sel pertama dan atur identifier gaya tabel.
//ExStepSummary:3: Terapkan opsi pemformatan ke tabel dan atur perilaku auto-fit.
//ExStepSummary:4: Tambahkan sel baris kepala ke tabel.
//ExStepSummary:5: Sisipkan data baris ke dalam tabel dengan nama butir dan jumlah.
//ExStepSummary:6: Simpan dokumen yang dirakit ke suatu berkas.
//ExStepImage:2:images/1.png
//ExStepImage:3:images/2.png
//ExStepImage:4:images/3.png

//ExStep:0-
// Buat objek Dokumen baru.
Document doc = new Document();
// Buat pembangun untuk memfasilitasi pembuatan dokumen.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Memulai tabel baru di dokumen.
Table table = builder.StartTable();

//ExStep:2-
// Masukkan sel pertama sebelum menata format.
builder.InsertCell();
// Atur identifier gaya bagi tabel.
table.StyleIdentifier = StyleIdentifier.MediumShading1Accent1;


//ExStep:3-
// Konfigurasikan opsi gaya.
table.StyleOptions = 
    TableStyleOptions.FirstColumn | TableStyleOptions.RowBands | TableStyleOptions.FirstRow;
// Mengatur perilaku autofit meja.	
table.AutoFit(AutoFitBehavior.AutoFitToContents);


//ExStep:4-
// Tulis header untuk kolom pertama.
builder.Writeln("Item");
// Atur padding kanan untuk sel.
builder.CellFormat.RightPadding = 40;
// Masukkan sel kedua untuk kolom berikutnya.
builder.InsertCell();
// Tulis header untuk kolom kedua.
builder.Writeln("Quantity (kg)");
// Akhiri pembuatan baris.
builder.EndRow();

//ExStep:5-
// Mulai baris baru dan masukkan sel pertama.
builder.InsertCell();
builder.Writeln("Apples");
builder.InsertCell();
builder.Writeln("20");
builder.EndRow();
 // Masukkan sel baris pertama untuk item kedua.
builder.InsertCell();
builder.Writeln("Bananas");
builder.InsertCell();
builder.Writeln("40");
builder.EndRow();
// Mulai baris baru untuk item ketiga.
builder.InsertCell();
builder.Writeln("Carrots");
builder.InsertCell(); 
builder.Writeln("50");
builder.EndRow();

//ExStep:6-
// Simpan dokumen ke berkas yang dinyatakan.
doc.Save("WorkingWithTableStylesAndFormatting.BuildTableWithStyle.docx");
//ExEnd
