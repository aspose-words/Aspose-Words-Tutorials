//ExStepSummary:1: Langkah ini memulai tabel dan memasukkan sel ke dalamnya.
//ExStepSummary:2: Langkah ini mengatur format untuk baris, termasuk aturan tinggi dan tinggi badan.
//ExStepSummary:3: Langkah ini mengatur padding untuk seluruh tabel, mempengaruhi semua baris.
//ExStepSummary:4: Langkah ini menambahkan teks ke sel dan mengakhiri baris.
//ExStepSummary:5: Langkah ini mengakhiri tabel dan menyimpan dokumen ke lokasi yang ditentukan.

//ExStep:0-
// Buat instansi Dokumen baru
Document doc = new Document();
// Inisialisasi DokumenBuilder dengan dokumen
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Mulai tabel baru
Table table = builder.StartTable();
// Masukkan sel pertama ke dalam tabel
builder.InsertCell();

//ExStep:2-
// Akses properti RowFormat dari pembangun
RowFormat rowFormat = builder.RowFormat;
// Menentukan tinggi baris
rowFormat.Height = 100;
// Tentukan aturan tinggi dengan tepat
rowFormat.HeightRule = HeightRule.Exactly;

//ExStep:3-
// Atur padding untuk tabel
table.LeftPadding = 30;
table.RightPadding = 30;
table.TopPadding = 30;
table.BottomPadding = 30;

//ExStep:4-
// Tambahkan teks ke sel kini
builder.Writeln("I'm a wonderful formatted row.");
// Akhir baris ini
builder.EndRow();
builder.EndTable();

//ExStep:5-
// Menyimpan dokumen pada lokasi yang ditentukan
doc.Save("WorkingWithTableStylesAndFormatting.DocumentBuilderSetTableRowFormatting.docx");
//ExEnd
