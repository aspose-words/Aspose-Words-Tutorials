//ExSummary: Tento kód ukazuje, jak vytvořit dokument Word se stylovou tabulkou pomocí Aspose.Words. 
//ExStepSummary:0: Inicializovat nový dokument a dokumentBuilder.
//ExStepSummary:1: Spusťte novou tabulku a vložte hlavičkové buňky pro "Název" a "Value."
//ExStepSummary:2: Vytvořit vlastní TableStyle s konkrétní hranou a nastavením vycpávky.
//ExStepSummary:3: Uložit dokument s novým stylem tabulky.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Vytvořit nový objekt Dokumentu
Document doc = new Document();
// Inicializovat dokumentBuilder pro sestavení dokumentu
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Spustit nový stůl
Table table = builder.StartTable();
builder.InsertCell();
builder.Write("Name");
builder.InsertCell();
builder.Write("Value");
builder.EndRow();
builder.InsertCell();
builder.InsertCell();
builder.EndTable();

//ExStep:2-
// Vytvořit nový styl tabulky
TableStyle tableStyle = (TableStyle) doc.Styles.Add(StyleType.Table, "MyTableStyle1");
tableStyle.Borders.LineStyle = LineStyle.Double;
// Nastavte šířku okraje
tableStyle.Borders.LineWidth = 1;
// Nastavte vycpávku.
tableStyle.LeftPadding = 18;
tableStyle.RightPadding = 18;
tableStyle.TopPadding = 12;
tableStyle.BottomPadding = 12;
// Použít vlastní styl tabulky na vytvořenou tabulku
table.Style = tableStyle;

//ExStep:3-
// Uložit dokument se stylovou tabulkou
doc.Save("WorkingWithTableStylesAndFormatting.CreateTableStyle.docx");
//ExEnd
