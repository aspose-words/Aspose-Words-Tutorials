//ExSummary: Tento kód ukazuje, jak vytvořit dokument Word se stylovou tabulkou pomocí Aspose.Words. Tabulka obsahuje hlavičky a datové řádky, formátované podle konkrétních stylových možností.
//ExStepSummary:0: Inicializovat nový dokument a dokumentBuilder pro správu struktury dokumentu.
//ExStepSummary:1: Spusťte novou tabulku v dokumentu pomocí DocumentBuilder.
//ExStepSummary:2: Vložte první buňku a nastavte identifikátor stylu tabulky.
//ExStepSummary:3: Aplikujte možnosti formátování do tabulky a nastavte automatické chování.
//ExStepSummary:4: Přidat buňky z řad hlaviček do tabulky.
//ExStepSummary:5: Vložte do tabulky datové řádky s názvy položek a množstvími.
//ExStepSummary:6: Uložit sestavený dokument do souboru.

//ExStep:0-
// Vytvořit nový objekt Dokumentu.
Document doc = new Document();
// Vytvořit stavitele pro usnadnění tvorby dokumentů.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Spusťte novou tabulku v dokumentu.
Table table = builder.StartTable();

//ExStep:2-
// Vložte první buňku před nastavením formátování.
builder.InsertCell();
// Nastavte identifikátor stylu tabulky.
table.StyleIdentifier = StyleIdentifier.MediumShading1Accent1;


//ExStep:3-
// Nastavit možnosti stylu.
table.StyleOptions = 
    TableStyleOptions.FirstColumn | TableStyleOptions.RowBands | TableStyleOptions.FirstRow;
// Nastavte automatické chování stolu.	
table.AutoFit(AutoFitBehavior.AutoFitToContents);


//ExStep:4-
// Napiš hlavičku do prvního sloupce.
builder.Writeln("Item");
// Nastavit správné vycpávky do cely.
builder.CellFormat.RightPadding = 40;
// Vložte druhou buňku do dalšího sloupce.
builder.InsertCell();
// Napiš hlavičku do druhého sloupce.
builder.Writeln("Quantity (kg)");
// Ukončete vytvoření řady.
builder.EndRow();

//ExStep:5-
// Spusťte novou řadu a vložte první buňku.
builder.InsertCell();
builder.Writeln("Apples");
builder.InsertCell();
builder.Writeln("20");
builder.EndRow();
 // Pro druhou položku vložte první buňku nového řádku.
builder.InsertCell();
builder.Writeln("Bananas");
builder.InsertCell();
builder.Writeln("40");
builder.EndRow();
// Začněte novou řadu pro třetí položku.
builder.InsertCell();
builder.Writeln("Carrots");
builder.InsertCell(); 
builder.Writeln("50");
builder.EndRow();

//ExStep:6-
// Uložit dokument do zadaného souboru.
doc.Save("WorkingWithTableStylesAndFormatting.BuildTableWithStyle.docx");
