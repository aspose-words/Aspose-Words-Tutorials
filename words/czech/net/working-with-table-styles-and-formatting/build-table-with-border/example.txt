//ExStepSummary:0: Následující kód nastaví dokument a získá z něj první tabulku.
//ExStepSummary:1: Následující kód vymaže všechny existující hranice z tabulky, aby se připravil na nové formátování.

//ExStep:0-
// Definovat adresář pro dokument
string dataDir = "Your Document Directory";
// Načíst dokument
Document doc = new Document(dataDir + "Tables.docx");

//ExStep:1-
// Získejte první tabulku z dokumentu
Table table = (Table) doc.GetChild(NodeType.Table, 0, true);

//ExStep:2-
// Vyčistit všechny existující hranice od tabulky
table.ClearBorders();

//ExStep:3-
// Nastavit zelenou hranici kolem a uvnitř stolu
table.SetBorders(LineStyle.Single, 1.5, Color.Green);
// Uložit dokument se změnami
doc.Save(dataDir + "WorkingWithTableStylesAndFormatting.BuildTableWithBorders.docx");
