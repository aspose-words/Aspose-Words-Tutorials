//ExSummary: Tento kód ukazuje, jak aplikovat obrysu hranice na tabulku v dokumentu Word pomocí Aspose.Slova. Soustředí tabulku na stránce, nastaví zelenou hranici kolem tabulky, vyčistí všechny existující hranice, a vyplní buňky tabulky s světle zelenou barvu pozadí. 
//ExStepSummary:0: Následující kód inicializuje dokument a získá první tabulku.
//ExStepSummary:1: Tento krok sladí tabulku se středem stránky.
//ExStepSummary:2: Tento krok vymaže všechny existující hranice od tabulky.
//ExStepSummary:3: Tento krok stanoví zelenou hranici kolem stolu, ale ne uvnitř.
//ExStepSummary:4: Tento krok naplní buňky světle zelenou pevnou barvou.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStepImage:4:images/4.png
//ExStep:0-
// Definovat adresář dokumentu
string dataDir = "Your Document Directory";
// Načíst dokument
Document doc = new Document(dataDir + "Tables.docx");

//ExStep:1-
// Získejte první tabulku z dokumentu
Table table = (Table)doc.GetChild(NodeType.Table, 0, true);
// Zarovnat stůl do středu stránky
table.Alignment = TableAlignment.Center;

//ExStep:2-
// Vyčistit všechny existující hranice od tabulky
table.ClearBorders();

//ExStep:3-
// Nastavit levou hranici
table.SetBorder(BorderType.Left, LineStyle.Single, 1.5, Color.Green, true);
// Nastavit pravou hranici
table.SetBorder(BorderType.Right, LineStyle.Single, 1.5, Color.Green, true);
// Nastavit horní hranici
table.SetBorder(BorderType.Top, LineStyle.Single, 1.5, Color.Green, true);
// Nastavit dolní hranici
table.SetBorder(BorderType.Bottom, LineStyle.Single, 1.5, Color.Green, true);

//ExStep:4-
// Naplňte buňky světle zelenou barvou
table.SetShading(TextureIndex.TextureSolid, Color.LightGreen, Color.Empty);
// Uložit upravený dokument
doc.Save(dataDir + "WorkingWithTableStylesAndFormatting.ApplyOutlineBorder.docx");
//ExEnd
