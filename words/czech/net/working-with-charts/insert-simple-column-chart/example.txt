//ExStepSummary:4: Tento kód ukazuje, jak se dostat k kolekci řady grafů a vytisknout počet řad.
//ExStepSummary:5: Tento kód vymaže všechny defaultní řady, které mohly být vytvořeny s grafem.
//ExStepSummary:6: Tento kód ukazuje, jak definovat názvy kategorií údajů grafu.
//ExStepSummary:7: Tento kód ukazuje, jak přidat řady do grafu s odpovídajícími názvy kategorií a hodnotami dat.

//ExStep:0-
// Vytvořit nový příklad dokumentu Word.
Document doc = new Document();

//ExStep:1-
// Vytvořit DocumentBuilder pro snadné vložení prvků do dokumentu.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:2-
// Vložte tvar sloupcového grafu se zadanou šířkou (432) a výškou (252).
Shape shape = builder.InsertChart(ChartType.Column, 432, 252);

//ExStep:3-
// Získejte objekt grafu z tvaru pro další manipulaci.
Chart chart = shape.Chart;

//ExStep:4-
// Přístup k sérii kolekci grafu.
ChartSeriesCollection seriesColl = chart.Series;
// Počet řad v grafu vytiskněte do konzole.
Console.WriteLine(seriesColl.Count);

//ExStep:5-
// Odstraňte všechny defaultní řady, které byly automaticky generovány při vytvoření grafu.
seriesColl.Clear();

//ExStep:6-
// Definujte řadu jmen kategorií pro data grafu.
string[] categories = new string[] { "Category 1", "Category 2" };

//ExStep:7-
// Do grafu přidejte více sérií s názvy kategorií a hodnotami dat.
seriesColl.Add("Aspose Series 1", categories, new double[] { 1, 2 });
seriesColl.Add("Aspose Series 2", categories, new double[] { 3, 4 });
seriesColl.Add("Aspose Series 3", categories, new double[] { 5, 6 });
seriesColl.Add("Aspose Series 4", categories, new double[] { 7, 8 });
seriesColl.Add("Aspose Series 5", categories, new double[] { 9, 10 });

//ExStep:8-
// Uložit dokument s vloženou mapou na zadanou cestu souboru.
doc.Save("WorkingWithCharts.InsertSimpleColumnChart.docx");
