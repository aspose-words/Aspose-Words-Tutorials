//ExStart
//ExSummary: Tento kód ukazuje, jak vložit Bubble graf do Word dokumentu pomocí Aspose.Words.
//ExStepSummary:0: Následující kód inicializuje nový dokument a dokumentBuilder.
//ExStepSummary:1: Do dokumentu se vytvoří a přidá mapa Bubble pomocí nástroje DocumentBuilder.
//ExStepSummary:2: Datová řada Bubbleova grafu je přidána s konkrétními hodnotami X, Y a velikostí.
//ExStepSummary:3: Nakonec je dokument uložen s vloženým Bubble grafem.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Inicializovat nový dokument a dokumentBuilder
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Vložit do dokumentu mapu bubliny
Shape shape = builder.InsertChart(ChartType.Bubble, 432, 252);

//ExStep:2-
// Přidat datové řady do Bubble grafu
Chart chart = shape.Chart;
chart.Series.Add("Aspose Series 1", new double[] { 0.7, 1.8, 2.6 }, // X values
    new double[] { 2.7, 3.2, 0.8 }, // Y values
    new double[] { 10, 4, 8 }); // Size values for the bubbles

//ExStep:3-
// Uložit dokument s Bubble grafem
doc.Save("WorkingWithCharts.InsertBubbleChart.docx");
//ExEnd
