//ExStepSummary:0: Následující kód ukazuje, jak vytvořit nový dokument a instantiate a DocumentBuilder:
//ExStepSummary:1: Následující kód ukazuje, jak do dokumentu vložit graf řádku:
//ExStepSummary:2: Následující kód vysvětluje, jak nastavit název grafu:
//ExStepSummary:3: Následující kód ukazuje, jak nastavit polohu a nastavení překrytí legendy grafu:
//ExStepImage:1:images/1.png

//ExStepImage:3:images/3.png
//ExStep:0-
// Vytvořit nový objekt Dokumentu
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Vložit do dokumentu graf řádku ve stanovených rozměrech
Shape shape = builder.InsertChart(ChartType.Line, 432, 252);

//ExStep:2-
// Přístup k objektu Graf z tvaru
Chart chart = shape.Chart;
// Povolit viditelnost názvu mapy
chart.Title.Show = true;
// Nastavit text názvu grafu
chart.Title.Text = "Line Chart Title";
// Ujistěte se, že název není překryv grafu
chart.Title.Overlay = false;

//ExStep:3-
// Nastavte polohu legendy grafu vlevo
chart.Legend.Position = LegendPosition.Left;
chart.Legend.Overlay = true;

//ExStep:4-
// Uložit dokument s grafem do souboru
doc.Save("WorkingWithCharts.CreateChartUsingShape.docx");
//ExEnd
