//ExStepSummary:0: Inicializace objektu Dokumentu a DocumentBuilder pro sestavení dokumentu.
//ExStepSummary:1: Vložení obrázku do dokumentu na aktuální pozici kurzoru s DocumentBuilderem.
//ExStepSummary:2: Uložení dokumentu na zadaný název souboru.
//ExStepImage:1:images/1.png
//ExStep:0-
// Inicializovat nový dokument a dokumentBuilder
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Vložit obrázek pomocí DocumentBuilder
builder.InsertImage(ImagesDir + "Transparent background logo.png");

//ExStep:2-
// Uložit upravený dokument do souboru
doc.Save("AddContentUsingDocumentBuilder.InsertInlineImage.docx");
