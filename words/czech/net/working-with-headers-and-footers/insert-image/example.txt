//ExSummary: Tento kód ukazuje, jak vložit obrázek do hlavičky dokumentu Word pomocí Aspose.Words pro .NET.
//ExStepSummary:0: Vytvořit nový dokument Word pomocí třídy Aspose.Words Document class.
//ExStepSummary:1: Přesuňte se do hlavní hlavičky dokumentu pomocí DocumentBuilder.
//ExStepSummary:2: Vložte obrázek do hlavičky se specifickými možnostmi seřízení, rozměrů a balení.
//ExStepSummary:3: Uložit dokument na zadanou cestu k souboru.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Vytvořit nový dokument Word
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Přesunout kurzor do hlavní hlavičky dokumentu
builder.MoveToHeaderFooter(HeaderFooterType.HeaderPrimary);

//ExStep:2-
// Vložte obrázek do hlavičky se specifickým nastavením a rozměry
builder.InsertImage(
    ImagesDir + "Logo.jpg", RelativeHorizontalPosition.RightMargin, 10,
    RelativeVerticalPosition.Page, 10, 50, 50, 
    WrapType.Through
);

//ExStep:3-
// Uložit dokument na zadanou cestu
doc.Save("WorkingWithHeadersAndFooters.InsertImage.docx");
//ExEnd
