//ExSummary:Tento kód ukazuje, jak vytvořit Word dokument pomocí Aspose.Words, formátování vlastnosti písma před napsáním vzor text na něj.
//ExStepSummary:0: Následující kód inicializuje nový dokument a dokumentBuilder.
//ExStepSummary:1: Následující kód nastavuje různé vlastnosti písma, jako je velikost, tučné, barva, název a podtržení stylu.
//ExStepSummary:2: Následující kód zapíše vzorový text do dokumentu pomocí nakonfigurovaného nastavení písma.
//ExStepSummary:3: Následující kód uloží dokument do zadaného adresáře s daným názvem souboru.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Vytvořit novou instanci dokumentu
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Přístup k vlastnosti písma prostřednictvím DocumentBuilder
Font font = builder.Font;
// Nastavte velikost písma na 16
font.Size = 16;
// Udělejte písmo tučné
font.Bold = true;
// Změnit barvu písma na modrou
font.Color = Color.Blue;
// Nastavte písmo na Arial
font.Name = "Arial";
// Na písmo naneste styl ztlumeného podtržení
font.Underline = Underline.Dash;

//ExStep:2-
builder.Write("Sample text.");

//ExStep:3-
// Uložit dokument do adresáře artefaktů
doc.Save("WorkingWithFonts.FontFormatting.docx");
//ExEnd
