//ExSummary: Tento kód ukazuje, jak vytvořit dokument a získat mezeru v řádku zadaného písma.
//ExStepSummary:0: Následující kód inicializuje nový dokument a dokumentBuilder.
//ExStepSummary:1: Kód nastaví písmo na "Calibri" a přidá řádek textu.
//ExStepSummary:2: Kód získá vzdálenost řádků prvního chodu v dokumentu.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Vytvořit novou instanci dokumentu
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Nastavte název písma na 'Calibri'
builder.Font.Name = "Calibri";
builder.Writeln("qText");

//ExStep:2-
// Přístup k písmu prvního spuštění
Font font = builder.Document.FirstSection.Body.FirstParagraph.Runs[0].Font;
// Výstupní rozestup linky do konzole
Console.WriteLine($"lineSpacing = {font.LineSpacing}");
//ExEnd
