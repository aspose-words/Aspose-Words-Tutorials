//ExSummary: This code demonstrates how to insert TC (Table of Content) fields into a document using Aspose.Words. It finds a specific text ("The Beginning") and replaces it with the TC field for chapter numbering.
//ExStepSummary:0: The following code initializes a new Document object.
//ExStepSummary:1: The following code sets up find and replace options with specific highlight and callback.
//ExStepSummary:2: The following code replaces the specified text with TC field.
//ExStepImage:1: images/insert_tc_fields.png
//ExStep:0-
// Create a new Document instance
Document doc = new Document();

//ExStep:1-
// Initialize find and replace options
FindReplaceOptions options = new FindReplaceOptions();
options.ApplyFont.HighlightColor = Color.DarkOrange;
options.ReplacingCallback = new InsertTCFieldHandler("Chapter 1", "\\l 1");

//ExStep:2-
// Replace specified text with TC field
doc.Range.Replace(new Regex("The Beginning"), "", options);
//ExEnd
