//ExSummary: Αυτός ο κώδικας δείχνει πώς να εισάγετε ένα διάγραμμα στήλης σε ένα έγγραφο χρησιμοποιώντας Aspose.Words. Δημιουργεί ένα έγγραφο,
//          προσθέτει ένα διάγραμμα με καθορισμένες κατηγορίες και τιμές, και στη συνέχεια αποθηκεύει το έγγραφο.
//ExStepSummary:0: Ο ακόλουθος κώδικας εισάγει ένα νέο έγγραφο και έναν κατασκευαστή εγγράφων.
//ExStepSummary:1: Το βήμα αυτό περιλαμβάνει την εισαγωγή ενός διαγράμματος στήλης με καθορισμένες διαστάσεις στο έγγραφο.
//ExStepSummary:2: Εδώ, τα δεδομένα σειράς προστίθενται στο διάγραμμα με τις κατηγορίες και τις αντίστοιχες τιμές.
//ExStepSummary:3: Αυτό το βήμα αποθηκεύει το έγγραφο με το εισάγον διάγραμμα σε μια καθορισμένη τοποθεσία αρχείου.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Αρχικοποίηση ενός νέου αντικειμένου εγγράφου
Document doc = new Document();
// Δημιουργία ενός DocumentBuilder για εύκολη χειραγώγηση εγγράφων
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Εισάγετε ένα διάγραμμα στήλης στο καθορισμένο μέγεθος (πλάτος x ύψος)
Shape shape = builder.InsertChart(ChartType.Column, 432, 252);

//ExStep:2-
// Πρόσβαση στο αντικείμενο του χάρτη από το εισάγον σχήμα
Chart chart = shape.Chart;
// Προσθήκη δεδομένων σειράς με κατηγορίες και τιμές
chart.Series.Add("Aspose Series 1", new string[] { "Category 1", "Category 2" }, new double[] { 1, 2 });

//ExStep:3-
// Αποθήκευση του εγγράφου με το διάγραμμα της στήλης
doc.Save("WorkingWithCharts.InsertColumnChart.docx"); 
//ExEnd
