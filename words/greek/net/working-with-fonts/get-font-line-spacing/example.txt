//ExStepSummary:0: Ο ακόλουθος κώδικας αρχικοποιείται ένα νέο έγγραφο και έγγραφοBuilder.
//ExStepSummary:1: Ο κώδικας θέτει τη γραμματοσειρά στο "Καλίμπρι" και προσθέτει μια γραμμή κειμένου.
//ExStepSummary:2: Ο κωδικός ανακτά την απόσταση γραμμής της πρώτης διαδρομής στο έγγραφο.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Δημιουργία μιας νέας περίπτωσης εγγράφου
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Ορισμός του ονόματος γραμματοσειράς στο 'Calibri'
builder.Font.Name = "Calibri";
builder.Writeln("qText");

//ExStep:2-
// Πρόσβαση στη γραμματοσειρά του πρώτου γύρου
Font font = builder.Document.FirstSection.Body.FirstParagraph.Runs[0].Font;
// Έξοδος της απόστασης γραμμής στην κονσόλα
Console.WriteLine($"lineSpacing = {font.LineSpacing}");
