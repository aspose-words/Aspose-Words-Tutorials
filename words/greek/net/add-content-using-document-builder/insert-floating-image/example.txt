//ExStepSummary:0: Ο ακόλουθος κώδικας αρχικοποιείται ένα νέο έγγραφο και έγγραφοBuilder.
//ExStepSummary:1: Ο ακόλουθος κωδικός εισάγει μια εικόνα στο έγγραφο.
//ExStepSummary:2: Ο ακόλουθος κωδικός αποθηκεύει το έγγραφο σε συγκεκριμένη τοποθεσία.
//ExStepImage:1:images/1.png
//ExStep:0-
// Δημιουργία μιας νέας περίπτωσης εγγράφου
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Εισαγωγή εικόνας από την καθορισμένη διαδρομή
builder.InsertImage(ImagesDir + "Transparent background logo.png", 
    RelativeHorizontalPosition.Margin,
    100, 
    RelativeVerticalPosition.Margin, 
    100, 
    200, 
    100, 
    WrapType.Square); 

//ExStep:2-
// Αποθήκευση του εγγράφου στο καθορισμένο αρχείο
doc.Save("AddContentUsingDocumentBuilder.InsertFloatingImage.docx");
//ExEnd
