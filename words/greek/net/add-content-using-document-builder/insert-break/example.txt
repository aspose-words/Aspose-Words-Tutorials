// Δημιουργία μιας νέας περίπτωσης εγγράφου
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Προσθήκη κειμένου για την πρώτη σελίδα
builder.Writeln("This is page 1.");

//ExStep:2-
// Εισάγετε ένα διάλειμμα σελίδας για να ξεκινήσετε μια νέα σελίδα
builder.InsertBreak(BreakType.PageBreak);

//ExStep:3-
// Προσθήκη κειμένου για τη δεύτερη σελίδα
builder.Writeln("This is page 2.");

//ExStep:4-
// Εισάγετε ένα άλλο διάλειμμα σελίδας για μια νέα σελίδα
builder.InsertBreak(BreakType.PageBreak);

//ExStep:5-
// Προσθήκη κειμένου για την τρίτη σελίδα
builder.Writeln("This is page 3.");

//ExStep:6-
// Αποθήκευση του εγγράφου στην καθορισμένη διαδρομή
doc.Save("AddContentUsingDocumentBuilder.InsertBreak.docx");
