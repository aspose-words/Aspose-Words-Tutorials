//ExStepSummary:3: Αυτό το βήμα εισάγει ένα διάλειμμα σελίδας για να ξεκινήσει το πραγματικό περιεχόμενο σε μια νέα σελίδα.
//ExStepSummary:4: Αυτό το βήμα θέτει την πρώτη επικεφαλίδα και γράφει "Κεφάλι 1" στο έγγραφο.
//ExStepSummary:5: Αυτό το βήμα ορίζει το δεύτερο επίπεδο επικεφαλίδας και γράφει "Κεφάλι 1.1" και "Κεφάλι 1.2."
//ExStepSummary:6: Αυτό το βήμα θέτει και πάλι το πρώτο επίπεδο και γράφει "Κεφάλι 2" και "Κεφάλι 3."
//ExStepSummary:7: Αυτό το βήμα ορίζει το δεύτερο επίπεδο επικεφαλίδας και γράφει "Κεφάλι 3.1."
//ExStepSummary:8: Το βήμα αυτό ορίζει το τρίτο επίπεδο κατεύθυνσης και γράφει τις διακρίσεις "Κεφάλι 3.1.1," "Κεφάλι 3.1.2" και "Κεφάλι 3.1.3."
//ExStepSummary:9: Αυτό το βήμα καθορίζει το δεύτερο επίπεδο κατεύθυνσης για άλλη μια φορά και γράφει "Κεφάλι 3.2" και "Κεφάλι 3.3."
//ExStepSummary:10: Τέλος, αυτό το βήμα ενημερώνει τα πεδία του εγγράφου και το αποθηκεύει ως .docx αρχείο.

//ExStep:0-
// Δημιουργία μιας νέας περίπτωσης εγγράφου
Document doc = new Document();

//ExStep:1-
// Αρχικοποίηση κατασκευής εγγράφου για τη χειραγώγηση εγγράφων
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:2-
// Εισάγετε έναν πίνακα περιεχομένων με συγκεκριμένες επιλογές
builder.InsertTableOfContents("\\o \"1-3\" \\h \\z \\u");

//ExStep:3-
// Εισάγετε ένα διάλειμμα σελίδας για να ξεκινήσετε το περιεχόμενο σε μια νέα σελίδα
builder.InsertBreak(BreakType.PageBreak);

//ExStep:4-
// Ορισμός στυλ στον τομέα 1
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading1;
builder.Writeln("Heading 1");

//ExStep:5-
// Ορισμός στυλ στον τομέα 2
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading2;
builder.Writeln("Heading 1.1");
builder.Writeln("Heading 1.2");

//ExStep:6-
// Ορισμός στυλ πίσω στον τομέα 1
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading1;
builder.Writeln("Heading 2");
builder.Writeln("Heading 3");

//ExStep:7-
// Ορισμός στυλ στον τομέα 2
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading2;
builder.Writeln("Heading 3.1");

//ExStep:8-
// Ορισμός στυλ στον τομέα 3
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading3;
builder.Writeln("Heading 3.1.1");
builder.Writeln("Heading 3.1.2");
builder.Writeln("Heading 3.1.3");

//ExStep:9-
// Ορισμός στυλ πίσω στον τομέα 2
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading2;
builder.Writeln("Heading 3.2");
builder.Writeln("Heading 3.3");

//ExStep:10-
// Ενημέρωση όλων των πεδίων του εγγράφου
doc.UpdateFields();
// Αποθήκευση του εγγράφου
doc.Save("AddContentUsingDocumentBuilder.InsertTableOfContents.docx");
