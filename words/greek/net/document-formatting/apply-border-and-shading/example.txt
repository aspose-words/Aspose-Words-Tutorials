//ExSummary: Αυτός ο κώδικας δείχνει πώς να εφαρμόσει τα σύνορα και σκίαση σε μια παράγραφο σε ένα έγγραφο Word χρησιμοποιώντας Aspose.Words.
//ExStepSummary:0: Ο ακόλουθος κώδικας εγκαινιάζει μια νέα περίπτωση Document and DocumentBuilder.
//ExStepSummary:1: Ο ακόλουθος κώδικας διαμορφώνει τα σύνορα της παραγράφου με απόσταση από το κείμενο.
//ExStepSummary:2: Ο ακόλουθος κωδικός εφαρμόζει διαγώνιο σκίαση της υφής στην παράγραφο με συγκεκριμένα χρώματα.
//ExStepSummary:3: Ο ακόλουθος κώδικας γράφει μια σχηματοποιημένη παράγραφο με τις εφαρμοζόμενες ρυθμίσεις.
//ExStepSummary:4: Ο ακόλουθος κώδικας αποθηκεύει το έγγραφο στην καθορισμένη διαδρομή εξόδου.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0-
// Αρχικοποίηση ενός νέου εγγράφου
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Πρόσβαση στα σύνορα της παραγράφου
BorderCollection borders = builder.ParagraphFormat.Borders;
borders.DistanceFromText = 20;
borders[BorderType.Left].LineStyle = LineStyle.Double;
borders[BorderType.Right].LineStyle = LineStyle.Double;
borders[BorderType.Top].LineStyle = LineStyle.Double;
borders[BorderType.Bottom].LineStyle = LineStyle.Double;

//ExStep:2-
// Πρόσβαση στις ιδιότητες σκίασης παραγράφου
Shading shading = builder.ParagraphFormat.Shading;
shading.Texture = TextureIndex.TextureDiagonalCross;
shading.BackgroundPatternColor = System.Drawing.Color.LightCoral; 
shading.ForegroundPatternColor = System.Drawing.Color.LightSalmon;

//ExStep:3-
// Γράψτε το μορφοποιημένο περιεχόμενο παραγράφου
builder.Write("I'm a formatted paragraph with double border and nice shading.");

//ExStep:4-
// Αποθήκευση του εγγράφου στην καθορισμένη διαδρομή
doc.Save(ArtifactsDir + "DocumentFormatting.ApplyBordersAndShadingToParagraph.doc");
//ExEnd
