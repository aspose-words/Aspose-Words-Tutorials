//ExStepSummary:0: Αυτό το βήμα εγκαινιάζει ένα νέο έγγραφο και τον κατασκευαστή εγγράφων.
//ExStepSummary:1: Αυτό το βήμα ξεκινάει το τραπέζι και εισάγει ένα κελί σε αυτό.
//ExStepSummary:2: Αυτό το βήμα καθορίζει τη διαμόρφωση για τη σειρά, συμπεριλαμβανομένου του κανόνα ύψους και ύψους.
//ExStepSummary:3: Αυτό το βήμα θέτει padding για ολόκληρο τον πίνακα, επηρεάζοντας όλες τις σειρές.
//ExStepSummary:4: Αυτό το βήμα προσθέτει κείμενο στο κελί και τελειώνει τη σειρά.
//ExStepSummary:5: Αυτό το βήμα τελειώνει τον πίνακα και αποθηκεύει το έγγραφο σε μια συγκεκριμένη διαδρομή.

//ExStep:0-
// Δημιουργία μιας νέας περίπτωσης εγγράφου
Document doc = new Document();
// Αρχικοποίηση κατασκευής εγγράφου με το έγγραφο
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Ξεκινήστε ένα νέο τραπέζι
Table table = builder.StartTable();
// Εισάγετε το πρώτο κελί στον πίνακα
builder.InsertCell();

//ExStep:2-
// Πρόσβαση στην ιδιοκτησία RowFormat του κατασκευαστή
RowFormat rowFormat = builder.RowFormat;
// Ορισμός του ύψους της γραμμής
rowFormat.Height = 100;
// Ορισμός κανόνα ύψους για την ακρίβεια
rowFormat.HeightRule = HeightRule.Exactly;

//ExStep:3-
// Ρυθμίστε την επένδυση για το τραπέζι
table.LeftPadding = 30;
table.RightPadding = 30;
table.TopPadding = 30;
table.BottomPadding = 30;

//ExStep:4-
// Προσθήκη κειμένου στο τρέχον κύτταρο
builder.Writeln("I'm a wonderful formatted row.");
// Τέλος της τρέχουσας γραμμής
builder.EndRow();
builder.EndTable();

//ExStep:5-
// Αποθήκευση του εγγράφου στην καθορισμένη διαδρομή
doc.Save("WorkingWithTableStylesAndFormatting.DocumentBuilderSetTableRowFormatting.docx");
//ExEnd
