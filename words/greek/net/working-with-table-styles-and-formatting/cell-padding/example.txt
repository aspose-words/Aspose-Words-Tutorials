//ExSummary: Ο παρακάτω κώδικας δείχνει πώς να δημιουργήσετε ένα έγγραφο Word με ένα σχηματοποιημένο κελί πίνακα που περιλαμβάνει padding γύρω από το περιεχόμενό του.
//ExStepSummary:0: Ο ακόλουθος κώδικας εισάγει ένα νέο έγγραφο και έναν κατασκευαστή εγγράφων.
//ExStepSummary:1: Ο ακόλουθος κωδικός αρχίζει έναν πίνακα στο έγγραφο.
//ExStepSummary:2: Ο ακόλουθος κωδικός εισάγει ένα κελί στον πίνακα.
//ExStepSummary:3: Ο ακόλουθος κωδικός καθορίζει την επένδυση για το περιεχόμενο του κυττάρου.
//ExStepSummary:4: Ο παρακάτω κωδικός προσθέτει κείμενο στο κελί και ολοκληρώνει την τρέχουσα σειρά.
//ExStepSummary:5: Ο παρακάτω κωδικός τελειώνει τον πίνακα και αποθηκεύει το έγγραφο σε ένα αρχείο.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0-
// Αρχικοποίηση ενός νέου εγγράφου
Document doc = new Document();
// Δημιουργία ενός κατασκευαστή εγγράφων για τη χειραγώγηση του εγγράφου
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Ξεκινήστε τη δημιουργία ενός πίνακα
builder.StartTable();
// Εισάγετε ένα νέο κελί στον πίνακα
builder.InsertCell();

//ExStep:2-
// Ρυθμίστε την επένδυση με 30 πόντους πάνω και κάτω, 50 πόντους αριστερά και δεξιά
builder.CellFormat.SetPaddings(30, 50, 30, 50);

//ExStep:3-
builder.Writeln("I'm a wonderful formatted cell.");
// Ολοκληρώστε την τρέχουσα σειρά στον πίνακα
builder.EndRow();
// Τελείωσε το τραπέζι.
builder.EndTable();

//ExStep:4-
// Αποθήκευση του εγγράφου σε ένα συγκεκριμένο αρχείο
doc.Save("WorkingWithTableStylesAndFormatting.CellPadding.docx");
//ExEnd
