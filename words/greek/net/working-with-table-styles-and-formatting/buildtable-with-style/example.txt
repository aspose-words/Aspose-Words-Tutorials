//ExStepSummary:0: Αρχικοποιήστε ένα νέο Document and DocumentBuilder για τη διαχείριση της δομής του εγγράφου.
//ExStepSummary:1: Ξεκινήστε έναν νέο πίνακα στο έγγραφο χρησιμοποιώντας το DocumentBuilder.
//ExStepSummary:2: Εισάγετε το πρώτο κελί και ρυθμίστε το αναγνωριστικό στυλ πίνακα.
//ExStepSummary:3: Εφαρμόστε τις επιλογές μορφοποίησης στον πίνακα και ορίστε αυτόματη συμπεριφορά.
//ExStepSummary:4: Προσθήκη κελιών γραμμής κεφαλίδας στον πίνακα.
//ExStepSummary:5: Εισάγετε σειρές δεδομένων στον πίνακα με τα ονόματα και τις ποσότητες των στοιχείων.
//ExStepSummary:6: Αποθηκεύστε το συναρμολογημένο έγγραφο σε ένα αρχείο.

//ExStep:0-
// Δημιουργία ενός νέου αντικειμένου εγγράφου.
Document doc = new Document();
// Δημιουργία ενός κατασκευαστή για τη διευκόλυνση της παραγωγής εγγράφων.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Ξεκινήστε ένα νέο τραπέζι στο έγγραφο.
Table table = builder.StartTable();

//ExStep:2-
// Εισάγετε το πρώτο κύτταρο πριν από τη διαμόρφωση.
builder.InsertCell();
// Ορίστε το αναγνωριστικό στυλ για τον πίνακα.
table.StyleIdentifier = StyleIdentifier.MediumShading1Accent1;


//ExStep:3-
// Ρύθμιση επιλογών στυλ.
table.StyleOptions = 
    TableStyleOptions.FirstColumn | TableStyleOptions.RowBands | TableStyleOptions.FirstRow;
// Ορίστε την αυτόματη συμπεριφορά του τραπεζιού.	
table.AutoFit(AutoFitBehavior.AutoFitToContents);


//ExStep:4-
// Γράψε την κεφαλίδα για την πρώτη στήλη.
builder.Writeln("Item");
// Βάλτε δεξιά επένδυση για το κελί.
builder.CellFormat.RightPadding = 40;
// Εισάγετε το δεύτερο κελί για την επόμενη στήλη.
builder.InsertCell();
// Γράψε την κεφαλίδα για τη δεύτερη στήλη.
builder.Writeln("Quantity (kg)");
// Τελείωσε τη δημιουργία της σειράς.
builder.EndRow();

//ExStep:5-
// Ξεκινήστε μια νέα σειρά και εισάγετε το πρώτο κελί.
builder.InsertCell();
builder.Writeln("Apples");
builder.InsertCell();
builder.Writeln("20");
builder.EndRow();
 // Εισάγετε το πρώτο κελί μιας νέας σειράς για το δεύτερο αντικείμενο.
builder.InsertCell();
builder.Writeln("Bananas");
builder.InsertCell();
builder.Writeln("40");
builder.EndRow();
// Ξεκινήστε μια νέα σειρά για το τρίτο αντικείμενο.
builder.InsertCell();
builder.Writeln("Carrots");
builder.InsertCell(); 
builder.Writeln("50");
builder.EndRow();

//ExStep:6-
// Αποθήκευση του εγγράφου σε ένα συγκεκριμένο αρχείο.
doc.Save("WorkingWithTableStylesAndFormatting.BuildTableWithStyle.docx");
