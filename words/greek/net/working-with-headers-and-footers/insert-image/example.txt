//ExStepSummary:3: Αποθήκευση του εγγράφου σε μια καθορισμένη διαδρομή αρχείου.

//ExStepImage:2:images/2.png
//ExStep:0-
// Δημιουργία ενός νέου εγγράφου Word
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Μετακίνηση του δρομέα στην κύρια κεφαλίδα του εγγράφου
builder.MoveToHeaderFooter(HeaderFooterType.HeaderPrimary);

//ExStep:2-
// Εισάγετε μια εικόνα στην κεφαλίδα με συγκεκριμένη ευθυγράμμιση και διαστάσεις
builder.InsertImage(
    ImagesDir + "Logo.jpg", RelativeHorizontalPosition.RightMargin, 10,
    RelativeVerticalPosition.Page, 10, 50, 50, 
    WrapType.Through
);

//ExStep:3-
// Αποθήκευση του εγγράφου στην καθορισμένη διαδρομή
doc.Save("WorkingWithHeadersAndFooters.InsertImage.docx");
