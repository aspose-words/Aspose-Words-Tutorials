//ExSummary: Ez a kód bemutatja, hogyan kell alkalmazni a határokat és az árnyékolást egy Word dokumentum egy bekezdésére az Aspose.Words használatával.
//ExStepSummary:0: A következő kód egy új Document and DocumentBuilder példányt inicializál.
//ExStepSummary:1: A következő kód a szövegtől való távolsággal állítja be a bekezdéshatárokat.
//ExStepSummary:2: A következő kód egy átlós kereszt textúrájú árnyékolást alkalmaz a megadott színekkel ellátott bekezdésre.
//ExStepSummary:3: A következő kód formázott bekezdést ír az alkalmazott beállításokkal.
//ExStepSummary:4: A következő kód menti a dokumentumot a megadott kimeneti útvonalra.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0-
// Új dokumentum inicializálása
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// A bekezdés határainak elérése
BorderCollection borders = builder.ParagraphFormat.Borders;
borders.DistanceFromText = 20;
borders[BorderType.Left].LineStyle = LineStyle.Double;
borders[BorderType.Right].LineStyle = LineStyle.Double;
borders[BorderType.Top].LineStyle = LineStyle.Double;
borders[BorderType.Bottom].LineStyle = LineStyle.Double;

//ExStep:2-
// A bekezdés árnyékoló tulajdonságainak elérése
Shading shading = builder.ParagraphFormat.Shading;
shading.Texture = TextureIndex.TextureDiagonalCross;
shading.BackgroundPatternColor = System.Drawing.Color.LightCoral; 
shading.ForegroundPatternColor = System.Drawing.Color.LightSalmon;

//ExStep:3-
// Írja be a formázott bekezdés tartalmát
builder.Write("I'm a formatted paragraph with double border and nice shading.");

//ExStep:4-
// A dokumentum mentése a megadott elérési útra
doc.Save(ArtifactsDir + "DocumentFormatting.ApplyBordersAndShadingToParagraph.doc");
//ExEnd
