//ExSummary: Ez a kód bemutatja, hogyan formázhatja a bekezdéseket egy Word dokumentumban az Aspose.Words használatával. Megmutatja, hogyan állíthatja be a bekezdés formázási beállításait, például az igazítást, a behúzást és a szóközt egy bekezdés után.
//ExStepSummary:0: Ez a lépés egy új dokumentumot és egy DocumentBuilder-t inicializál a dokumentum létrehozásához.
//ExStepSummary:1: Ez a lépés beállítja a bekezdés formázási beállításait, beleértve az igazítást, a bal oldali behúzást, a jobb oldali behúzást és a bekezdés utáni szóközt.
//ExStepSummary:2: Ez a lépés egy jól formázott bekezdést ír a megadott formázási beállításokkal a dokumentumhoz.
//ExStepSummary:3: Ez a lépés egy másik formázott bekezdést ír, amely bemutatja a bekezdés utáni tér hatását.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0- 
// Új dokumentumpéldány létrehozása.
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1- 
// Állítsa be a bekezdésformázási beállításokat.
ParagraphFormat paragraphFormat = builder.ParagraphFormat;
paragraphFormat.Alignment = ParagraphAlignment.Center;
paragraphFormat.LeftIndent = 50;
paragraphFormat.RightIndent = 50;
paragraphFormat.SpaceAfter = 25;

//ExStep:2- 
// Írja be az első formázott bekezdést.
builder.Writeln(
	"I'm a very nice formatted paragraph. I'm intended to demonstrate how the left and right indents affect word wrapping.");

//ExStep:3- 
// Írja be a második formázott bekezdést.
builder.Writeln(
	"I'm another nice formatted paragraph. I'm intended to demonstrate how the space after paragraph looks like.");

//ExStep:4- 
// Mentse a dokumentumot a megadott útvonalra.
doc.Save(ArtifactsDir + "DocumentFormatting.ParagraphFormatting.docx");
//ExEnd
