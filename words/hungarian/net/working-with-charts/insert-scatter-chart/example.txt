//ExSummary:Ez a kód bemutatja, hogyan lehet létrehozni egy Word dokumentumot egy scatter diagrammal az Aspose.Words segítségével. Ez inicializálja a dokumentumot, építi a tartalmat, beilleszti a scatter diagramot, hozzáadja az adatokat, és menti a dokumentumot.
//ExStepSummary:0: A dokumentum és a DocumentBuilder objektumok inicializálása a dokumentum működésének megkezdéséhez.
//ExStepSummary:1: A scatter chart beillesztése a dokumentumba.
//ExStepSummary:2: Adatpontok sorozatának hozzáadása a scatter charthoz.
//ExStepSummary:3: A dokumentum mentése a scatter diagrammal egy .docx fájlba.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Új dokumentum inicializálása
Document doc = new Document();
// Hozzon létre egy építőt a tartalom beillesztéséhez
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Szórási diagram beszúrása a dokumentumba
Shape shape = builder.InsertChart(ChartType.Scatter, 432, 252); 

//ExStep:2-
// A diagramobjektum beszúrása a beszúrt alakzatból
Chart chart = shape.Chart;
// Sorozatadatok hozzáadása a diagramhoz
chart.Series.Add("Aspose Series 1", new double[] { 0.7, 1.8, 2.6 }, new double[] { 2.7, 3.2, 0.8 });

//ExStep:3-
// A dokumentum mentése a scatter chart segítségével
doc.Save("WorkingWithCharts.InsertScatterChart.docx");
//ExEnd
