//ExStepSummary:5: Ez a kód törli a diagrammal létrehozott alapértelmezett sorozatokat.
//ExStepSummary:6: Ez a kód azt mutatja be, hogyan lehet kategórianeveket definiálni a diagram adataihoz.

//ExStep:0-
// Hozzon létre egy új Word dokumentumpéldányt.
Document doc = new Document();

//ExStep:1-
// Hozzon létre egy DocumentBuilder-t, hogy könnyen beilleszthesse az elemeket a dokumentumba.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:2-
// A megadott szélességű (432) és magasságú (252) oszlopdiagram-alakzat beszúrása.
Shape shape = builder.InsertChart(ChartType.Column, 432, 252);

//ExStep:3-
// Töltse le a diagram objektumot az alakzatból a további manipulációhoz.
Chart chart = shape.Chart;

//ExStep:4-
// Hozzáférés a diagram sorozatgyűjteményéhez.
ChartSeriesCollection seriesColl = chart.Series;
// Nyomtassa ki a diagramban lévő sorozatok számát a konzolra.
Console.WriteLine(seriesColl.Count);

//ExStep:5-
// Távolítson el minden olyan alapértelmezett sorozatot, amely automatikusan generálódott a diagram létrehozásakor.
seriesColl.Clear();

//ExStep:6-
// Adja meg a diagramadatok kategórianeveit.
string[] categories = new string[] { "Category 1", "Category 2" };

//ExStep:7-
// Több sorozat hozzáadása a diagramhoz kategórianevekkel és adatértékekkel.
seriesColl.Add("Aspose Series 1", categories, new double[] { 1, 2 });
seriesColl.Add("Aspose Series 2", categories, new double[] { 3, 4 });
seriesColl.Add("Aspose Series 3", categories, new double[] { 5, 6 });
seriesColl.Add("Aspose Series 4", categories, new double[] { 7, 8 });
seriesColl.Add("Aspose Series 5", categories, new double[] { 9, 10 });

//ExStep:8-
// Mentse a dokumentumot a beillesztett diagrammal a megadott fájl elérési útjára.
doc.Save("WorkingWithCharts.InsertSimpleColumnChart.docx");
