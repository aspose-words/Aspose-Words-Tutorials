A táblázatot az oldalra helyezi, zöld szegélyt állít be az asztal körül, törli a meglévő határokat, és kitölti a táblázatcellákat világos zöld háttérszínnel. 
//ExStepSummary:0: A következő kód inicializálja a dokumentumot, és beolvassa az első táblát.
//ExStepSummary:1: Ez a lépés a táblázatot az oldal közepére igazítja.
//ExStepSummary:2: Ez a lépés törli a meglévő határokat az asztalról.
//ExStepSummary:3: Ez a lépés zöld határt állít fel az asztal körül, de nem belül.
//ExStepSummary:4: Ez a lépés világoszöld, szilárd színnel tölti ki a sejteket.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStepImage:4:images/4.png
//ExStep:0-
// A dokumentum könyvtárának megadása
string dataDir = "Your Document Directory";
// A dokumentum betöltése
Document doc = new Document(dataDir + "Tables.docx");

//ExStep:1-
// Az első táblázat letöltése a dokumentumból
Table table = (Table)doc.GetChild(NodeType.Table, 0, true);
// A táblázat igazítása az oldal közepére
table.Alignment = TableAlignment.Center;

//ExStep:2-
// Törölje a meglévő határokat az asztalról
table.ClearBorders();

//ExStep:3-
// Bal oldali szegély beállítása
table.SetBorder(BorderType.Left, LineStyle.Single, 1.5, Color.Green, true);
// Jobb oldali szegély beállítása
table.SetBorder(BorderType.Right, LineStyle.Single, 1.5, Color.Green, true);
// Felső határ beállítása
table.SetBorder(BorderType.Top, LineStyle.Single, 1.5, Color.Green, true);
// Alsó szegély beállítása
table.SetBorder(BorderType.Bottom, LineStyle.Single, 1.5, Color.Green, true);

//ExStep:4-
// Töltse fel a sejteket világos zöld színnel
table.SetShading(TextureIndex.TextureSolid, Color.LightGreen, Color.Empty);
// A módosított dokumentum mentése
doc.Save(dataDir + "WorkingWithTableStylesAndFormatting.ApplyOutlineBorder.docx");
//ExEnd
