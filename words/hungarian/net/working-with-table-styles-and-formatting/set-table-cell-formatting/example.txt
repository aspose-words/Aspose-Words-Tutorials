//ExStepSummary:0: Új dokumentum és DocumentBuilder inicializálása a Word-dokumentum felépítésének megkezdéséhez.
//ExStepSummary:1: Hozzon létre egy táblázatot, és helyezzen be egy cellát a formázáshoz.
//ExStepSummary:2: Állítsa be a cella formázási tulajdonságait, beleértve a párnázást és a szélességet.
//ExStepSummary:3: Helyezzen be szöveget a formázott cellába.
//ExStepSummary:4: Zárja be a sort és a táblázatot, és mentse a dokumentumot egy megadott fájl elérési útjára.

//ExStepImage:3:images/3.png
//ExStep:0-
// Új dokumentumobjektum létrehozása
Document doc = new Document();
// A DocumentBuilder inicializálása az új dokumentummal
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Hozzon létre egy táblázatot
builder.StartTable();
// Az első cella beszúrása a táblázatba
builder.InsertCell();

//ExStep:2-
// A cella formázási tulajdonságainak elérése
CellFormat cellFormat = builder.CellFormat;
// A cella szélességének beállítása
cellFormat.Width = 250;
// Állítsa be a párnázást
cellFormat.LeftPadding = 30;
cellFormat.RightPadding = 30;
cellFormat.TopPadding = 30;
cellFormat.BottomPadding = 30;

//ExStep:3-
// Szöveg beillesztése a cellába
builder.Writeln("I'm a wonderful formatted cell.");

//ExStep:4-
// A táblázat aktuális sorának befejezése
builder.EndRow();
// Az asztal létrehozásának befejezése
builder.EndTable();
// A dokumentum mentése
doc.Save("WorkingWithTableStylesAndFormatting.DocumentBuilderSetTableCellFormatting.docx");
//ExEnd
