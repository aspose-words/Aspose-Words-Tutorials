//ExStepSummary:1: A következő kód elindít egy táblázatot a dokumentumon belül.
//ExStepSummary:2: A következő kód beszúr egy cellát a táblázatba.
//ExStepSummary:3: A következő kód határozza meg a cella tartalmának kitöltését.
//ExStepSummary:4: A következő kód szöveget ad a cellához, és befejezi az aktuális sort.
//ExStepSummary:5: A következő kód befejezi a táblázatot, és a dokumentumot fájlba menti.

//ExStep:0-
// Új dokumentum inicializálása
Document doc = new Document();
// Dokumentumszerkesztő létrehozása a dokumentum manipulálásához
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Hozzon létre egy táblázatot
builder.StartTable();
// Új cella beszúrása a táblázatba
builder.InsertCell();

//ExStep:2-
// Állítsa be a párnázást 30 ponttal a tetején és az alján, 50 ponttal a bal és a jobb oldalon
builder.CellFormat.SetPaddings(30, 50, 30, 50);

//ExStep:3-
builder.Writeln("I'm a wonderful formatted cell.");
// Az aktuális sor befejezése a táblázatban
builder.EndRow();
// Fejezd be az asztalt
builder.EndTable();

//ExStep:4-
// A dokumentum mentése egy megadott fájlba
doc.Save("WorkingWithTableStylesAndFormatting.CellPadding.docx");
//ExEnd
