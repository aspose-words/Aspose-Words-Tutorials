//ExStep:0-
// A dokumentum könyvtárának megadása
string dataDir = "Your Document Directory";
// A dokumentum betöltése
Document doc = new Document(dataDir + "Tables.docx");

//ExStep:1-
// Szerezd meg az első táblázatot a dokumentumból
Table table = (Table) doc.GetChild(NodeType.Table, 0, true);

//ExStep:2-
// Törölje a meglévő határokat az asztalról
table.ClearBorders();

//ExStep:3-
// Állítson be egy zöld szegélyt az asztal körül és azon belül
table.SetBorders(LineStyle.Single, 1.5, Color.Green);
// A dokumentum mentése módosításokkal
doc.Save(dataDir + "WorkingWithTableStylesAndFormatting.BuildTableWithBorders.docx");
//ExEnd
