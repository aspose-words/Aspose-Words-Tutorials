//ExStepSummary:1:Ez a lépés lekéri az első táblázatot a dokumentumból, és beállítja annak címét és leírását.

//ExStepSummary:3:Ez a lépés optimalizálja a Word 2016 dokumentum kompatibilitási beállításait.

//ExStep:0-
// Dokumentum inicializálása és betöltése az adatkönyvtárból
string dataDir = "Your Document Directory";
Document doc = new Document(dataDir + "Tables.docx");

//ExStep:1-
// Töltse le az első táblázatot, és állítsa be a címet és a leírást
Table table = (Table)doc.GetChild(NodeType.Table, 0, true);
table.Title = "Test title";
table.Description = "Test description";

//ExStep:2-
// A dokumentum mentési beállításainak konfigurálása
OoxmlSaveOptions options = new OoxmlSaveOptions { Compliance = OoxmlCompliance.Iso29500_2008_Strict };

//ExStep:3-
// A Word 2016 kompatibilitási lehetőségeinek optimalizálása
doc.CompatibilityOptions.OptimizeFor(Aspose.Words.Settings.MsWordVersion.Word2016);

//ExStep:4-
// A módosított dokumentum mentése a megadott könyvtárba
doc.Save(dataDir + "WorkingWithTableStylesAndFormatting.TableTitleAndDescription.docx", options);
//ExEnd
