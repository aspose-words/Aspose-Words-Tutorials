//ExStepSummary:2: Helyezzen be egy vízszintes szabályalakot a dokumentumba.

//ExStep:0-
// Új dokumentumobjektum létrehozása a tartalom tárolásához
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc); 

//ExStep:1-
// Írjon egy sort a dokumentumba
builder.Writeln("Insert a horizontal rule shape into the document.");

//ExStep:2-
// Vízszintes szabályalak beszúrása a dokumentumba
builder.InsertHorizontalRule();

//ExStep:3-
// A dokumentum mentése a megadott helyre
doc.Save("AddContentUsingDocumentBuilder.InsertHorizontalRule.docx");
