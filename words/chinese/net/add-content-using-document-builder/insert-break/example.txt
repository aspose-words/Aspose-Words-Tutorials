//ExStepSummary:0: 此步骤初始化了一个新的文档和一个用于操作文档的文档构建器 。
//ExStepSummary:1: 此步骤将文字添加到首页 。
//ExStepSummary:2: 此步骤插入一个分页符以启动新页面 。
//ExStepSummary:3: 此步骤将文字添加到第二页。
//ExStepSummary:4: 此步骤为新页面插入另一个分页符。
//ExStepSummary:5: 此步骤将文字添加到第三页。
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStepImage:4:images/4.png
//ExStepImage:5:images/5.png

//ExStep:0-
// 创建新文档实例
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// 添加首页文本
builder.Writeln("This is page 1.");

//ExStep:2-
// 插入分页符以开始新页面
builder.InsertBreak(BreakType.PageBreak);

//ExStep:3-
// 添加第二页的案文
builder.Writeln("This is page 2.");

//ExStep:4-
// 为新页面插入另一个分页符
builder.InsertBreak(BreakType.PageBreak);

//ExStep:5-
// 添加第三页的文字
builder.Writeln("This is page 3.");

//ExStep:6-
// 将文档保存到指定的路径
doc.Save("AddContentUsingDocumentBuilder.InsertBreak.docx");
//ExEnd
