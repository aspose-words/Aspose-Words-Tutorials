//ExStart
//ExSummary:本代码演示如何使用 Aspose. words 将内嵌图像插入 Word 文档。
//ExStepSummary:0: 文件对象的初始化和建造文件的建筑。
//ExStepSummary:1: 将图像插入到文档构建器当前光标位置的文档中。
//ExStepSummary:2: 将文档保存到指定的文件名 。
//ExStepImage:1:images/1.png
//ExStep:0-
// 初始化新的文档和文档构建器
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// 使用文档建筑插入图像
builder.InsertImage(ImagesDir + "Transparent background logo.png");

//ExStep:2-
// 将修改后的文档保存到文件
doc.Save("AddContentUsingDocumentBuilder.InsertInlineImage.docx");
//ExEnd
