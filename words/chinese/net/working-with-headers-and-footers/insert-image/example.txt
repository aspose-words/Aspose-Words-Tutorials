//ExStart
//ExSummary:本代码演示如何使用 apose.Words for.NET在 Word 文档页眉中插入图像。
//ExStepSummary:0: 使用 Apose. words 文档类创建新的 Word 文档 。
//ExStepSummary:1: 使用“文档构建器”移动到文档主页眉。
//ExStepSummary:2: 在页眉中插入图像, 包含特定的对齐、 尺寸和包包选项 。
//ExStepSummary:3: 将文档保存到指定的文件路径 。
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// 创建新 Word 文档
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// 将光标移动到文档的主标题
builder.MoveToHeaderFooter(HeaderFooterType.HeaderPrimary);

//ExStep:2-
// 将图像插入带有特定对齐和尺寸的页眉中的图像
builder.InsertImage(
    ImagesDir + "Logo.jpg", RelativeHorizontalPosition.RightMargin, 10,
    RelativeVerticalPosition.Page, 10, 50, 50, 
    WrapType.Through
);

//ExStep:3-
// 将文档保存到指定的路径
doc.Save("WorkingWithHeadersAndFooters.InsertImage.docx");
//ExEnd
