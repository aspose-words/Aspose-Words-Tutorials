//ExStart
//ExSummary:本代码演示如何使用 Aspose. words 创建单词文档 。
//ExStepSummary:0: 以下代码初始化了一个新的文档和一个文档构建器 。
//ExStepSummary:1: 以下代码设置各种字体属性,如大小、粗体、颜色、名称和下划线样式。
//ExStepSummary:2: 以下代码使用已配置的字体设置,将样本文本写入文档。
//ExStepSummary:3: 以下代码将文档保存到指定的目录中, 并使用给定的文件名 。
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// 创建新文档实例
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// 通过文档构建访问字体属性
Font font = builder.Font;
// 将字体大小设为 16
font.Size = 16;
// 使字体以粗体显示
font.Bold = true;
// 将字体颜色更改为蓝色
font.Color = Color.Blue;
// 将字体设为 Arial 字体
font.Name = "Arial";
// 对字体应用破折的下划线样式
font.Underline = Underline.Dash;

//ExStep:2-
builder.Write("Sample text.");

//ExStep:3-
// 将文档保存到文物目录
doc.Save("WorkingWithFonts.FontFormatting.docx");
//ExEnd
