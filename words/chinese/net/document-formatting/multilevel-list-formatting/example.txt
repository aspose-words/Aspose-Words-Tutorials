//ExSummary:此代码演示如何使用 Aspose.Words 在 Word 文档中创建多级别列表。它初始化文档,应用编号格式,创建主项和子项,最后将文档保存为.docx 文件。
//ExStepSummary:0: 以下代码初始化了一个新的文档和文档构建实例 。
//ExStepSummary:1: 以下代码对列表应用默认编号样式,并添加顶级项目。
//ExStepSummary:2: 以下代码缩进用于在第二个顶级项目下创建分项目。
//ExStepSummary:3: 以下代码进一步缩进第二个分项目下的分项目。
//ExStepSummary:4: 以下代码从上一级缩回到上一级,在二级添加另一个项目。
//ExStepSummary:5: 下面的代码会向上缩进到顶层,以便在列表中添加最后的项目 。
//ExStepSummary:6: 以下代码在保存前从列表中删除编号 。
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStepImage:4:images/4.png
//ExStepImage:5:images/5.png
//ExStepImage:6:images/6.png
//ExStep:0-
// 初始化新文档
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// 应用默认编号格式
builder.ListFormat.ApplyNumberDefault();
builder.Writeln("Item 1");
builder.Writeln("Item 2");

//ExStep:2-
// 分项目的记分项目
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.1");
builder.Writeln("Item 2.2");

//ExStep:3-
// 分分项目的记分项目
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.2.1");
builder.Writeln("Item 2.2.2");

//ExStep:4-
// 后退到第二层
builder.ListFormat.ListOutdent();
builder.Writeln("Item 2.3");

//ExStep:5-
// 后退到顶层
builder.ListFormat.ListOutdent();
builder.Writeln("Item 3");

//ExStep:6-
// 从列表中删除编号
builder.ListFormat.RemoveNumbers();

//ExStep:7-
// 保存文档
doc.Save("DocumentFormatting.MultilevelListFormatting.docx");
//ExEnd
