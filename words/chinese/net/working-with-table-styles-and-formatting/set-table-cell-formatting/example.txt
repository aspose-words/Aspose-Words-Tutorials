//ExSummary:以下代码演示如何使用 Aspose.Word 中的文档构建器类创建和格式化单词文档中的表格。
//ExStepSummary:0: 初始化一个新的文档和文档构建器, 以开始构建 Word 文档 。
//ExStepSummary:1: 创建表格并插入用于格式化的单元格。
//ExStepSummary:2: 设置单元格的格式属性,包括草坪和宽度。
//ExStepSummary:3: 在格式化单元格中插入文字。
//ExStepSummary:4: 结束行和表格,并将文档保存到指定的文件路径。
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0-
// 创建新的文档对象
Document doc = new Document();
// 正在初始化与新文档相连接的文档构建器
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// 开始创建表格
builder.StartTable();
// 在表格中插入第一个单元格
builder.InsertCell();

//ExStep:2-
// 访问单元格格式化属性
CellFormat cellFormat = builder.CellFormat;
// 设置单元格宽度
cellFormat.Width = 250;
// 设置折叠
cellFormat.LeftPadding = 30;
cellFormat.RightPadding = 30;
cellFormat.TopPadding = 30;
cellFormat.BottomPadding = 30;

//ExStep:3-
// 将文字插入单元格
builder.Writeln("I'm a wonderful formatted cell.");

//ExStep:4-
// 结束表格当前行
builder.EndRow();
// 结束表格创建
builder.EndTable();
// 保存文档
doc.Save("WorkingWithTableStylesAndFormatting.DocumentBuilderSetTableCellFormatting.docx");
//ExEnd
