//ExSummary:Este código muestra cómo crear un documento de Word con un gráfico de dispersión usando Aspose.Words. Inicializa un documento, construye contenido, inserta un gráfico de dispersión, agrega datos y guarda el documento.
//ExStepSummary:0: Inicialización de los objetos Document y DocumentBuilder para comenzar a trabajar en el documento.
//ExStepSummary:1: Inserción de un gráfico de dispersión en el documento.
//ExStepSummary:2: Añadir una serie de puntos de datos al gráfico de dispersión.
//ExStepSummary:3: Guardar el documento con el gráfico de dispersión en un archivo .docx.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Inicializa un nuevo documento
Document doc = new Document();
// Crear un constructor para insertar contenido
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Insertar un gráfico de dispersión en el documento
Shape shape = builder.InsertChart(ChartType.Scatter, 432, 252); 

//ExStep:2-
// Obtener el objeto gráfico de la forma insertada
Chart chart = shape.Chart;
// Añadir datos de series al gráfico
chart.Series.Add("Aspose Series 1", new double[] { 0.7, 1.8, 2.6 }, new double[] { 2.7, 3.2, 0.8 });

//ExStep:3-
// Guardar el documento con el gráfico de dispersión
doc.Save("WorkingWithCharts.InsertScatterChart.docx");
//ExEnd
