// Crear una nueva instancia de documento de Word.
Document doc = new Document();

//ExStep:1-
// Cree un DocumentBuilder para insertar fácilmente elementos en el documento.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:2-
// Inserte una forma de gráfico de columna con el ancho (432) y la altura (252) especificados.
Shape shape = builder.InsertChart(ChartType.Column, 432, 252);

//ExStep:3-
// Recupera el objeto gráfico de la forma para su posterior manipulación.
Chart chart = shape.Chart;

//ExStep:4-
// Acceda a la colección de series del gráfico.
ChartSeriesCollection seriesColl = chart.Series;
// Imprima el número de series en el gráfico en la consola.
Console.WriteLine(seriesColl.Count);

//ExStep:5-
// Elimine las series predeterminadas que se generaron automáticamente cuando se creó el gráfico.
seriesColl.Clear();

//ExStep:6-
// Defina una matriz de nombres de categorías para los datos de gráficos.
string[] categories = new string[] { "Category 1", "Category 2" };

//ExStep:7-
// Agregue varias series al gráfico con los nombres de las categorías y los valores de los datos.
seriesColl.Add("Aspose Series 1", categories, new double[] { 1, 2 });
seriesColl.Add("Aspose Series 2", categories, new double[] { 3, 4 });
seriesColl.Add("Aspose Series 3", categories, new double[] { 5, 6 });
seriesColl.Add("Aspose Series 4", categories, new double[] { 7, 8 });
seriesColl.Add("Aspose Series 5", categories, new double[] { 9, 10 });

//ExStep:8-
// Guarde el documento con el gráfico insertado en la ruta de archivo especificada.
doc.Save("WorkingWithCharts.InsertSimpleColumnChart.docx");
