//ExStart
//ExSummary: Este código muestra cómo insertar un gráfico de área en un documento de Word usando Aspose.Words.
//ExStepSummary:0: El siguiente código inicializa un nuevo documento y un generador de documentos.
//ExStepSummary:1: El siguiente código inserta un gráfico de área en el documento.
//ExStepSummary:2: El siguiente código agrega una serie al gráfico con puntos de datos especificados.
//ExStepSummary:3: El siguiente código guarda el documento con el gráfico de área insertado.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Crear un nuevo documento
Document doc = new Document();
// Inicializa el generador de documentos
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Insertar un gráfico de área con las dimensiones especificadas
Shape shape = builder.InsertChart(ChartType.Area, 432, 252);

//ExStep:2-
// Acceder al objeto chart
Chart chart = shape.Chart;
// Añadir una nueva serie al gráfico
chart.Series.Add("Aspose Series 1", new [] 
{
	// Punto de datos 1
    new DateTime(2002, 05, 01),
	// Punto de datos 2
    new DateTime(2002, 06, 01),
	// Punto de datos 3
    new DateTime(2002, 07, 01),
	// Punto de datos 4
    new DateTime(2002, 08, 01),
	// Punto de datos 5
    new DateTime(2002, 09, 01)
}, 
// Valores correspondientes para los puntos de datos
new double[] { 32, 32, 28, 12, 15 });

//ExStep:3-
// Guardar el documento con el gráfico insertado
doc.Save("WorkingWithCharts.InsertAreaChart.docx");
//ExEnd
