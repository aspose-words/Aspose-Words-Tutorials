//ExStepSummary:0: El siguiente código inicializa un nuevo documento y una instancia de DocumentBuilder.
//ExStepSummary:1: El siguiente código aplica un estilo de numeración predeterminado a la lista y agrega elementos de nivel superior.
//ExStepSummary:2: El siguiente código indents para crear sub-elementos en el segundo elemento de nivel superior.
//ExStepSummary:3: En el siguiente código se insertan además sangrías para crear sub-sub- elementos bajo el segundo sub-elemento.
//ExStepSummary:4: El siguiente código vuelve al nivel anterior para agregar otro elemento en el segundo nivel.
//ExStepSummary:5: El siguiente código vuelve al nivel superior para agregar el elemento final en la lista.
//ExStepSummary:6: El siguiente código elimina la numeración de la lista antes de guardarla.

//ExStepImage:2:images/2.png

//ExStep:0-
// Inicializa un nuevo documento
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Aplicar el formato de numeración predeterminado
builder.ListFormat.ApplyNumberDefault();
builder.Writeln("Item 1");
builder.Writeln("Item 2");

//ExStep:2-
// Indentación para los subtemas
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.1");
builder.Writeln("Item 2.2");

//ExStep:3-
// Indentación para los subtemas
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.2.1");
builder.Writeln("Item 2.2.2");

//ExStep:4-
// Vuelta al segundo nivel
builder.ListFormat.ListOutdent();
builder.Writeln("Item 2.3");

//ExStep:5-
// Outdent volver al nivel superior
builder.ListFormat.ListOutdent();
builder.Writeln("Item 3");

//ExStep:6-
// Eliminar la numeración de la lista
builder.ListFormat.RemoveNumbers();

//ExStep:7-
// Guardar el documento
doc.Save("DocumentFormatting.MultilevelListFormatting.docx");
