//ExSummary: Este código muestra cómo crear un documento de Word e insertar una imagen con un diseño flotante usando Aspose.Words.
//ExStepSummary:0: El siguiente código inicializa un nuevo documento y DocumentBuilder.
//ExStepSummary:1: El siguiente código inserta una imagen en el documento.
//ExStepSummary:2: El siguiente código guarda el documento en una ubicación especificada.
//ExStepImage:1:images/1.png
//ExStep:0-
// Crear una nueva instancia de documento
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Insertar imagen desde la ruta especificada
builder.InsertImage(ImagesDir + "Transparent background logo.png", 
    RelativeHorizontalPosition.Margin,
    100, 
    RelativeVerticalPosition.Margin, 
    100, 
    200, 
    100, 
    WrapType.Square); 

//ExStep:2-
// Guardar el documento en el archivo especificado
doc.Save("AddContentUsingDocumentBuilder.InsertFloatingImage.docx");
//ExEnd
