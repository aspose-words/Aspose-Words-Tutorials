//ExStep:0-
// Crear una nueva instancia de documento
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Añadir texto para la primera página
builder.Writeln("This is page 1.");

//ExStep:2-
// Insertar un salto de página para iniciar una nueva página
builder.InsertBreak(BreakType.PageBreak);

//ExStep:3-
// Añadir texto para la segunda página
builder.Writeln("This is page 2.");

//ExStep:4-
// Insertar otro salto de página para una nueva página
builder.InsertBreak(BreakType.PageBreak);

//ExStep:5-
// Añadir texto para la tercera página
builder.Writeln("This is page 3.");

//ExStep:6-
// Guardar el documento en la ruta especificada
doc.Save("AddContentUsingDocumentBuilder.InsertBreak.docx");
