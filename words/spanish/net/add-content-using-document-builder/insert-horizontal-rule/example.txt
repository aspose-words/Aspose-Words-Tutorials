//ExStepSummary:1: Utilice DocumentBuilder para diseñar el diseño del documento e insertar un mensaje.

//ExStepSummary:3: Guarde el documento en la ruta de archivo especificada.

//ExStep:0-
// Crear un nuevo objeto Document para mantener el contenido
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc); 

//ExStep:1-
// Escribe una línea en el documento
builder.Writeln("Insert a horizontal rule shape into the document.");

//ExStep:2-
// Insertar una forma de regla horizontal en el documento
builder.InsertHorizontalRule();

//ExStep:3-
// Guardar el documento en la ubicación especificada
doc.Save("AddContentUsingDocumentBuilder.InsertHorizontalRule.docx");
