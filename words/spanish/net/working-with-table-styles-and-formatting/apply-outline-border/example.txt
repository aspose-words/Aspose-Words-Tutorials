//ExSummary: Este código muestra cómo aplicar un borde de contorno a una tabla en un documento de Word usando Aspose.Words. Centra la tabla en la página, establece un borde verde alrededor de la tabla, borra cualquier borde existente y llena las celdas de la tabla con un color de fondo verde claro. 
//ExStepSummary:0: El siguiente código inicializa el documento y recupera la primera tabla.
//ExStepSummary:1: Este paso alinea la tabla al centro de la página.
//ExStepSummary:2: Este paso elimina cualquier frontera existente de la tabla.
//ExStepSummary:3: Este paso establece un borde verde alrededor de la mesa, pero no dentro.
//ExStepSummary:4: Este paso llena las células con un color sólido verde claro.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStepImage:4:images/4.png
//ExStep:0-
// Definir el directorio del documento
string dataDir = "Your Document Directory";
// Cargar el documento
Document doc = new Document(dataDir + "Tables.docx");

//ExStep:1-
// Recuperar la primera tabla del documento
Table table = (Table)doc.GetChild(NodeType.Table, 0, true);
// Alinear la tabla al centro de la página
table.Alignment = TableAlignment.Center;

//ExStep:2-
// Borrar cualquier frontera existente de la tabla
table.ClearBorders();

//ExStep:3-
// Establecer borde izquierdo
table.SetBorder(BorderType.Left, LineStyle.Single, 1.5, Color.Green, true);
// Establecer borde derecho
table.SetBorder(BorderType.Right, LineStyle.Single, 1.5, Color.Green, true);
// Establecer borde superior
table.SetBorder(BorderType.Top, LineStyle.Single, 1.5, Color.Green, true);
// Establecer borde inferior
table.SetBorder(BorderType.Bottom, LineStyle.Single, 1.5, Color.Green, true);

//ExStep:4-
// Llena las celdas de color verde claro
table.SetShading(TextureIndex.TextureSolid, Color.LightGreen, Color.Empty);
// Guardar el documento modificado
doc.Save(dataDir + "WorkingWithTableStylesAndFormatting.ApplyOutlineBorder.docx");
//ExEnd
