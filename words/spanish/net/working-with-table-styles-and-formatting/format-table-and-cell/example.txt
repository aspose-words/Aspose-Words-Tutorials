//ExSummary:Este código muestra cómo crear un documento de Word con una tabla que tiene diferentes estilos de borde y sombreado para sus celdas.
//ExStepSummary:0: Inicializar el documento y DocumentBuilder para crear un nuevo documento de Word.
//ExStepSummary:1: Inicie una nueva tabla y establezca los bordes para toda la tabla.
//ExStepSummary:2: Agregue la primera celda con sombreado de fondo rojo y texto.
//ExStepSummary:3: Agregue la segunda celda con sombreado de fondo verde y texto.
//ExStepSummary:4: Finalizar la primera fila y borrar el formato de celda anterior.
//ExStepSummary:5: Agregue una nueva fila con bordes más grandes personalizados para la primera celda.
//ExStepSummary:6: Agregue la segunda celda de la nueva fila con formato y texto predeterminados.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStepImage:4:images/4.png
//ExStepImage:5:images/5.png

//ExStep:0-
// Crear un nuevo documento.
Document doc = new Document();
// Cree un DocumentBuilder para manipular el documento.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Inicie una nueva mesa.
Table table = builder.StartTable();
// Inserte la primera celda en la tabla.
builder.InsertCell();
// Establezca los límites para toda la mesa.
table.SetBorders(LineStyle.Single, 2.0, Color.Black); 

//ExStep:2-
// Establezca el color de fondo de la celda en rojo.
builder.CellFormat.Shading.BackgroundPatternColor = Color.Red;
builder.Writeln("Cell #1");
// Inserte la segunda celda.
builder.InsertCell();

//ExStep:3-
// Establezca el color de fondo de la celda en verde.
builder.CellFormat.Shading.BackgroundPatternColor = Color.Green;
builder.Writeln("Cell #2");
// Termina la fila actual.
builder.EndRow();

//ExStep:4-
// Borre cualquier formato de celda para comenzar de nuevo.
builder.CellFormat.ClearFormatting();
// Inserte una nueva celda en la siguiente fila.
builder.InsertCell();
// Cree bordes más grandes para la primera celda de esta fila.
builder.CellFormat.Borders.Left.LineWidth = 4.0;
builder.CellFormat.Borders.Right.LineWidth = 4.0;
builder.CellFormat.Borders.Top.LineWidth = 4.0;
builder.CellFormat.Borders.Bottom.LineWidth = 4.0;
builder.Writeln("Cell #3");
//ExStep:5-
// Inserte la cuarta celda.
builder.InsertCell();
builder.CellFormat.ClearFormatting();
builder.Writeln("Cell #4");
//ExStep:6-
// Guarde el documento.
doc.Save("WorkingWithTableStylesAndFormatting.FormatTableAndCellWithDifferentBorders.docx"); 
//ExEnd
