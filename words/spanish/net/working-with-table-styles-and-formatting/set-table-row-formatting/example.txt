//ExStepSummary:0: Este paso inicializa un nuevo documento y el constructor de documentos.
//ExStepSummary:1: Este paso inicia la tabla e inserta una celda en ella.
//ExStepSummary:2: Este paso establece el formato de la fila, incluida la regla de altura y altura.
//ExStepSummary:3: Este paso establece el relleno para toda la tabla, afectando a todas las filas.
//ExStepSummary:4: Este paso agrega texto a la celda y finaliza la fila.
//ExStepSummary:5: Este paso finaliza la tabla y guarda el documento en una ruta especificada.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStepImage:4:images/4.png

//ExStep:0-
// Crear una nueva instancia de documento
Document doc = new Document();
// Inicializa DocumentBuilder con el documento
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Iniciar una nueva mesa
Table table = builder.StartTable();
// Insertar la primera celda en la tabla
builder.InsertCell();

//ExStep:2-
// Acceda a la propiedad RowFormat del constructor
RowFormat rowFormat = builder.RowFormat;
// Fijar la altura de la fila
rowFormat.Height = 100;
// Establezca la regla de altura exactamente
rowFormat.HeightRule = HeightRule.Exactly;

//ExStep:3-
// Set de relleno para la mesa
table.LeftPadding = 30;
table.RightPadding = 30;
table.TopPadding = 30;
table.BottomPadding = 30;

//ExStep:4-
// Añadir texto a la celda actual
builder.Writeln("I'm a wonderful formatted row.");
// Finalizar la fila actual
builder.EndRow();
builder.EndTable();

//ExStep:5-
// Guardar el documento en la ruta especificada
doc.Save("WorkingWithTableStylesAndFormatting.DocumentBuilderSetTableRowFormatting.docx");
//ExEnd
