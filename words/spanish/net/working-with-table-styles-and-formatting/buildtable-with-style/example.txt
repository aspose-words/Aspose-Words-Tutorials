//ExStepSummary:0: Inicializar un nuevo documento y DocumentBuilder para administrar la estructura del documento.
//ExStepSummary:1: Inicie una nueva tabla en el documento con DocumentBuilder.
//ExStepSummary:2: Inserte la primera celda y establezca el identificador de estilo de tabla.
//ExStepSummary:3: Aplicar opciones de formato a la tabla y establecer el comportamiento de ajuste automático.
//ExStepSummary:4: Agregue celdas de fila de encabezado a la tabla.
//ExStepSummary:5: Inserte filas de datos en la tabla con nombres y cantidades de elementos.
//ExStepSummary:6: Guarde el documento ensamblado en un archivo.
//ExStepImage:2:images/1.png

//ExStep:0-
// Crear un nuevo objeto de documento.
Document doc = new Document();
// Cree un constructor para facilitar la generación de documentos.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Inicie una nueva tabla en el documento.
Table table = builder.StartTable();

//ExStep:2-
// Inserte la primera celda antes de establecer el formato.
builder.InsertCell();
// Establezca el identificador de estilo para la tabla.
table.StyleIdentifier = StyleIdentifier.MediumShading1Accent1;


//ExStep:3-
// Configure las opciones de estilo.
table.StyleOptions = 
    TableStyleOptions.FirstColumn | TableStyleOptions.RowBands | TableStyleOptions.FirstRow;
// Establezca el comportamiento de ajuste automático de la tabla.	
table.AutoFit(AutoFitBehavior.AutoFitToContents);


//ExStep:4-
// Escribe el encabezado de la primera columna.
builder.Writeln("Item");
// Ajuste el relleno adecuado para la celda.
builder.CellFormat.RightPadding = 40;
// Inserte la segunda celda para la siguiente columna.
builder.InsertCell();
// Escribe el encabezado de la segunda columna.
builder.Writeln("Quantity (kg)");
// Finaliza la creación de la fila.
builder.EndRow();

//ExStep:5-
// Inicie una nueva fila e inserte la primera celda.
builder.InsertCell();
builder.Writeln("Apples");
builder.InsertCell();
builder.Writeln("20");
builder.EndRow();
 // Inserte la primera celda de una nueva fila para el segundo elemento.
builder.InsertCell();
builder.Writeln("Bananas");
builder.InsertCell();
builder.Writeln("40");
builder.EndRow();
// Comience una nueva fila para el tercer elemento.
builder.InsertCell();
builder.Writeln("Carrots");
builder.InsertCell(); 
builder.Writeln("50");
builder.EndRow();

//ExStep:6-
// Guarde el documento en un archivo especificado.
doc.Save("WorkingWithTableStylesAndFormatting.BuildTableWithStyle.docx");
