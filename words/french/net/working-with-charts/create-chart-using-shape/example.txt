// Créer un nouvel objet Document
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Insérer un diagramme de ligne dans le document aux dimensions spécifiées
Shape shape = builder.InsertChart(ChartType.Line, 432, 252);

//ExStep:2-
// Accéder à l'objet Chart depuis la forme
Chart chart = shape.Chart;
// Activer la visibilité du titre du graphique
chart.Title.Show = true;
// Définir le texte du titre pour le graphique
chart.Title.Text = "Line Chart Title";
// S'assurer que le titre ne recouvre pas le graphique
chart.Title.Overlay = false;

//ExStep:3-
// Définissez la position de la légende du graphique à gauche
chart.Legend.Position = LegendPosition.Left;
chart.Legend.Overlay = true;

//ExStep:4-
// Enregistrer le document avec le graphique dans un fichier
doc.Save("WorkingWithCharts.CreateChartUsingShape.docx");
