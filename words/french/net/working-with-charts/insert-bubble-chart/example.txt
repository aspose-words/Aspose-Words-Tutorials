//ExStart
//ExSummary: Ce code montre comment insérer un graphique Bubble dans un document Word en utilisant Aspose.Words.
//ExStepSummary:0: Le code suivant initialise un nouveau Document et DocumentBuilder.
//ExStepSummary:1: Un diagramme de bulles est créé et ajouté au document à l'aide du documentBuilder.
//ExStepSummary:2: Les séries de données pour le graphique Bubble sont ajoutées avec des valeurs spécifiques de X, Y et de taille.
//ExStepSummary:3: Enfin, le document est enregistré avec le graphique Bubble inséré.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Initialiser un nouveau document et documentBuilder
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Insérer un diagramme de bulles dans le document
Shape shape = builder.InsertChart(ChartType.Bubble, 432, 252);

//ExStep:2-
// Ajouter des séries de données au graphique Bubble
Chart chart = shape.Chart;
chart.Series.Add("Aspose Series 1", new double[] { 0.7, 1.8, 2.6 }, // X values
    new double[] { 2.7, 3.2, 0.8 }, // Y values
    new double[] { 10, 4, 8 }); // Size values for the bubbles

//ExStep:3-
// Enregistrez le document avec le graphique Bubble
doc.Save("WorkingWithCharts.InsertBubbleChart.docx");
//ExEnd
