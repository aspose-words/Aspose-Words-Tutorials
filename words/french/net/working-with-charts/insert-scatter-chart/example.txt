//ExSummary:Ce code montre comment créer un document Word avec un graphique scatter en utilisant Aspose.Words. Il initialise un document, construit du contenu, insère un graphique scatter, y ajoute des données et enregistre le document.
//ExStepSummary:0: Initialisation des objets Document et DocumentBuilder pour commencer à travailler sur le document.
//ExStepSummary:1: Insertion d'un graphique d'éparpillement dans le document.
//ExStepSummary:2: Ajout d'une série de points de données au graphique de dispersion.
//ExStepSummary:3: Enregistrer le document avec le graphique scatter dans un fichier .docx.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Initialiser un nouveau document
Document doc = new Document();
// Créer un constructeur pour insérer du contenu
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Insérer un graphique de dispersion dans le document
Shape shape = builder.InsertChart(ChartType.Scatter, 432, 252); 

//ExStep:2-
// Obtenez l'objet graphique de la forme insérée
Chart chart = shape.Chart;
// Ajouter des données de série au graphique
chart.Series.Add("Aspose Series 1", new double[] { 0.7, 1.8, 2.6 }, new double[] { 2.7, 3.2, 0.8 });

//ExStep:3-
// Enregistrer le document avec le graphique de dispersion
doc.Save("WorkingWithCharts.InsertScatterChart.docx");
//ExEnd
