//ExStepSummary:0: Le code suivant initialise un nouveau document et une instance DocumentBuilder.
//ExStepSummary:1: Le code suivant applique un style de numérotation par défaut à la liste et ajoute des éléments de haut niveau.
//ExStepSummary:2: Les tirets de code suivants pour créer des sous-éléments dans le deuxième élément de niveau supérieur.
//ExStepSummary:3: Les sous-alinéas suivants visent à créer des sous-alinéas au titre du deuxième sous-alinéa.
//ExStepSummary:4: Le code qui suit renvoie au niveau précédent pour ajouter un autre élément au deuxième niveau.
//ExStepSummary:5: Le code qui suit retourne au niveau supérieur pour ajouter l'élément final dans la liste.
//ExStepSummary:6: Le code suivant supprime la numérotation de la liste avant d'enregistrer.

//ExStep:0-
// Initialiser un nouveau document
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Appliquer le format de numérotation par défaut
builder.ListFormat.ApplyNumberDefault();
builder.Writeln("Item 1");
builder.Writeln("Item 2");

//ExStep:2-
// Indicatif pour les sous-points
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.1");
builder.Writeln("Item 2.2");

//ExStep:3-
// Indique pour les sous-sous-points
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.2.1");
builder.Writeln("Item 2.2.2");

//ExStep:4-
// Retour au deuxième niveau
builder.ListFormat.ListOutdent();
builder.Writeln("Item 2.3");

//ExStep:5-
// Retour au niveau supérieur
builder.ListFormat.ListOutdent();
builder.Writeln("Item 3");

//ExStep:6-
// Supprimer la numérotation de la liste
builder.ListFormat.RemoveNumbers();

//ExStep:7-
// Enregistrer le document
doc.Save("DocumentFormatting.MultilevelListFormatting.docx");
