//ExStart
//ExSummary: Ce code montre comment appliquer les bordures et l'ombrage à un paragraphe d'un document Word utilisant Aspose.Words.
//ExStepSummary:0: Le code suivant initialise une nouvelle instance Document et DocumentBuilder.
//ExStepSummary:1: Le code suivant configure les bordures du paragraphe à une distance du texte.
//ExStepSummary:2: Le code suivant applique une ombrage diagonale de texture croisée au paragraphe avec des couleurs spécifiées.
//ExStepSummary:3: Le code suivant écrit un paragraphe formaté avec les paramètres appliqués.
//ExStepSummary:4: Le code suivant enregistre le document sur le chemin de sortie spécifié.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0-
// Initialiser un nouveau document
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Accès aux frontières du paragraphe
BorderCollection borders = builder.ParagraphFormat.Borders;
borders.DistanceFromText = 20;
borders[BorderType.Left].LineStyle = LineStyle.Double;
borders[BorderType.Right].LineStyle = LineStyle.Double;
borders[BorderType.Top].LineStyle = LineStyle.Double;
borders[BorderType.Bottom].LineStyle = LineStyle.Double;

//ExStep:2-
// Accéder aux propriétés d'ombrage du paragraphe
Shading shading = builder.ParagraphFormat.Shading;
shading.Texture = TextureIndex.TextureDiagonalCross;
shading.BackgroundPatternColor = System.Drawing.Color.LightCoral; 
shading.ForegroundPatternColor = System.Drawing.Color.LightSalmon;

//ExStep:3-
// Écrire le contenu du paragraphe formaté
builder.Write("I'm a formatted paragraph with double border and nice shading.");

//ExStep:4-
// Enregistrer le document sur le chemin spécifié
doc.Save(ArtifactsDir + "DocumentFormatting.ApplyBordersAndShadingToParagraph.doc");
//ExEnd
