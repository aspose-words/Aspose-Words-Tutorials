// Créer un nouvel objet Document pour conserver le contenu
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc); 

//ExStep:1-
// Écrire une ligne dans le document
builder.Writeln("Insert a horizontal rule shape into the document.");

//ExStep:2-
// Insérer une forme de règle horizontale dans le document
builder.InsertHorizontalRule();

//ExStep:3-
// Enregistrer le document à l'emplacement spécifié
doc.Save("AddContentUsingDocumentBuilder.InsertHorizontalRule.docx");
