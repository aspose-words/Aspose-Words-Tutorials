//ExSummary: Ce code montre comment créer un document Word avec une table en utilisant Aspose.Words. Il construit une table avec deux lignes et différents formats d'orientation et de texte de cellule.
//ExStepSummary:0: Le code suivant initialise un nouveau fabricant de documents et de documents.
//ExStepSummary:1: Le code suivant démarre une nouvelle table et insère la première ligne de cellules.
//ExStepSummary:2: Le code suivant configure le formatage pour la deuxième ligne et ses cellules.
//ExStepSummary:3: Le code suivant configure le formatage pour la deuxième ligne et ses cellules.
//ExStepSummary:4: Le code suivant termine la table et enregistre le document dans un fichier.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png

//ExStep:0-
// Initialiser un nouvel objet Document
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Démarrer une nouvelle table
Table table = builder.StartTable();
// Commencer la première cellule
builder.InsertCell();
// Régler l'alignement vertical de la première cellule
builder.CellFormat.VerticalAlignment = CellVerticalAlignment.Center; 
builder.Write("This is row 1 cell 1");
// Insérer la deuxième cellule
builder.InsertCell();
builder.Write("This is row 1 cell 2");
builder.EndRow();

//ExStep:2-
// Insérer une cellule pour la deuxième ligne
builder.InsertCell();
// Configurer le formatage pour la deuxième ligne
builder.RowFormat.Height = 100;
builder.RowFormat.HeightRule = HeightRule.Exactly;
builder.CellFormat.Orientation = TextOrientation.Upward;
builder.Writeln("This is row 2 cell 1");
//ExStep:3-
// Insérer la deuxième cellule pour la deuxième ligne
builder.InsertCell();
builder.CellFormat.Orientation = TextOrientation.Downward; 
builder.Writeln("This is row 2 cell 2");
builder.EndRow();
builder.EndTable();

//ExStep:4-
// Ajustement automatique de la table avec des largeurs de colonnes fixes
table.AutoFit(AutoFitBehavior.FixedColumnWidths);
// Enregistrer le document dans un fichier
doc.Save("AddContentUsingDocumentBuilder.BuildTable.docx");
//ExEnd
