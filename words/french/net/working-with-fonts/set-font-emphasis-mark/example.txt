//ExSummary:Ce code montre comment mettre une marque d'accent dans un document Word en utilisant Aspose.words.
//ExStepSummary:0:Création d'une nouvelle instance Document avec laquelle travailler.
//ExStepSummary:1:Configuration d'un documentBuilder pour manipuler le document.
//ExStepSummary:2:Appliquer une marque d'accent à la police du constructeur.
//ExStepSummary:3:Écrire un texte d'accent sur le document.
//ExStepSummary:4:Écrire un texte simple sans aucun accent.
//ExStepSummary:5: Ce code enregistre le document avec la marque de mise en évidence de la police.
//ExStepImage:2:images/1.png
//ExStepImage:3:images/2.png
//ExStepImage:4:images/3.png
//ExStep:0-
// Créer un nouveau document
Document document = new Document();

//ExStep:1-
// Initialiser DocumentBuilder pour la manipulation de documents
DocumentBuilder builder = new DocumentBuilder(document);

//ExStep:2-
// Mettre l'accent sur UnderSolidCircle
builder.Font.EmphasisMark = EmphasisMark.UnderSolidCircle;

//ExStep:3-
// Écrire le texte de l'accent sur le document
builder.Write("Emphasis text");
// Déplacez-vous vers la ligne suivante
builder.Writeln();
// Effacer le formatage de la police pour le texte suivant
builder.Font.ClearFormatting();

//ExStep:4-
// Écrire un texte simple sans mise en évidence
builder.Write("Simple text");

//ExStep:5-
// Enregistrer le document sur le chemin spécifié
document.Save("WorkingWithFonts.SetFontEmphasisMark.docx");
//ExEnd
