Het centreert de tabel op de pagina, zet een groene rand rond de tabel, verwijdert eventuele bestaande randen, en vult de tafelcellen met een lichtgroene achtergrondkleur. 
//ExStepSummary:0: De volgende code initialiseert het document en haalt de eerste tabel.
//ExStepSummary:1: Deze stap brengt de tabel naar het midden van de pagina.
//ExStepSummary:2: Deze stap verwijdert alle bestaande grenzen van de tabel.
//ExStepSummary:3: Deze stap zet een groene rand rond de tafel maar niet binnen.
//ExStepSummary:4: Deze stap vult de cellen met een lichtgroene effen kleur.

// De map van het document definiÃ«ren
string dataDir = "Your Document Directory";
// Het document laden
Document doc = new Document(dataDir + "Tables.docx");

//ExStep:1-
// De eerste tabel uit het document ophalen
Table table = (Table)doc.GetChild(NodeType.Table, 0, true);
// De tabel uitlijnen naar het midden van de pagina
table.Alignment = TableAlignment.Center;

//ExStep:2-
// Alle bestaande randen uit de tabel wissen
table.ClearBorders();

//ExStep:3-
// Linkerrand instellen
table.SetBorder(BorderType.Left, LineStyle.Single, 1.5, Color.Green, true);
// Rechterrand instellen
table.SetBorder(BorderType.Right, LineStyle.Single, 1.5, Color.Green, true);
// Bovenrand instellen
table.SetBorder(BorderType.Top, LineStyle.Single, 1.5, Color.Green, true);
// Onderrand instellen
table.SetBorder(BorderType.Bottom, LineStyle.Single, 1.5, Color.Green, true);

//ExStep:4-
// Vul de cellen met lichtgroene kleur
table.SetShading(TextureIndex.TextureSolid, Color.LightGreen, Color.Empty);
// Het gewijzigde document opslaan
doc.Save(dataDir + "WorkingWithTableStylesAndFormatting.ApplyOutlineBorder.docx");
