//ExStepSummary:0: Initialiseer een nieuw Document en DocumentBuilder voor het beheren van de documentstructuur.
//ExStepSummary:1: Start een nieuwe tabel in het document met de DocumentBuilder.
//ExStepSummary:2: Voer de eerste cel in en stel de tabelstijl-identificatie in.
//ExStepSummary:3: Opmaakopties toepassen op de tabel en auto-fit gedrag instellen.
//ExStepSummary:4: Voeg header rij cellen aan de tabel.
//ExStepSummary:5: Gegevensrijen in de tabel met itemnamen en -hoeveelheden invoegen.
//ExStepSummary:6: Het samengevoegde document opslaan in een bestand.

//ExStep:0-
// Een nieuw document-object aanmaken.
Document doc = new Document();
// Maak een bouwer aan om het genereren van documenten te vergemakkelijken.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Start een nieuwe tabel in het document.
Table table = builder.StartTable();

//ExStep:2-
// Voer de eerste cel in voordat u de opmaak instelt.
builder.InsertCell();
// Stel de stijl-identifier in voor de tabel.
table.StyleIdentifier = StyleIdentifier.MediumShading1Accent1;


//ExStep:3-
// Stijlopties configureren.
table.StyleOptions = 
    TableStyleOptions.FirstColumn | TableStyleOptions.RowBands | TableStyleOptions.FirstRow;
// Stel het automatische gedrag van de tafel in.	
table.AutoFit(AutoFitBehavior.AutoFitToContents);


//ExStep:4-
// Schrijf de koptekst voor de eerste kolom.
builder.Writeln("Item");
// Zet rechts opvulling voor de cel.
builder.CellFormat.RightPadding = 40;
// De tweede cel invoegen voor de volgende kolom.
builder.InsertCell();
// Schrijf de koptekst voor de tweede kolom.
builder.Writeln("Quantity (kg)");
// BeÃ«indig de rij aanmaak.
builder.EndRow();

//ExStep:5-
// Start een nieuwe rij en plaats de eerste cel.
builder.InsertCell();
builder.Writeln("Apples");
builder.InsertCell();
builder.Writeln("20");
builder.EndRow();
 // Plaats de eerste cel van een nieuwe rij voor het tweede item.
builder.InsertCell();
builder.Writeln("Bananas");
builder.InsertCell();
builder.Writeln("40");
builder.EndRow();
// Begin een nieuwe rij voor het derde item.
builder.InsertCell();
builder.Writeln("Carrots");
builder.InsertCell(); 
builder.Writeln("50");
builder.EndRow();

//ExStep:6-
// Het document opslaan in een opgegeven bestand.
doc.Save("WorkingWithTableStylesAndFormatting.BuildTableWithStyle.docx");
