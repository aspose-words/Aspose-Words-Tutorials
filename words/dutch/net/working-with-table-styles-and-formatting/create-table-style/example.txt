//ExStart
//ExSummary: Deze code laat zien hoe je een Word-document maakt met een gestyleerde tabel met behulp van Aspose.Words. 
//ExStepSummary:0: Initialiseer een nieuw Document en DocumentBuilder.
//ExStepSummary:1: Start een nieuwe tabel en plaats header cellen voor "Name" en "Value."
//ExStepSummary:2: Maak een aangepaste TableStyle met specifieke rand- en padding-instellingen.
//ExStepSummary:3: Het document opslaan met de nieuwe tabelstijl toegepast.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Een nieuw document-object aanmaken
Document doc = new Document();
// Initialiseer de DocumentBuilder om het document te bouwen
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Een nieuwe tabel starten
Table table = builder.StartTable();
builder.InsertCell();
builder.Write("Name");
builder.InsertCell();
builder.Write("Value");
builder.EndRow();
builder.InsertCell();
builder.InsertCell();
builder.EndTable();

//ExStep:2-
// Een nieuwe tabelstijl aanmaken
TableStyle tableStyle = (TableStyle) doc.Styles.Add(StyleType.Table, "MyTableStyle1");
tableStyle.Borders.LineStyle = LineStyle.Double;
// Randbreedte instellen
tableStyle.Borders.LineWidth = 1;
// Opvulling instellen
tableStyle.LeftPadding = 18;
tableStyle.RightPadding = 18;
tableStyle.TopPadding = 12;
tableStyle.BottomPadding = 12;
// Pas de aangepaste tabelstijl toe op de aangemaakte tabel
table.Style = tableStyle;

//ExStep:3-
// Het document opslaan met de gestileerde tabel
doc.Save("WorkingWithTableStylesAndFormatting.CreateTableStyle.docx");
//ExEnd
