//ExStepSummary:1: Plaats een grafiek in het document en haal het grafiekobject op. 
//ExStepSummary:2: Voeg een nieuwe serie toe aan de grafiek met gespecificeerde categorieÃ«n en waarden.

//ExStep:0-
// Een nieuwe document-initiatie aanmaken.
Document doc = new Document();
// Initialiseer DocumentBuilder om het document te bouwen.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Voeg een kolomdiagram in het document.
Shape shape = builder.InsertChart(ChartType.Column, 432, 252);
// Haal het grafiekobject uit de vorm.
Chart chart = shape.Chart;
// Verwijder alle bestaande reeksen uit de grafiek voor nieuwe gegevens.
chart.Series.Clear();

//ExStep:2-
// Definieer een nieuwe serie met de naam "Aspose Series 1."
chart.Series.Add("Aspose Series 1",
    new string[] { "Item 1", "Item 2", "Item 3", "Item 4", "Item 5" }, // Set categories for the series.
    new double[] { 1.2, 0.3, 2.1, 2.9, 4.2 });

//ExStep:3-
// Stel de minimumgrens van de Y-as in.
chart.AxisY.Scaling.Minimum = new AxisBound(0);
chart.AxisY.Scaling.Maximum = new AxisBound(6);

//ExStep:4-
// Sla het document met de grafiek op naar het opgegeven pad.
doc.Save("WorkingWithCharts.BoundsOfAxis.docx");
