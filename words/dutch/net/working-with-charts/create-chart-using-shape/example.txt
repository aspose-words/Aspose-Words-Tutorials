// Een nieuw document-object aanmaken
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Plaats een lijndiagram in het document bij gespecificeerde afmetingen
Shape shape = builder.InsertChart(ChartType.Line, 432, 252);

//ExStep:2-
// Bekijk het grafiek-object vanuit de vorm
Chart chart = shape.Chart;
// De titelweergave van de kaart inschakelen
chart.Title.Show = true;
// De titeltekst voor de grafiek instellen
chart.Title.Text = "Line Chart Title";
// Zorg ervoor dat de titel de grafiek niet overlapt
chart.Title.Overlay = false;

//ExStep:3-
// Stel de positie van de legende in naar links
chart.Legend.Position = LegendPosition.Left;
chart.Legend.Overlay = true;

//ExStep:4-
// Het document met de grafiek opslaan in een bestand
doc.Save("WorkingWithCharts.CreateChartUsingShape.docx");
