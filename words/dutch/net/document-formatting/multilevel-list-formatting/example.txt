//ExStepSummary:2: De volgende code streepjes om sub-items te maken onder het tweede top-level item.
//ExStepSummary:3: De volgende code extra streepjes om subsubposten te creÃ«ren onder de tweede subpost.
//ExStepSummary:4: De volgende code gaat terug naar het vorige niveau om een ander item toe te voegen op het tweede niveau.
//ExStepSummary:5: De volgende code gaat terug naar het bovenste niveau om het laatste item in de lijst toe te voegen.

//ExStep:0-
// Een nieuw document initialiseren
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Standaard nummeringsformaat toepassen
builder.ListFormat.ApplyNumberDefault();
builder.Writeln("Item 1");
builder.Writeln("Item 2");

//ExStep:2-
// Inspringen voor subitems
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.1");
builder.Writeln("Item 2.2");

//ExStep:3-
// Inspringen voor sub-sub-items
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.2.1");
builder.Writeln("Item 2.2.2");

//ExStep:4-
// Terugspringen naar het tweede niveau
builder.ListFormat.ListOutdent();
builder.Writeln("Item 2.3");

//ExStep:5-
// Terugspringen naar het bovenste niveau
builder.ListFormat.ListOutdent();
builder.Writeln("Item 3");

//ExStep:6-
// Nummering uit de lijst verwijderen
builder.ListFormat.RemoveNumbers();

//ExStep:7-
// Het document opslaan
doc.Save("DocumentFormatting.MultilevelListFormatting.docx");
