//ExSummary: Deze code laat zien hoe je een tekst watermerk met specifieke opties toevoegt aan een Word-document met behulp van Aspose.Words.
//ExStepSummary:0: De volgende code initialiseert het documentpad en maakt een nieuw documentobject aan.
//ExStepSummary:1: De volgende code stelt de opties voor de tekst watermerk, met inbegrip van lettertype familie, grootte, kleur en lay-out.
//ExStepSummary:2: De volgende code is van toepassing op de tekst watermerk met de opgegeven opties op het document.
//ExStepSummary:3: De volgende code slaat het gewijzigde document op in een opgegeven map.
//ExStepImage:2:images/1.png
//ExStepImage:3:images/2.png
//ExStep:0-
// Het mappad voor het document instellen
string dataDir = "Your Document Directory";

//ExStep:1-
// Het Word-document laden
Document doc = new Document(dataDir + "Document.docx");

//ExStep:2-
// Opties voor watermerk aanmaken
TextWatermarkOptions options = new TextWatermarkOptions()
{
    FontFamily = "Arial", // Set font family for the watermark text
    FontSize = 36, // Set font size for the watermark text
    Color = Color.Black, // Set color of the watermark text
    Layout = WatermarkLayout.Horizontal, // Specify layout of the watermark
    IsSemitrasparent = false // Set transparency option
};

//ExStep:3-
// De watermerktekst met opties toepassen
doc.Watermark.SetText("Test", options);

//ExStep:4-
// Het document opslaan met het watermerk
doc.Save(dataDir + "WorkWithWatermark.AddTextWatermarkWithSpecificOptions.docx");
//ExEnd
