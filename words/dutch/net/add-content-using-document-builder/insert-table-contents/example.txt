//ExSummary: Deze code laat zien hoe je een Word-document kunt maken met een Inhoudsopgave met behulp van DocumentBuilder. Het bevat rubrieken van verschillende niveaus en slaat het document op als een .docx-bestand.
//ExStepSummary:0: Deze stap initialiseert een nieuw documentobject.
//ExStepSummary:1: Deze stap maakt een DocumentBuilder object aan voor het document manipuleren.
//ExStepSummary:2: Deze stap plaatst de Inhoudsopgave aan het begin van het document.
//ExStepSummary:3: Deze stap introduceert een paginapauze om de werkelijke inhoud op een nieuwe pagina te starten.
//ExStepSummary:4: Deze stap stelt de eerste kop in en schrijft "Rubriek 1" naar het document.
//ExStepSummary:5: Deze stap stelt het niveau van de tweede rubriek in en schrijft "Rubriek 1.1" en "Rubriek 1.2."
//ExStepSummary:6: Deze stap stelt opnieuw het niveau van de eerste rubriek in en schrijft "Rubriek 2" en "Rubriek 3."
//ExStepSummary:7: Deze stap stelt het niveau van de tweede rubriek in en schrijft "Rubriek 3.1."
//ExStepSummary:8: Deze stap stelt het niveau van de derde rubriek vast en schrijft de onderverdelingen "Rubriek 3.1.1," "Rubriek 3.1.2" en "Rubriek 3.1.3."
//ExStepSummary:9: Deze stap stelt het niveau van de tweede rubriek opnieuw in en schrijft "Rubriek 3.2" en "Rubriek 3.3."
//ExStepSummary:10: Tot slot, deze stap update de velden in het document en slaat het op als een .docx bestand.

//ExStep:0-
// Een nieuwe document-instantie aanmaken
Document doc = new Document();

//ExStep:1-
// DocumentBuilder initialiseren voor documentmanipulatie
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:2-
// Inhoudsopgave met gespecificeerde opties invoegen
builder.InsertTableOfContents("\\o \"1-3\" \\h \\z \\u");

//ExStep:3-
// Een paginapauze invoegen om inhoud op een nieuwe pagina te starten
builder.InsertBreak(BreakType.PageBreak);

//ExStep:4-
// Stijl instellen op rubriek 1
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading1;
builder.Writeln("Heading 1");

//ExStep:5-
// Stijl instellen op rubriek 2
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading2;
builder.Writeln("Heading 1.1");
builder.Writeln("Heading 1.2");

//ExStep:6-
// Stijl terugzetten naar rubriek 1
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading1;
builder.Writeln("Heading 2");
builder.Writeln("Heading 3");

//ExStep:7-
// Stijl instellen op rubriek 2
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading2;
builder.Writeln("Heading 3.1");

//ExStep:8-
// Stijl instellen op rubriek 3
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading3;
builder.Writeln("Heading 3.1.1");
builder.Writeln("Heading 3.1.2");
builder.Writeln("Heading 3.1.3");

//ExStep:9-
// Stijl terugzetten naar rubriek 2
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading2;
builder.Writeln("Heading 3.2");
builder.Writeln("Heading 3.3");

//ExStep:10-
// Alle velden in het document bijwerken
doc.UpdateFields();
// Het document opslaan
doc.Save("AddContentUsingDocumentBuilder.InsertTableOfContents.docx");
//ExEnd
