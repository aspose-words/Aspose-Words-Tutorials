//ExStepSummary:1: De volgende code start een nieuwe tabel en plaatst de eerste rij cellen.
//ExStepSummary:2: De volgende code configureert de opmaak voor de tweede rij en de cellen.
//ExStepSummary:3: De volgende code configureert de opmaak voor de tweede rij en de cellen.

//ExStep:0-
// Een nieuw document-object initialiseren
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Een nieuwe tabel starten
Table table = builder.StartTable();
// Eerste cel beginnen
builder.InsertCell();
// Vertikale uitlijning instellen voor de eerste cel
builder.CellFormat.VerticalAlignment = CellVerticalAlignment.Center; 
builder.Write("This is row 1 cell 1");
// Tweede cel invoegen
builder.InsertCell();
builder.Write("This is row 1 cell 2");
builder.EndRow();

//ExStep:2-
// Een cel invoegen voor de tweede rij
builder.InsertCell();
// Opmaak instellen voor de tweede rij
builder.RowFormat.Height = 100;
builder.RowFormat.HeightRule = HeightRule.Exactly;
builder.CellFormat.Orientation = TextOrientation.Upward;
builder.Writeln("This is row 2 cell 1");
//ExStep:3-
// Tweede cel invoegen voor de tweede rij
builder.InsertCell();
builder.CellFormat.Orientation = TextOrientation.Downward; 
builder.Writeln("This is row 2 cell 2");
builder.EndRow();
builder.EndTable();

//ExStep:4-
// De tabel automatisch aanpassen met vaste kolombreedtes
table.AutoFit(AutoFitBehavior.FixedColumnWidths);
// Het document opslaan in een bestand
doc.Save("AddContentUsingDocumentBuilder.BuildTable.docx");
