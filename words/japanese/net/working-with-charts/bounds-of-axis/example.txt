//ExSummary: このコードは、Asfo.words を使用してグラフを使ってドキュメントを作成する方法を示します。 Y 軸の境界を設定し、ドキュメントを DOCX ファイルとして保存します。 
// 各ステップは、Astrpostのチャートを操作する作業に係る工程について明確に理解できるように分割されます。
//ExStepSummary:0: 新しい文書と文書ビルダーのインスタンスを初期化します。 
//ExStepSummary:1: ドキュメントにグラフを挿入し、グラフオブジェクトを取得します。 
//ExStepSummary:2: カテゴリと値を指定して、グラフに新しいシリーズを追加します。 
//ExStepSummary:3: グラフの Y 軸の最小値と最大値を設定します。 
//ExStepSummary:4: ドキュメントを指定したパスに保存します。 
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0-
// 新しい文書インスタンスを作成します。
Document doc = new Document();
// ドキュメント作成を初期化します。
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// ドキュメントに列グラフを挿入します。
Shape shape = builder.InsertChart(ChartType.Column, 432, 252);
// シェイプからグラフオブジェクトを取得します。
Chart chart = shape.Chart;
// グラフから既存の系列をクリアします。
chart.Series.Clear();

//ExStep:2-
// 2020年1月15日閲覧. ^ a b c d e f g h i "アスキーシリーズ1"を新シリーズに指定".
chart.Series.Add("Aspose Series 1",
    new string[] { "Item 1", "Item 2", "Item 3", "Item 4", "Item 5" }, // Set categories for the series.
    new double[] { 1.2, 0.3, 2.1, 2.9, 4.2 });

//ExStep:3-
// Y 軸の最小境界を設定します。
chart.AxisY.Scaling.Minimum = new AxisBound(0);
chart.AxisY.Scaling.Maximum = new AxisBound(6);

//ExStep:4-
// ドキュメントを、グラフで指定したパスに保存します。
doc.Save("WorkingWithCharts.BoundsOfAxis.docx");
