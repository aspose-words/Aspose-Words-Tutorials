//ExStart
//ExSummary: Dieser Code zeigt, wie Formen (Textboxen) in ein Word-Dokument mit Aspose.Words eingefügt werden. Das Dokument wird im DOCX-Format mit spezifischen Compliance-Optionen für ISO-Standards gespeichert.
//ExStepSummary:0: Der Code initialisiert ein neues Dokument und einen Dokument-Builder zur Manipulation des Dokuments.
//ExStepSummary:1: Eine Form (Textbox) wird an einer absoluten Position im Dokument eingefügt und die Rotation der ersten Form auf 30 Grad eingestellt.
//ExStepSummary:2: Eine neue Form wird an einer relativen Position eingefügt und die Rotation der zweiten Form wird ebenfalls auf 30 Grad eingestellt.
//ExStepSummary:3: Das Dokument wird mit spezifischen OoxmlSaveOptionen gespeichert.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png

//ExStep:0-
// Erstellt eine neue Dokument-Instanz
Document doc = new Document();
// Initialisiert einen DocumentBuilder für das Dokument
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Fügt eine Form des Typs TextBox ein
Shape shape = builder.InsertShape(ShapeType.TextBox, RelativeHorizontalPosition.Page, 100, 
    RelativeVerticalPosition.Page, 100, 50, 50, WrapType.None); // Sets the position and size of the shape
// Dreht die Form um 30 Grad
shape.Rotation = 30.0;

//ExStep:2-
builder.Writeln();
// Fügt eine andere Form des Typs TextBox an einer relativen Position ein
shape = builder.InsertShape(ShapeType.TextBox, 50, 50); 
// Dreht die zweite Form um 30 Grad
shape.Rotation = 30.0;

//ExStep:3-
// Bereitet Optionen für das Dokument zu speichern
OoxmlSaveOptions saveOptions = new OoxmlSaveOptions(SaveFormat.Docx)
{
    Compliance = OoxmlCompliance.Iso29500_2008_Transitional // Sets compliance to ISO 29500:2008 Transitional
};

// Speichert das Dokument mit den angegebenen Speicheroptionen
doc.Save("WorkingWithShapes.InsertShape.docx", saveOptions); 
//ExEnd
