//ExStepSummary:0: Der folgende Code initialisiert ein neues Dokument und DocumentBuilder.
//ExStepSummary:1: Der folgende Code fügt ein Bild in das Dokument ein.

//ExStep:0-
// Eine neue Dokument-Instanz erstellen
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Bild aus dem angegebenen Pfad einfügen
builder.InsertImage(ImagesDir + "Transparent background logo.png", 
    RelativeHorizontalPosition.Margin,
    100, 
    RelativeVerticalPosition.Margin, 
    100, 
    200, 
    100, 
    WrapType.Square); 

//ExStep:2-
// Das Dokument in der angegebenen Datei speichern
doc.Save("AddContentUsingDocumentBuilder.InsertFloatingImage.docx");
