//ExStart
//ExSummary: Dieser Code zeigt, wie man mit Aspose.Words ein Bubble-Diagramm in ein Word- Dokument einfügt.
//ExStepSummary:0: Der folgende Code initialisiert ein neues Dokument und DocumentBuilder.
//ExStepSummary:1: Mit dem DocumentBuilder wird ein Bubble-Diagramm erstellt und dem Dokument hinzugefügt.
//ExStepSummary:2: Die Datenreihe für das Bubble-Diagramm wird mit spezifischen X-, Y- und Größenwerten hinzugefügt.
//ExStepSummary:3: Schließlich wird das Dokument mit dem eingefügten Bubble-Diagramm gespeichert.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Initialisieren eines neuen Dokuments und DocumentBuilders
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Fügen Sie ein Bubble-Diagramm in das Dokument ein
Shape shape = builder.InsertChart(ChartType.Bubble, 432, 252);

//ExStep:2-
// Datenreihen zum Bubble-Diagramm hinzufügen
Chart chart = shape.Chart;
chart.Series.Add("Aspose Series 1", new double[] { 0.7, 1.8, 2.6 }, // X values
    new double[] { 2.7, 3.2, 0.8 }, // Y values
    new double[] { 10, 4, 8 }); // Size values for the bubbles

//ExStep:3-
// Speichern Sie das Dokument mit dem Bubble-Diagramm
doc.Save("WorkingWithCharts.InsertBubbleChart.docx");
//ExEnd
