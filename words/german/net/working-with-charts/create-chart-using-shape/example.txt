//ExStep:0-
// Ein neues Dokumentobjekt erstellen
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Einfügen eines Liniendiagramms in das Dokument bei vorgegebenen Abmessungen
Shape shape = builder.InsertChart(ChartType.Line, 432, 252);

//ExStep:2-
// Zugriff auf das Chart-Objekt aus der Form
Chart chart = shape.Chart;
// Die Sichtbarkeit des Diagrammtitels aktivieren
chart.Title.Show = true;
// Stellen Sie den Titeltext für das Diagramm
chart.Title.Text = "Line Chart Title";
// Stellen Sie sicher, dass der Titel das Diagramm nicht überlagert
chart.Title.Overlay = false;

//ExStep:3-
// Legen Sie die Position der Chartlegende nach links
chart.Legend.Position = LegendPosition.Left;
chart.Legend.Overlay = true;

//ExStep:4-
// Das Dokument mit dem Diagramm in eine Datei speichern
doc.Save("WorkingWithCharts.CreateChartUsingShape.docx");
//ExEnd
