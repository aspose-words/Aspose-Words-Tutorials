//ExStepSummary:0: Initialisieren Sie das Dokument und DocumentBuilder, um ein neues Word- Dokument zu erstellen.
//ExStepSummary:1: Starten Sie einen neuen Tisch und legen Sie die Grenzen für den gesamten Tisch fest.
//ExStepSummary:2: Fügen Sie die erste Zelle mit roten Hintergrund-Schatten und Text hinzu.
//ExStepSummary:3: Fügen Sie die zweite Zelle mit grüner Hintergrund-Schattierung und Text hinzu.
//ExStepSummary:4: Beenden Sie die erste Zeile und löschen Sie die vorherige Zellenformatierung.
//ExStepSummary:5: Fügen Sie eine neue Zeile mit benutzerdefinierten größeren Grenzen für die erste Zelle hinzu.
//ExStepSummary:6: Fügen Sie die zweite Zelle der neuen Zeile mit Standardformatierung und Text hinzu.

//ExStep:0-
// Erstellen Sie ein neues Dokument.
Document doc = new Document();
// Erstellen Sie einen DocumentBuilder, um das Dokument zu manipulieren.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Starten Sie einen neuen Tisch.
Table table = builder.StartTable();
// Legen Sie die erste Zelle in die Tabelle ein.
builder.InsertCell();
// Stellen Sie die Grenzen für den gesamten Tisch.
table.SetBorders(LineStyle.Single, 2.0, Color.Black); 

//ExStep:2-
// Setzen Sie die Hintergrundfarbe der Zelle auf rot.
builder.CellFormat.Shading.BackgroundPatternColor = Color.Red;
builder.Writeln("Cell #1");
// Legen Sie die zweite Zelle ein.
builder.InsertCell();

//ExStep:3-
// Setzen Sie die Hintergrundfarbe der Zelle auf grün.
builder.CellFormat.Shading.BackgroundPatternColor = Color.Green;
builder.Writeln("Cell #2");
// Beenden Sie die aktuelle Zeile.
builder.EndRow();

//ExStep:4-
// Löschen Sie jede Zelle Formatierung, um neu zu starten.
builder.CellFormat.ClearFormatting();
// Fügen Sie eine neue Zelle in der nächsten Zeile ein.
builder.InsertCell();
// Erstellen Sie größere Ränder für die erste Zelle dieser Zeile.
builder.CellFormat.Borders.Left.LineWidth = 4.0;
builder.CellFormat.Borders.Right.LineWidth = 4.0;
builder.CellFormat.Borders.Top.LineWidth = 4.0;
builder.CellFormat.Borders.Bottom.LineWidth = 4.0;
builder.Writeln("Cell #3");
//ExStep:5-
// Legen Sie die vierte Zelle ein.
builder.InsertCell();
builder.CellFormat.ClearFormatting();
builder.Writeln("Cell #4");
//ExStep:6-
// Speichern Sie das Dokument.
doc.Save("WorkingWithTableStylesAndFormatting.FormatTableAndCellWithDifferentBorders.docx"); 
//ExEnd
