//ExSummary: Dieser Code zeigt, wie man Zeilenformatierung in einem Word-Dokument mit Aspose.Words anwendet.
//ExStepSummary:0: Initialisiert ein neues Dokument und beginnt mit dem Aufbau einer Tabelle mit formatierten Zeilen.
//ExStepSummary:1: Konfiguriert das Zeilenformat.
//ExStepSummary:2: Konfiguriert die spezifischen Höheneinstellungen.
//ExStepSummary:3: Setzt die Polsterung für die gesamte Tabelle, die alle Zeilen betrifft.
//ExStepSummary:4: Schreibt Text in die Tabellenzelle und beendet die Zeile.
//ExStepSummary:5: Ergänzt den Tabellenaufbau und speichert das Dokument.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStepImage:4:images/4.png
//ExStep:0-
// Ein neues Dokument erstellen
Document doc = new Document();
// Erstellen eines DocumentBuilders zur Änderung des Dokuments
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Eine neue Tabelle starten
Table table = builder.StartTable();
// Eine neue Zelle in die Tabelle einfügen
builder.InsertCell();
//ExStep:2-
// Zugriff auf die Zeilenformatierungseigenschaften
RowFormat rowFormat = builder.RowFormat;
// Die Höhe der Zeile einstellen
rowFormat.Height = 100;
// Geben Sie an, dass die Höhe genau ist
rowFormat.HeightRule = HeightRule.Exactly;

//ExStep:3-
// Polsterung für den Tisch einstellen
table.LeftPadding = 30; 
table.RightPadding = 30;
table.TopPadding = 30;
table.BottomPadding = 30;

//ExStep:4-
// Text in die Zelle schreiben
builder.Writeln("I'm a wonderful formatted row.");
// Beenden Sie die aktuelle Zeile
builder.EndRow();
// Beenden Sie das Tischgebäude
builder.EndTable();

//ExStep:5-
// Dokument speichern
doc.Save("WorkingWithTableStylesAndFormatting.ApplyRowFormatting.docx");
//ExEnd
