//ExStart
//ExSummary: Der folgende Code zeigt, wie man eine Tabelle in einem Word-Dokument mit der DocumentBuilder-Klasse von Aspose.Words erstellt und formatiert.
//ExStepSummary:0: Initialisieren Sie ein neues Dokument und DocumentBuilder, um mit dem Bau des Word-Dokuments zu beginnen.
//ExStepSummary:1: Erstellen Sie eine Tabelle und fügen Sie eine Zelle für die Formatierung ein.
//ExStepSummary:2: Legen Sie die Formatierungseigenschaften für die Zelle, einschließlich Polsterung und Breite, fest.
//ExStepSummary:3: Text in die formatierte Zelle einfügen.
//ExStepSummary:4: Beenden Sie die Zeile und Tabelle und speichern Sie das Dokument in einem angegebenen Dateipfad.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0-
// Ein neues Dokumentobjekt erstellen
Document doc = new Document();
// Initialisieren von DocumentBuilder mit dem neuen Dokument
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Beginnen Sie mit der Erstellung einer Tabelle
builder.StartTable();
// Die erste Zelle in die Tabelle einfügen
builder.InsertCell();

//ExStep:2-
// Zugriff auf die Formatierungseigenschaften der Zelle
CellFormat cellFormat = builder.CellFormat;
// Einstellung der Zellbreite
cellFormat.Width = 250;
// Die Polsterung einstellen
cellFormat.LeftPadding = 30;
cellFormat.RightPadding = 30;
cellFormat.TopPadding = 30;
cellFormat.BottomPadding = 30;

//ExStep:3-
// Text in die Zelle einfügen
builder.Writeln("I'm a wonderful formatted cell.");

//ExStep:4-
// Beenden der aktuellen Zeile der Tabelle
builder.EndRow();
// Ende der Tabellenerstellung
builder.EndTable();
// Das Dokument speichern
doc.Save("WorkingWithTableStylesAndFormatting.DocumentBuilderSetTableCellFormatting.docx");
//ExEnd
