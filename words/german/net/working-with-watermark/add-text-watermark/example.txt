//ExSummary: Dieser Code zeigt, wie man mit Aspose.Words ein Text-Wasserzeichen mit bestimmten Optionen zu einem Word-Dokument hinzufügt.
//ExStepSummary:0: Der folgende Code initialisiert den Dokumentpfad und erzeugt ein neues Dokumentobjekt.
//ExStepSummary:1: Der folgende Code legt die Optionen für das Text-Wasserzeichen fest, einschließlich Schriftfamilie, Größe, Farbe und Layout.
//ExStepSummary:2: Der folgende Code wendet das Text-Wasserzeichen mit den angegebenen Optionen auf das Dokument an.
//ExStepSummary:3: Der folgende Code speichert das geänderte Dokument in einem angegebenen Verzeichnis.
//ExStepImage:2:images/1.png
//ExStepImage:3:images/2.png
//ExStep:0-
// Stellen Sie den Verzeichnispfad für das Dokument
string dataDir = "Your Document Directory";

//ExStep:1-
// Das Word-Dokument laden
Document doc = new Document(dataDir + "Document.docx");

//ExStep:2-
// Wasserzeichen-Optionen erstellen
TextWatermarkOptions options = new TextWatermarkOptions()
{
    FontFamily = "Arial", // Set font family for the watermark text
    FontSize = 36, // Set font size for the watermark text
    Color = Color.Black, // Set color of the watermark text
    Layout = WatermarkLayout.Horizontal, // Specify layout of the watermark
    IsSemitrasparent = false // Set transparency option
};

//ExStep:3-
// Den Wasserzeichentext mit Optionen anwenden
doc.Watermark.SetText("Test", options);

//ExStep:4-
// Das Dokument mit dem Wasserzeichen speichern
doc.Save(dataDir + "WorkWithWatermark.AddTextWatermarkWithSpecificOptions.docx");
//ExEnd
