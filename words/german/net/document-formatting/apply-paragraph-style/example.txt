//ExStepSummary:0: Dieser Abschnitt initialisiert eine neue Document- und DocumentBuilder- Instanz.
//ExStepSummary:1: Dieser Abschnitt wendet einen Titel Absatzstil mit dem DocumentBuilder an.
//ExStepSummary:2: Dieser Abschnitt schreibt "Hallo" an das Dokument.
//ExStepSummary:3: Dieser Abschnitt speichert das Dokument in einem angegebenen Dateipfad.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Initialisieren einer neuen Dokument-Instanz
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Den Absatzstil auf Titel setzen
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Title;

//ExStep:2-
// Schreiben Sie "Hallo" mit dem Builder, unter Anwendung der angegebenen Stil
builder.Write("Hello");

//ExStep:3-
// Speichern Sie das Dokument in eine .docx-Datei
doc.Save("DocumentFormatting.ApplyParagraphStyle.docx");
