//ExStart
//ExSummary: Dieser Code zeigt, wie man mit Aspose.Words Grenzen und Schattierungen auf einen Absatz in einem Word-Dokument anwendet.
//ExStepSummary:0: Der folgende Code initialisiert eine neue Document- und DocumentBuilder- Instanz.
//ExStepSummary:1: Der folgende Code konfiguriert die Absatzgrenzen mit einem Abstand vom Text.
//ExStepSummary:2: Der folgende Code bezieht eine diagonale Kreuzstruktur-Schattierung auf den Absatz mit spezifizierten Farben.
//ExStepSummary:3: Der folgende Code schreibt einen formatierten Absatz mit den angewendeten Einstellungen.
//ExStepSummary:4: Der folgende Code speichert das Dokument auf den angegebenen Ausgabepfad.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0-
// Initialisieren eines neuen Dokuments
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Zugang zu den Grenzen des Absatzes
BorderCollection borders = builder.ParagraphFormat.Borders;
borders.DistanceFromText = 20;
borders[BorderType.Left].LineStyle = LineStyle.Double;
borders[BorderType.Right].LineStyle = LineStyle.Double;
borders[BorderType.Top].LineStyle = LineStyle.Double;
borders[BorderType.Bottom].LineStyle = LineStyle.Double;

//ExStep:2-
// Zugriff auf die Absatz-Schatteneigenschaften
Shading shading = builder.ParagraphFormat.Shading;
shading.Texture = TextureIndex.TextureDiagonalCross;
shading.BackgroundPatternColor = System.Drawing.Color.LightCoral; 
shading.ForegroundPatternColor = System.Drawing.Color.LightSalmon;

//ExStep:3-
// Schreiben Sie den formatierten Absatzinhalt
builder.Write("I'm a formatted paragraph with double border and nice shading.");

//ExStep:4-
// Das Dokument in den angegebenen Pfad speichern
doc.Save(ArtifactsDir + "DocumentFormatting.ApplyBordersAndShadingToParagraph.doc");
//ExEnd
