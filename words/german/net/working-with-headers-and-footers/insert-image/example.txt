//ExStart
//ExSummary: Dieser Code zeigt, wie man ein Bild mit Aspose.Words für .NET in den Header eines Word-Dokuments einfügt.
//ExStepSummary:0: Erstellen Sie ein neues Word-Dokument mit der Klasse Aspose.Words Document.
//ExStepSummary:1: Wechseln Sie mit dem DocumentBuilder in den primären Header des Dokuments.
//ExStepSummary:2: Fügen Sie ein Bild in den Header mit spezifischen Ausrichtungs-, Abmessungen- und Umhüllungsoptionen ein.
//ExStepSummary:3: Speichern Sie das Dokument in einen bestimmten Dateipfad.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Ein neues Word-Dokument erstellen
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Bewegen Sie den Cursor in den primären Kopf des Dokuments
builder.MoveToHeaderFooter(HeaderFooterType.HeaderPrimary);

//ExStep:2-
// Ein Bild in den Header mit spezifischer Ausrichtung und Abmessungen einfügen
builder.InsertImage(
    ImagesDir + "Logo.jpg", RelativeHorizontalPosition.RightMargin, 10,
    RelativeVerticalPosition.Page, 10, 50, 50, 
    WrapType.Through
);

//ExStep:3-
// Das Dokument in den angegebenen Pfad speichern
doc.Save("WorkingWithHeadersAndFooters.InsertImage.docx");
//ExEnd
