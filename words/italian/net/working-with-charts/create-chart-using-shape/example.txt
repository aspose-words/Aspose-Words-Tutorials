// Crea un nuovo oggetto Documento
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Inserisci un grafico a righe nel documento alle dimensioni specificate
Shape shape = builder.InsertChart(ChartType.Line, 432, 252);

//ExStep:2-
// Accedere all'oggetto Grafico dalla forma
Chart chart = shape.Chart;
// Abilita la visibilità del titolo del grafico
chart.Title.Show = true;
// Imposta il testo del titolo per il grafico
chart.Title.Text = "Line Chart Title";
// Assicurati che il titolo non sovrapponga il grafico
chart.Title.Overlay = false;

//ExStep:3-
// Imposta la posizione della legenda del grafico a sinistra
chart.Legend.Position = LegendPosition.Left;
chart.Legend.Overlay = true;

//ExStep:4-
// Salva il documento con il grafico in un file
doc.Save("WorkingWithCharts.CreateChartUsingShape.docx");
