//ExSummary: Questo codice dimostra come inserire un grafico Bubble in un documento Word usando Aspose.Words.
//ExStepSummary:0: Il seguente codice inizializza un nuovo documento e DocumentBuilder.
//ExStepSummary:1: Un grafico a bolle viene creato e aggiunto al documento usando DocumentBuilder.
//ExStepSummary:2: La serie di dati per il grafico Bubble Ã¨ aggiunta con valori specifici di X, Y e dimensione.
//ExStepSummary:3: Infine, il documento viene salvato con il grafico Bubble inserito.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Inizializzare un nuovo documento e costruire un documento
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Inserisci un grafico a bolle nel documento
Shape shape = builder.InsertChart(ChartType.Bubble, 432, 252);

//ExStep:2-
// Aggiungi serie di dati al grafico Bubble
Chart chart = shape.Chart;
chart.Series.Add("Aspose Series 1", new double[] { 0.7, 1.8, 2.6 }, // X values
    new double[] { 2.7, 3.2, 0.8 }, // Y values
    new double[] { 10, 4, 8 }); // Size values for the bubbles

//ExStep:3-
// Salva il documento con il grafico Bubble
doc.Save("WorkingWithCharts.InsertBubbleChart.docx");
//ExEnd
