//ExStepSummary:2: I seguenti trattini di codice per creare sottovoci sotto la seconda voce di primo livello.

//ExStepSummary:4: Il seguente codice Ã¨ in ritardo rispetto al livello precedente per aggiungere un altro elemento al secondo livello.

// Inizializza un nuovo documento
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Applica il formato predefinito di numerazione
builder.ListFormat.ApplyNumberDefault();
builder.Writeln("Item 1");
builder.Writeln("Item 2");

//ExStep:2-
// Indentazione delle sottovoci
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.1");
builder.Writeln("Item 2.2");

//ExStep:3-
// Indentazione per sotto-sotto-voci
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.2.1");
builder.Writeln("Item 2.2.2");

//ExStep:4-
// Outdent torna al secondo livello
builder.ListFormat.ListOutdent();
builder.Writeln("Item 2.3");

//ExStep:5-
// Outdent back to the top level
builder.ListFormat.ListOutdent();
builder.Writeln("Item 3");

//ExStep:6-
// Rimuove la numerazione dall'elenco
builder.ListFormat.RemoveNumbers();

//ExStep:7-
// Salva il documento
doc.Save("DocumentFormatting.MultilevelListFormatting.docx");
