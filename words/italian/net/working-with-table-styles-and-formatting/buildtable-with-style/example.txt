//ExSummary: Questo codice dimostra come creare un documento di Word con una tabella in stile utilizzando Aspose.Words. La tabella include intestazioni e righe di dati, formattati in base a specifiche opzioni di stile.
//ExStepSummary:0: Inizializzare un nuovo documento e DocumentBuilder per la gestione della struttura del documento.
//ExStepSummary:1: Avvia una nuova tabella nel documento usando DocumentBuilder.
//ExStepSummary:2: Inserisci la prima cella e imposta l'identificatore dello stile della tabella.
//ExStepSummary:3: Applicare le opzioni di formattazione alla tabella e impostare il comportamento auto-fit.
//ExStepSummary:4: Aggiungere le celle di riga di intestazione al tavolo.
//ExStepSummary:5: Inserire righe di dati nella tabella con i nomi degli elementi e le quantit√†.
//ExStepSummary:6: Salva il documento assemblato in un file.
//ExStepImage:2:images/1.png
//ExStepImage:3:images/2.png

//ExStep:0-
// Crea un nuovo oggetto Documento.
Document doc = new Document();
// Creare un costruttore per facilitare la generazione di documenti.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Avvia una nuova tabella nel documento.
Table table = builder.StartTable();

//ExStep:2-
// Inserire la prima cella prima di impostare la formattazione.
builder.InsertCell();
// Imposta l' identificatore di stile per la tabella.
table.StyleIdentifier = StyleIdentifier.MediumShading1Accent1;


//ExStep:3-
// Configura le opzioni di stile.
table.StyleOptions = 
    TableStyleOptions.FirstColumn | TableStyleOptions.RowBands | TableStyleOptions.FirstRow;
// Impostare il comportamento di auto-fit della tabella.	
table.AutoFit(AutoFitBehavior.AutoFitToContents);


//ExStep:4-
// Scrivi l'intestazione per la prima colonna.
builder.Writeln("Item");
// Impostare l'imbottitura destra per la cella.
builder.CellFormat.RightPadding = 40;
// Inserisci la seconda cella per la colonna successiva.
builder.InsertCell();
// Scrivi l'intestazione per la seconda colonna.
builder.Writeln("Quantity (kg)");
// Termina la creazione della riga.
builder.EndRow();

//ExStep:5-
// Avviare una nuova riga e inserire la prima cella.
builder.InsertCell();
builder.Writeln("Apples");
builder.InsertCell();
builder.Writeln("20");
builder.EndRow();
 // Inserire la prima cella di una nuova riga per il secondo elemento.
builder.InsertCell();
builder.Writeln("Bananas");
builder.InsertCell();
builder.Writeln("40");
builder.EndRow();
// Iniziare una nuova riga per il terzo elemento.
builder.InsertCell();
builder.Writeln("Carrots");
builder.InsertCell(); 
builder.Writeln("50");
builder.EndRow();

//ExStep:6-
// Salva il documento in un file specificato.
doc.Save("WorkingWithTableStylesAndFormatting.BuildTableWithStyle.docx");
