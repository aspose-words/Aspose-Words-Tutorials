//ExSummary: Questo codice dimostra come creare un documento Word e inserire un'immagine con un layout fluttuante utilizzando Aspose.Words.
//ExStepSummary:0: Il seguente codice inizializza un nuovo documento e DocumentBuilder.
//ExStepSummary:1: Il seguente codice inserisce un'immagine nel documento.
//ExStepSummary:2: Il seguente codice salva il documento in una posizione specificata.
//ExStepImage:1:images/1.png
//ExStep:0-
// Crea una nuova istanza di documento
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Inserire l'immagine dal percorso specificato
builder.InsertImage(ImagesDir + "Transparent background logo.png", 
    RelativeHorizontalPosition.Margin,
    100, 
    RelativeVerticalPosition.Margin, 
    100, 
    200, 
    100, 
    WrapType.Square); 

//ExStep:2-
// Salva il documento nel file specificato
doc.Save("AddContentUsingDocumentBuilder.InsertFloatingImage.docx");
//ExEnd
