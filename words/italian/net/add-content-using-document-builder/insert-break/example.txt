// Crea una nuova istanza di documento
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Aggiungi testo per la prima pagina
builder.Writeln("This is page 1.");

//ExStep:2-
// Inserisci una interruzione di pagina per avviare una nuova pagina
builder.InsertBreak(BreakType.PageBreak);

//ExStep:3-
// Aggiungi testo per la seconda pagina
builder.Writeln("This is page 2.");

//ExStep:4-
// Inserisci un'altra pausa pagina per una nuova pagina
builder.InsertBreak(BreakType.PageBreak);

//ExStep:5-
// Aggiungi testo per la terza pagina
builder.Writeln("This is page 3.");

//ExStep:6-
// Salva il documento nel percorso specificato
doc.Save("AddContentUsingDocumentBuilder.InsertBreak.docx");
