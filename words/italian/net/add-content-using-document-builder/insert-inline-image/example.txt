//ExSummary:Questo codice dimostra come inserire un'immagine in linea in un documento di Word usando Aspose.Words.
//ExStepSummary:0: Inizializzazione dell'oggetto Documento e del DocumentBuilder per la costruzione del documento.
//ExStepSummary:1: Inserimento di un'immagine nel documento nella posizione attuale del cursore con DocumentBuilder.
//ExStepSummary:2: Salvataggio del documento in un nome di file specificato.
//ExStepImage:1:images/1.png
//ExStep:0-
// Inizializzare un nuovo documento e costruire un documento
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Inserisci l'immagine usando DocumentBuilder
builder.InsertImage(ImagesDir + "Transparent background logo.png");

//ExStep:2-
// Salva il documento modificato in un file
doc.Save("AddContentUsingDocumentBuilder.InsertInlineImage.docx");
