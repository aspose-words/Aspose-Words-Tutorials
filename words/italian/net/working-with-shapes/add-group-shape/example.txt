//ExStepSummary:0: Inizializza un nuovo documento e garantisce il rispetto dei requisiti minimi.
//ExStepSummary:1: Crea una forma di gruppo che terr√† altre forme.
//ExStepSummary:2: Aggiunge una forma di callout al bordo dell'accento alla forma del gruppo.
//ExStepSummary:3: Aggiunge una forma pulsante azione alla forma di gruppo.
//ExStepSummary:4: Imposta le dimensioni e le coordinate per la forma del gruppo.
//ExStepSummary:5: Inserisce la forma del gruppo nel documento usando un DocumentBuilder.
//ExStepSummary:6: Salva il documento contenente la forma del gruppo in una posizione specificata.

//ExStep:0-
// Crea un nuovo documento
Document doc = new Document();
// Assicurare che il documento soddisfi i requisiti minimi
doc.EnsureMinimum();

//ExStep:1-
// Crea una nuova istanza di GroupShape
GroupShape groupShape = new GroupShape(doc);

//ExStep:2-
// Crea una forma di callout al bordo dell'accento
Shape accentBorderShape = new Shape(doc, ShapeType.AccentBorderCallout1) 
{
    Width = 100, 
    Height = 100
};
// Aggiungi la forma alla forma del gruppo
groupShape.AppendChild(accentBorderShape);

//ExStep:3-
// Crea una forma pulsante di azione
Shape actionButtonShape = new Shape(doc, ShapeType.ActionButtonBeginning)
{
	// Imposta la posizione sinistra della forma
    Left = 100, 
	// Imposta la larghezza della forma
    Width = 100, 
	// Imposta l'altezza della forma
    Height = 200
};
// Aggiungi il pulsante azione alla forma del gruppo
groupShape.AppendChild(actionButtonShape);

//ExStep:4-
// Imposta la larghezza della forma del gruppo
groupShape.Width = 200;
// Imposta l'altezza della forma del gruppo
groupShape.Height = 200;
// Imposta la dimensione delle coordinate per la forma del gruppo
groupShape.CoordSize = new Size(200, 200);

//ExStep:5-
// Crea un documentoCostruisci per manipolare il documento
DocumentBuilder builder = new DocumentBuilder(doc);
// Inserisci la forma del gruppo nel documento
builder.InsertNode(groupShape);

//ExStep:6-
// Salva il documento in una posizione specificata
doc.Save("WorkingWithShapes.AddGroupShape.docx");
//ExEnd
