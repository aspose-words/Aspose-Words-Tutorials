//ExSummaryEste código demonstra como criar um documento do Word com uma tabela que tem diferentes estilos de borda e sombreamento para suas células.
//ExStepSummary:0: Inicialize o Documento e o DocumentBuilder para criar um novo documento do Word.
//ExStepSummary:1: Inicie uma nova mesa e defina as fronteiras para toda a mesa.
//ExStepSummary:2: Adicione a primeira célula com sombreamento de fundo vermelho e texto.
//ExStepSummary:3: Adicione a segunda célula com sombreamento de fundo verde e texto.
//ExStepSummary:4: Terminar a primeira linha e limpar a formatação celular anterior.
//ExStepSummary:5: Adicione uma nova linha com bordas maiores personalizadas para a primeira célula.
//ExStepSummary:6: Adicione a segunda célula da nova linha com formatação padrão e texto.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStepImage:4:images/4.png
//ExStepImage:5:images/5.png

//ExStep:0-
// Crie um novo documento.
Document doc = new Document();
// Crie um DocumentBuilder para manipular o documento.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Comece uma nova mesa.
Table table = builder.StartTable();
// Insira a primeira célula na tabela.
builder.InsertCell();
// Defina as fronteiras para toda a mesa.
table.SetBorders(LineStyle.Single, 2.0, Color.Black); 

//ExStep:2-
// Definir a cor de fundo da célula para vermelho.
builder.CellFormat.Shading.BackgroundPatternColor = Color.Red;
builder.Writeln("Cell #1");
// Insira a segunda célula.
builder.InsertCell();

//ExStep:3-
// Definir a cor de fundo da célula para verde.
builder.CellFormat.Shading.BackgroundPatternColor = Color.Green;
builder.Writeln("Cell #2");
// Fim da linha atual.
builder.EndRow();

//ExStep:4-
// Limpe qualquer formatação de célula para começar de novo.
builder.CellFormat.ClearFormatting();
// Insira uma nova célula na próxima linha.
builder.InsertCell();
// Crie fronteiras maiores para a primeira célula desta linha.
builder.CellFormat.Borders.Left.LineWidth = 4.0;
builder.CellFormat.Borders.Right.LineWidth = 4.0;
builder.CellFormat.Borders.Top.LineWidth = 4.0;
builder.CellFormat.Borders.Bottom.LineWidth = 4.0;
builder.Writeln("Cell #3");
//ExStep:5-
// Inserir a quarta célula.
builder.InsertCell();
builder.CellFormat.ClearFormatting();
builder.Writeln("Cell #4");
//ExStep:6-
// Salve o documento.
doc.Save("WorkingWithTableStylesAndFormatting.FormatTableAndCellWithDifferentBorders.docx"); 
//ExEnd
