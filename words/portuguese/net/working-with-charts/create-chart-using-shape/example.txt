//ExStep:0-
// Criar um novo objeto Documento
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Insira um gráfico de linha no documento em dimensões especificadas
Shape shape = builder.InsertChart(ChartType.Line, 432, 252);

//ExStep:2-
// Acessar o objeto Gráfico a partir da forma
Chart chart = shape.Chart;
// Ativar a visibilidade do título do gráfico
chart.Title.Show = true;
// Definir o texto do título para o gráfico
chart.Title.Text = "Line Chart Title";
// Certifique-se de que o título não sobreponha o gráfico
chart.Title.Overlay = false;

//ExStep:3-
// Definir a posição da legenda do gráfico à esquerda
chart.Legend.Position = LegendPosition.Left;
chart.Legend.Overlay = true;

//ExStep:4-
// Salvar o documento com o gráfico em um arquivo
doc.Save("WorkingWithCharts.CreateChartUsingShape.docx");
