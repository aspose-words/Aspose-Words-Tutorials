//ExSummary: Este código demonstra como criar um documento do Word com um gráfico de dispersão usando Aspose.Words. Ele inicializa um documento, constrói conteúdo, insere um gráfico de dispersão, adiciona dados a ele e salva o documento.
//ExStepSummary:0: Inicialização dos objetos Document e DocumentBuilder para começar a trabalhar no documento.
//ExStepSummary:1: Inserção de um gráfico de dispersão no documento.
//ExStepSummary:2: Adicionando uma série de pontos de dados ao gráfico de dispersão.
//ExStepSummary:3: Salvando o documento com o gráfico de dispersão em um arquivo .docx.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Inicializar um novo documento
Document doc = new Document();
// Criar um construtor para inserir conteúdo
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Insira um gráfico de dispersão no documento
Shape shape = builder.InsertChart(ChartType.Scatter, 432, 252); 

//ExStep:2-
// Obter o objeto gráfico da forma inserida
Chart chart = shape.Chart;
// Adicionar dados da série ao gráfico
chart.Series.Add("Aspose Series 1", new double[] { 0.7, 1.8, 2.6 }, new double[] { 2.7, 3.2, 0.8 });

//ExStep:3-
// Salve o documento com o gráfico de dispersão
doc.Save("WorkingWithCharts.InsertScatterChart.docx");
//ExEnd
