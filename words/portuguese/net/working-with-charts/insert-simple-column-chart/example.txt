//ExStepSummary:4: Este código mostra como acessar a coleção de séries de gráficos e imprimir a contagem de séries.

//ExStep:0-
// Crie uma nova instância de documento do Word.
Document doc = new Document();

//ExStep:1-
// Crie um DocumentBuilder para inserir facilmente elementos no documento.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:2-
// Insira uma forma de gráfico de coluna com largura especificada (432) e altura (252).
Shape shape = builder.InsertChart(ChartType.Column, 432, 252);

//ExStep:3-
// Recupere o objeto do gráfico da forma para posterior manipulação.
Chart chart = shape.Chart;

//ExStep:4-
// Acesse a coleção de séries do gráfico.
ChartSeriesCollection seriesColl = chart.Series;
// Imprima o número de séries no gráfico para o console.
Console.WriteLine(seriesColl.Count);

//ExStep:5-
// Remova qualquer série padrão que foi gerada automaticamente quando o gráfico foi criado.
seriesColl.Clear();

//ExStep:6-
// Defina uma matriz de nomes de categorias para dados de gráficos.
string[] categories = new string[] { "Category 1", "Category 2" };

//ExStep:7-
// Adicione várias séries ao gráfico com nomes de categorias e valores de dados.
seriesColl.Add("Aspose Series 1", categories, new double[] { 1, 2 });
seriesColl.Add("Aspose Series 2", categories, new double[] { 3, 4 });
seriesColl.Add("Aspose Series 3", categories, new double[] { 5, 6 });
seriesColl.Add("Aspose Series 4", categories, new double[] { 7, 8 });
seriesColl.Add("Aspose Series 5", categories, new double[] { 9, 10 });

//ExStep:8-
// Salve o documento com o gráfico inserido no caminho do arquivo especificado.
doc.Save("WorkingWithCharts.InsertSimpleColumnChart.docx");
