//ExSummary: Este código demonstra como criar um documento do Word com uma tabela usando Aspose.Words. Ele constrói uma tabela com duas linhas e várias orientações e formatos de texto celular.
//ExStepSummary:0: O código a seguir inicializa um novo documento e construtor de documentos.
//ExStepSummary:1: O código a seguir inicia uma nova tabela e insere a primeira linha de células.
//ExStepSummary:2: O código a seguir configura a formatação para a segunda linha e suas células.
//ExStepSummary:3: O código a seguir configura a formatação para a segunda linha e suas células.
//ExStepSummary:4: O código a seguir termina a tabela e salva o documento em um arquivo.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png

//ExStep:0-
// Inicializar um novo objeto Documento
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Iniciar uma nova mesa
Table table = builder.StartTable();
// Comece a primeira célula
builder.InsertCell();
// Definir alinhamento vertical para a primeira célula
builder.CellFormat.VerticalAlignment = CellVerticalAlignment.Center; 
builder.Write("This is row 1 cell 1");
// Inserir segunda célula
builder.InsertCell();
builder.Write("This is row 1 cell 2");
builder.EndRow();

//ExStep:2-
// Inserir uma célula para a segunda linha
builder.InsertCell();
// Configurar formatação para a segunda linha
builder.RowFormat.Height = 100;
builder.RowFormat.HeightRule = HeightRule.Exactly;
builder.CellFormat.Orientation = TextOrientation.Upward;
builder.Writeln("This is row 2 cell 1");
//ExStep:3-
// Inserir segunda célula para a segunda linha
builder.InsertCell();
builder.CellFormat.Orientation = TextOrientation.Downward; 
builder.Writeln("This is row 2 cell 2");
builder.EndRow();
builder.EndTable();

//ExStep:4-
// Ajuste automático da tabela com larguras de coluna fixas
table.AutoFit(AutoFitBehavior.FixedColumnWidths);
// Salvar o documento em um arquivo
doc.Save("AddContentUsingDocumentBuilder.BuildTable.docx");
//ExEnd
