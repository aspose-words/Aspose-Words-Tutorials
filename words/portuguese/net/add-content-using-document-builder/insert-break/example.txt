//ExStepSummary:0: Esta etapa inicializa um novo Documento e um DocumentBuilder para manipular o documento.
//ExStepSummary:1: Esta etapa adiciona texto à primeira página.
//ExStepSummary:2: Esta etapa insere uma quebra de página para iniciar uma nova página.
//ExStepSummary:3: Esta etapa adiciona texto à segunda página.
//ExStepSummary:4: Esta etapa insere outra quebra de página para uma nova página.
//ExStepSummary:5: Esta etapa adiciona texto à terceira página.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStepImage:4:images/4.png
//ExStepImage:5:images/5.png

//ExStep:0-
// Criar uma nova instância de documento
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Adicionar texto para a primeira página
builder.Writeln("This is page 1.");

//ExStep:2-
// Inserir uma quebra de página para iniciar uma nova página
builder.InsertBreak(BreakType.PageBreak);

//ExStep:3-
// Adicionar texto para a segunda página
builder.Writeln("This is page 2.");

//ExStep:4-
// Inserir outra quebra de página para uma nova página
builder.InsertBreak(BreakType.PageBreak);

//ExStep:5-
// Adicionar texto para a terceira página
builder.Writeln("This is page 3.");

//ExStep:6-
// Salve o documento no caminho especificado
doc.Save("AddContentUsingDocumentBuilder.InsertBreak.docx");
//ExEnd
