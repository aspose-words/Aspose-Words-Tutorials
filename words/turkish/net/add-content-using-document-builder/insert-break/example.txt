//ExStepSummary:0: Bu adım, belgeyi manipüle etmek için yeni bir Belge ve bir Belge Oluşturucuyu başlatır.
//ExStepSummary:1: Bu adım ilk sayfaya metin ekler.
//ExStepSummary:2: Bu adım, yeni bir sayfa başlatmak için bir sayfa molası ekler.
//ExStepSummary:3: Bu adım ikinci sayfaya metin ekler.
//ExStepSummary:4: Bu adım, yeni bir sayfa için başka bir sayfa molası ekler.
//ExStepSummary:5: Bu adım üçüncü sayfaya metin ekler.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStepImage:4:images/4.png
//ExStepImage:5:images/5.png

//ExStep:0-
// Yeni bir Belge örneği oluştur
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// İlk sayfa için metin ekle
builder.Writeln("This is page 1.");

//ExStep:2-
// Yeni bir sayfa başlatmak için bir sayfa molası ekleyin
builder.InsertBreak(BreakType.PageBreak);

//ExStep:3-
// İkinci sayfa için metin ekle
builder.Writeln("This is page 2.");

//ExStep:4-
// Yeni bir sayfa için başka bir sayfa molası ekle
builder.InsertBreak(BreakType.PageBreak);

//ExStep:5-
// Üçüncü sayfa için metin ekle
builder.Writeln("This is page 3.");

//ExStep:6-
// Belgeyi belirtilen yola kaydet
doc.Save("AddContentUsingDocumentBuilder.InsertBreak.docx");
//ExEnd
