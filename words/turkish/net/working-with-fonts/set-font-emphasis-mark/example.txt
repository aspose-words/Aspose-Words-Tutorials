//ExSummary:Bu kod, Aspose.words kullanarak bir Word belgesinde vurgu işaretinin nasıl ayarlanacağını gösterir.
//ExStepSummary:0:Birlikte çalışmak için yeni bir Belge örneği oluşturma.
//ExStepSummary:1:Belgeyi manipüle etmek için bir DocumentBuilder kurmak.
//ExStepSummary:2:Yapıcının yazı tipine bir vurgu işareti uygulamak.
//ExStepSummary:3:Belgeye vurgu metni yazmak.
//ExStepSummary:4:Herhangi bir vurgu olmadan basit bir metin yazmak.
//ExStepSummary:5: Bu kod, yazı tipine vurgu işareti ile belgeyi kaydeder.
//ExStepImage:2:images/1.png
//ExStepImage:3:images/2.png
//ExStepImage:4:images/3.png
//ExStep:0-
// Yeni bir Belge Oluştur
Document document = new Document();

//ExStep:1-
// Belge manipülasyonu için DocumentBuilder'ı başlat
DocumentBuilder builder = new DocumentBuilder(document);

//ExStep:2-
// Vurgu işaretini UnderSolidCircle olarak ayarla
builder.Font.EmphasisMark = EmphasisMark.UnderSolidCircle;

//ExStep:3-
// Vurgu metnini belgeye yazın
builder.Write("Emphasis text");
// Bir sonraki satıra geç
builder.Writeln();
// Sonraki metin için yazı tipi biçimlendirmesini temizle
builder.Font.ClearFormatting();

//ExStep:4-
// Vurgulamadan basit metin yazın
builder.Write("Simple text");

//ExStep:5-
// Belgeyi belirtilen yola kaydet
document.Save("WorkingWithFonts.SetFontEmphasisMark.docx");
//ExEnd
