//ExSummary: Bu kod Aspose.Words kullanarak bir belgeye sütun grafiğinin nasıl yerleştirileceğini gösterir. Bir belge oluşturur,
//          Tanımlanmış kategoriler ve değerler içeren bir grafik ekler ve ardından belgeyi kaydeder.
//ExStepSummary:0: Aşağıdaki kod yeni bir belge ve belge oluşturucuyu başlatır.
//ExStepSummary:1: Bu adım, belgenin içine belirtilen boyutlara sahip bir sütun grafiğinin eklenmesini içerir.
//ExStepSummary:2: Burada, dizi verileri, kategoriler ve karşılık gelen değerlerle çizelgeye eklenir.
//ExStepSummary:3: Bu adım, eklenen grafikle belgeyi belirli bir dosya konumuna kaydeder.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Yeni bir Belge nesnesini başlat
Document doc = new Document();
// Kolay belge manipülasyonu için bir DocumentBuilder oluşturun
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Belirtilen boyuta bir sütun grafiği ekle (genişlik x yükseklik)
Shape shape = builder.InsertChart(ChartType.Column, 432, 252);

//ExStep:2-
// Girilen şekilden grafik nesnesine erişin
Chart chart = shape.Chart;
// Kategoriler ve değerlerle seri verileri ekle
chart.Series.Add("Aspose Series 1", new string[] { "Category 1", "Category 2" }, new double[] { 1, 2 });

//ExStep:3-
// Belgeyi eklenen sütun grafiğiyle kaydet
doc.Save("WorkingWithCharts.InsertColumnChart.docx"); 
//ExEnd
