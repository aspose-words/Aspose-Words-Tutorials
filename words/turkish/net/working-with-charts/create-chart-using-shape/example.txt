//ExStepSummary:0: Aşağıdaki kod, yeni bir Belgenin nasıl oluşturulacağını ve bir DocumentBuilder'ın nasıl hazırlanacağını gösterir:
//ExStepSummary:1: Aşağıdaki kod, belgeye bir Satır grafiğinin nasıl yerleştirileceğini gösterir:
//ExStepSummary:2: Aşağıdaki kod, grafik için başlığın nasıl ayarlanacağını açıklar:
//ExStepSummary:3: Aşağıdaki kod, grafik efsanesinin konumunu ve bindirme ayarlarının nasıl yapılandırılacağını gösterir:

//ExStep:0-
// Yeni bir Belge nesnesi oluştur
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Belgeye belirtilen boyutlarda bir Satır grafiği ekle
Shape shape = builder.InsertChart(ChartType.Line, 432, 252);

//ExStep:2-
// Grafik nesnesine formdan erişin
Chart chart = shape.Chart;
// Grafik başlığı görünürlüğünü etkinleştir
chart.Title.Show = true;
// Grafik için başlık metnini ayarla
chart.Title.Text = "Line Chart Title";
// Başlığın grafiği kaplamadığından emin olun
chart.Title.Overlay = false;

//ExStep:3-
// Grafik efsanesinin konumunu sola ayarla
chart.Legend.Position = LegendPosition.Left;
chart.Legend.Overlay = true;

//ExStep:4-
// Belgeyi grafikle bir dosyaya kaydet
doc.Save("WorkingWithCharts.CreateChartUsingShape.docx");
//ExEnd
