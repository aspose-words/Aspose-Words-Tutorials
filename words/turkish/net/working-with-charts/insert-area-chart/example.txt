//ExSummary: Bu kod, Aspose.Words kullanarak bir Word belgesine bir alan grafiğinin nasıl yerleştirileceğini gösterir.
//ExStepSummary:0: Aşağıdaki kod yeni bir belge ve belge oluşturucuyu başlatır.
//ExStepSummary:1: Aşağıdaki kod belgeye bir alan grafiği ekler.
//ExStepSummary:2: Aşağıdaki kod, belirtilen veri noktalarına sahip çizelgeye bir dizi ekler.
//ExStepSummary:3: Aşağıdaki kod, eklenen alan grafiği ile belgeyi kaydeder.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Yeni bir belge oluştur
Document doc = new Document();
// Belge oluşturucuyu başlat
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Belirtilen boyutlara sahip bir alan grafiği ekle
Shape shape = builder.InsertChart(ChartType.Area, 432, 252);

//ExStep:2-
// Grafik nesnesine eriş
Chart chart = shape.Chart;
// Çizelgeye yeni bir dizi ekle
chart.Series.Add("Aspose Series 1", new [] 
{
	// Veri noktası 1
    new DateTime(2002, 05, 01),
	// Veri noktası 2
    new DateTime(2002, 06, 01),
	// Veri noktası 3
    new DateTime(2002, 07, 01),
	// Veri noktası 4
    new DateTime(2002, 08, 01),
	// Veri noktası 5
    new DateTime(2002, 09, 01)
}, 
// Veri noktaları için karşılık gelen değerler
new double[] { 32, 32, 28, 12, 15 });

//ExStep:3-
// Belgeyi eklenen grafikle kaydet
doc.Save("WorkingWithCharts.InsertAreaChart.docx");
//ExEnd
