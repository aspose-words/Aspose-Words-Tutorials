//ExStepSummary:0: Bu kod, Aspose.Words kullanarak bir Word belgesinin nasıl oluşturulacağını ve içine bir sütun grafiğinin nasıl yerleştirileceğini gösterir.
//ExStepSummary:1: Bu kod, Word içeriğini manipüle etmek için bir belge oluşturucu nesnesinin nasıl oluşturulacağını gösterir.
//ExStepSummary:2: Bu kod, belgenin içine belirli boyutlara sahip bir sütun grafiği şeklinin nasıl yerleştirileceğini gösterir.
//ExStepSummary:3: Bu kod, daha fazla özelleştirme için eklenen formdan grafik nesnesini alır.
//ExStepSummary:4: Bu kod, grafik serisi koleksiyonuna nasıl erişileceğini ve seri sayısını nasıl yazdırılacağını gösterir.
//ExStepSummary:5: Bu kod, grafikle oluşturulmuş olabilecek varsayılan serileri temizler.
//ExStepSummary:6: Bu kod, grafik verileri için kategori adlarının nasıl tanımlanacağını gösterir.
//ExStepSummary:7: Bu kod, ilgili kategori adları ve veri değerleri ile çizelgeye serilerin nasıl ekleneceğini gösterir.
//ExStepSummary:8: Bu kod, eklenen grafikle belgeyi belirli bir dosyaya kaydeder.

//ExStep:0-
// Yeni bir Word belgesi örneği oluşturun.
Document doc = new Document();

//ExStep:1-
// Öğeleri belgeye kolayca eklemek için bir DocumentBuilder oluşturun.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:2-
// Belirtilen genişlik (432) ve yükseklik (252) ile bir sütun grafik şekli ekleyin.
Shape shape = builder.InsertChart(ChartType.Column, 432, 252);

//ExStep:3-
// Daha fazla manipülasyon için grafik nesnesini formdan alın.
Chart chart = shape.Chart;

//ExStep:4-
// Çizelgenin seri koleksiyonuna erişin.
ChartSeriesCollection seriesColl = chart.Series;
// Grafikteki seri sayısını konsola yazdır.
Console.WriteLine(seriesColl.Count);

//ExStep:5-
// Grafik oluşturulduğunda otomatik olarak oluşturulan varsayılan serileri kaldırın.
seriesColl.Clear();

//ExStep:6-
// Grafik verileri için bir dizi kategori adı tanımla.
string[] categories = new string[] { "Category 1", "Category 2" };

//ExStep:7-
// Kategori adları ve veri değerleri ile tabloya birden fazla dizi ekleyin.
seriesColl.Add("Aspose Series 1", categories, new double[] { 1, 2 });
seriesColl.Add("Aspose Series 2", categories, new double[] { 3, 4 });
seriesColl.Add("Aspose Series 3", categories, new double[] { 5, 6 });
seriesColl.Add("Aspose Series 4", categories, new double[] { 7, 8 });
seriesColl.Add("Aspose Series 5", categories, new double[] { 9, 10 });

//ExStep:8-
// Eklenen grafikle belgeyi belirtilen dosya yoluna kaydedin.
doc.Save("WorkingWithCharts.InsertSimpleColumnChart.docx");
//ExEnd
