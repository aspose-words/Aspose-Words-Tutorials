//ExSummary: Bu kod, Aspose.Words kullanarak bir Word belgesinde çok seviyeli bir listenin nasıl oluşturulacağını gösterir. Bir belgeyi başlatır, bir numaralandırma formatı uygular, ana ve alt öğeler oluşturur ve sonunda belgeyi bir .docx dosyası olarak kaydeder.
//ExStepSummary:0: Aşağıdaki kod yeni bir belge ve DocumentBuilder örneğini başlatır.
//ExStepSummary:1: Aşağıdaki kod, listeye varsayılan numaralandırma stili uygular ve üst düzey öğeler ekler.
//ExStepSummary:2: Aşağıdaki kod, ikinci üst düzey öğenin altında alt öğeler oluşturmak için girintiler.
//ExStepSummary:3: Aşağıdaki kod, ikinci alt-tem altında alt-temler oluşturmak için daha da girintilenir.
//ExStepSummary:4: Aşağıdaki kod, ikinci seviyeye başka bir öğe eklemek için bir önceki seviyeye geri gider.
//ExStepSummary:5: Aşağıdaki kod, listedeki son öğeyi eklemek için en üst seviyeye geri gider.
//ExStepSummary:6: Aşağıdaki kod, kaydetmeden önce numaralandırmayı listeden kaldırır.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStepImage:4:images/4.png
//ExStepImage:5:images/5.png
//ExStepImage:6:images/6.png
//ExStep:0-
// Yeni bir belgeyi başlat
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Varsayılan numaralandırma biçimini uygula
builder.ListFormat.ApplyNumberDefault();
builder.Writeln("Item 1");
builder.Writeln("Item 2");

//ExStep:2-
// Alt itemler için girinti
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.1");
builder.Writeln("Item 2.2");

//ExStep:3-
// Sub-sub-items için girinti
builder.ListFormat.ListIndent();
builder.Writeln("Item 2.2.1");
builder.Writeln("Item 2.2.2");

//ExStep:4-
// İkinci seviyeye geri döndü
builder.ListFormat.ListOutdent();
builder.Writeln("Item 2.3");

//ExStep:5-
// En üst seviyeye geri dön
builder.ListFormat.ListOutdent();
builder.Writeln("Item 3");

//ExStep:6-
// Listeden numaralandırmayı kaldır
builder.ListFormat.RemoveNumbers();

//ExStep:7-
// Belgeyi kaydet
doc.Save("DocumentFormatting.MultilevelListFormatting.docx");
//ExEnd
