//ExStepSummary:0: Belge yapısını yönetmek için yeni bir Belge ve Belge Oluşturucu başlatın.
//ExStepSummary:1: DocumentBuilder'ı kullanarak belgede yeni bir tablo başlatın.
//ExStepSummary:2: İlk hücreyi yerleştirin ve tablo stili tanımlayıcısını ayarlayın.
//ExStepSummary:3: Biçimlendirme seçeneklerini tabloya uygulayın ve otomatik uyum davranışını ayarlayın.
//ExStepSummary:4: Başlık satırı hücrelerini tabloya ekleyin.
//ExStepSummary:5: Veri satırlarını öğe adları ve miktarları ile tabloya ekleyin.
//ExStepSummary:6: Birleştirilmiş belgeyi bir dosyaya kaydedin.

//ExStep:0-
// Yeni bir Belge nesnesi oluştur.
Document doc = new Document();
// Belge oluşturmayı kolaylaştırmak için bir oluşturucu oluşturun.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Belgede yeni bir tablo başlatın.
Table table = builder.StartTable();

//ExStep:2-
// Biçimlendirmeyi ayarlamadan önce ilk hücreyi yerleştirin.
builder.InsertCell();
// Tablo için stil tanımlayıcısını ayarlayın.
table.StyleIdentifier = StyleIdentifier.MediumShading1Accent1;


//ExStep:3-
// Stil seçeneklerini yapılandırın.
table.StyleOptions = 
    TableStyleOptions.FirstColumn | TableStyleOptions.RowBands | TableStyleOptions.FirstRow;
// Masanın otomatik uyum davranışını ayarlayın.	
table.AutoFit(AutoFitBehavior.AutoFitToContents);


//ExStep:4-
// İlk sütun için başlığı yazın.
builder.Writeln("Item");
// Hücre için sağ dolguyu ayarlayın.
builder.CellFormat.RightPadding = 40;
// Sonraki sütun için ikinci hücreyi yerleştirin.
builder.InsertCell();
// İkinci sütun için başlığı yazın.
builder.Writeln("Quantity (kg)");
// Sıra yaratımına son verin.
builder.EndRow();

//ExStep:5-
// Yeni bir satır başlatın ve ilk hücreyi takın.
builder.InsertCell();
builder.Writeln("Apples");
builder.InsertCell();
builder.Writeln("20");
builder.EndRow();
 // İkinci öğe için yeni bir satırın ilk hücresini ekleyin.
builder.InsertCell();
builder.Writeln("Bananas");
builder.InsertCell();
builder.Writeln("40");
builder.EndRow();
// Üçüncü madde için yeni bir satır başlatın.
builder.InsertCell();
builder.Writeln("Carrots");
builder.InsertCell(); 
builder.Writeln("50");
builder.EndRow();

//ExStep:6-
// Belgeyi belirtilen bir dosyaya kaydedin.
doc.Save("WorkingWithTableStylesAndFormatting.BuildTableWithStyle.docx");
