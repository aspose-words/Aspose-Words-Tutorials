//ExSummary: Aşağıdaki kod, Aspose.Words'tan DocumentBuilder sınıfını kullanarak bir Word belgesinde bir tablonun nasıl oluşturulacağını ve biçimlendirileceğini gösterir.
//ExStepSummary:0: Word belgesini oluşturmaya başlamak için yeni bir Belge ve Belge Oluşturucuyu başlatın.
//ExStepSummary:1: Bir tablo oluşturun ve biçimlendirme için bir hücre ekleyin.
//ExStepSummary:2: Dolgu ve genişlik dahil olmak üzere hücrenin biçimlendirme özelliklerini ayarlayın.
//ExStepSummary:3: Biçimlendirilmiş hücreye metin ekle.
//ExStepSummary:4: Satır ve tabloyu sonlandırın ve belgeyi belirtilen bir dosya yoluna kaydedin.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0-
// Yeni bir Belge nesnesi oluşturma
Document doc = new Document();
// DocumentBuilder yeni Document ile başlatılıyor
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Bir tablo oluşturmaya başla
builder.StartTable();
// İlk hücreyi tabloya ekle
builder.InsertCell();

//ExStep:2-
// Hücrenin biçimlendirme özelliklerine erişme
CellFormat cellFormat = builder.CellFormat;
// Hücre genişliğinin ayarlanması
cellFormat.Width = 250;
// Dolguyu ayarla
cellFormat.LeftPadding = 30;
cellFormat.RightPadding = 30;
cellFormat.TopPadding = 30;
cellFormat.BottomPadding = 30;

//ExStep:3-
// Hücreye metin ekleme
builder.Writeln("I'm a wonderful formatted cell.");

//ExStep:4-
// Tablonun geçerli sırasını sonlandırmak
builder.EndRow();
// Tablo oluşturulmasının sona ermesi
builder.EndTable();
// Belge kaydediliyor
doc.Save("WorkingWithTableStylesAndFormatting.DocumentBuilderSetTableCellFormatting.docx");
//ExEnd
