//ExSummaryيوضح هذا الرمز كيفية إنشاء مستند Word باستخدام فئة DocumentBuilder وإدراج فقرة نمطية فيه.
//ExStepSummary:0: يقوم الكود التالي بتهيئة مستند وDocumentBuilder لإنشاء مستند.
//ExStepSummary:1: يقوم الكود التالي بتكوين إعدادات الخط للفقرة.
//ExStepSummary:2: يحدد الرمز التالي إعدادات تنسيق الفقرة للمحاذاة والمسافة البادئة.
//ExStepSummary:3: الرمز التالي يكتب فقرة في الوثيقة.
//ExStepSummary:4: يحفظ الكود التالي المستند كملف .docx.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0-
// تهيئة كائن مستند جديد
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// الوصول إلى إعدادات الخط من DocumentBuilder
Font font = builder.Font;
// اضبط حجم الخط على 16
font.Size = 16;
// اجعل الخط غامقًا
font.Bold = true;
// اضبط لون الخط على الأزرق
font.Color = Color.Blue;
// اضبط اسم الخط على Arial
font.Name = "Arial";
// أضف سطرًا سفليًا متقطعًا إلى الخط
font.Underline = Underline.Dash;

//ExStep:2-
// الوصول إلى إعدادات تنسيق الفقرة
ParagraphFormat paragraphFormat = builder.ParagraphFormat;
paragraphFormat.FirstLineIndent = 8;
paragraphFormat.Alignment = ParagraphAlignment.Justify;
paragraphFormat.KeepTogether = true;

//ExStep:3-
// اكتب فقرة من النص في المستند
builder.Writeln("A whole paragraph.");

//ExStep:4-
// احفظ المستند كملف .docx
doc.Save("AddContentUsingDocumentBuilder.InsertParagraph.docx");
