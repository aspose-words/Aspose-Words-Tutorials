//ExStepSummary:0: يؤدي الكود التالي إلى تهيئة مستند جديد وDocumentBuilder.
//ExStepSummary:1: يحدد الكود التالي خصائص الخط المختلفة مثل الحجم واللون واللون والاسم وأسلوب التسطير.
//ExStepSummary:2: يقوم الكود التالي بكتابة نص عينة إلى المستند باستخدام إعدادات الخط التي تم تكوينها.

//ExStep:0-
// إنشاء مثيل مستند جديد
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// الوصول إلى خاصية الخط من خلال DocumentBuilder
Font font = builder.Font;
// اضبط حجم الخط على 16
font.Size = 16;
// اجعل الخط غامقًا
font.Bold = true;
// تغيير لون الخط إلى الأزرق
font.Color = Color.Blue;
// اضبط الخط على Arial
font.Name = "Arial";
// تطبيق نمط التسطير المتقطع على الخط
font.Underline = Underline.Dash;

//ExStep:2-
builder.Write("Sample text.");

//ExStep:3-
// احفظ المستند في دليل القطع الأثرية
doc.Save("WorkingWithFonts.FontFormatting.docx");
