//ExStepSummary:0: تهيئة الوثيقة وDocumentBuilder كائنات لبدء العمل على الوثيقة.
//ExStepSummary:1: إدراج مخطط مبعثر في الوثيقة.
//ExStepSummary:2: إضافة سلسلة من نقاط البيانات إلى المخطط المبعثر.
//ExStepSummary:3: حفظ المستند مع المخطط المبعثر إلى ملف .docx.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// تهيئة مستند جديد
Document doc = new Document();
// إنشاء منشئ لإدراج المحتوى
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// أدرج مخططاً مبعثراً في المستند
Shape shape = builder.InsertChart(ChartType.Scatter, 432, 252); 

//ExStep:2-
// احصل على كائن المخطط من الشكل المدرج
Chart chart = shape.Chart;
// إضافة بيانات السلسلة إلى الرسم البياني
chart.Series.Add("Aspose Series 1", new double[] { 0.7, 1.8, 2.6 }, new double[] { 2.7, 3.2, 0.8 });

//ExStep:3-
// احفظ المستند باستخدام المخطط المبعثر
doc.Save("WorkingWithCharts.InsertScatterChart.docx");
//ExEnd
