//ExStepSummary:0: يقوم هذا القسم بتهيئة مثيل جديد لـ Document and DocumentBuilder.
//ExStepSummary:1: يطبق هذا القسم نمط فقرة العنوان باستخدام DocumentBuilder.
//ExStepSummary:2: هذا القسم يكتب "مرحبًا" على المستند.
//ExStepSummary:3: يحفظ هذا القسم المستند إلى مسار ملف محدد.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// تهيئة مثيل مستند جديد
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// اضبط نمط الفقرة على العنوان
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Title;

//ExStep:2-
// اكتب "Hello" باستخدام منشئ ، وتطبيق النمط المحدد
builder.Write("Hello");

//ExStep:3-
// احفظ المستند في ملف .docx
doc.Save("DocumentFormatting.ApplyParagraphStyle.docx");
