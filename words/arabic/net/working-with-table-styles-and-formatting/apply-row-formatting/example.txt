//ExStepSummary:2: يقوم بتكوين إعدادات الارتفاع المحددة.
//ExStepSummary:3: مجموعات الحشو للجدول بأكمله يؤثر على جميع الصفوف.
//ExStepSummary:4: يكتب النص في خلية الجدول وينهي الصف.
//ExStepSummary:5: يكمل عملية بناء الجدول ويحفظ المستند.

//ExStep:0-
// أنشئ مستندًا جديدًا
Document doc = new Document();
// إنشاء مستندبناء لتعديل المستند
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// ابدأ منضدة جديدة
Table table = builder.StartTable();
// إدراج a جديد خلية بوصة جدول
builder.InsertCell();
//ExStep:2-
// الوصول إلى خصائص تنسيق الصف
RowFormat rowFormat = builder.RowFormat;
// اضبط ارتفاع الصف
rowFormat.Height = 100;
// حدد أن الارتفاع دقيق
rowFormat.HeightRule = HeightRule.Exactly;

//ExStep:3-
// تعيين الحشو للجدول
table.LeftPadding = 30; 
table.RightPadding = 30;
table.TopPadding = 30;
table.BottomPadding = 30;

//ExStep:4-
// اكتب النص في الخلية
builder.Writeln("I'm a wonderful formatted row.");
// أنه الصف الحالي
builder.EndRow();
// إنهاء بناء الطاولة
builder.EndTable();

//ExStep:5-
// احفظ المستند
doc.Save("WorkingWithTableStylesAndFormatting.ApplyRowFormatting.docx");
