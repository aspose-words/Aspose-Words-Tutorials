//ExStepSummary:1: اكتب نصًا وأدخل فواصل أقسام لإنشاء أقسام متعددة في المستند.

//ExStepSummary:3: قم بإعداد المحتوى من القسم الأول إلى القسم الثالث.
//ExStepSummary:4: قم بإرفاق المحتوى من القسم الثاني إلى القسم الثالث.

//ExStep:0-
// تهيئة المستند وDocumentBuilder
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// اكتب نصًا وأدخل فواصل أقسام لإنشاء ثلاثة أقسام
builder.Write("Section 1");
builder.InsertBreak(BreakType.SectionBreakNewPage);
// إضافة محتوى للقسم الثاني
builder.Write("Section 2");
builder.InsertBreak(BreakType.SectionBreakNewPage);
// إضافة محتوى للقسم الثالث
builder.Write("Section 3");

//ExStep:2-
// استرجع القسم الثالث من المستند
Section section = doc.Sections[2];

//ExStep:3-
// قم بإعداد المحتوى من القسم الأول إلى القسم الثالث.
Section sectionToPrepend = doc.Sections[0];
section.PrependContent(sectionToPrepend);

//ExStep:4-
// تذييل المحتوى من القسم الثاني إلى القسم الثالث
Section sectionToAppend = doc.Sections[1];
section.AppendContent(sectionToAppend);
//ExEnd
