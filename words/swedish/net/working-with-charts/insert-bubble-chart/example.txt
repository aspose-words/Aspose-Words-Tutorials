//ExStart
//ExSummary: Denna kod visar hur man infogar ett Bubble diagram i ett Word-dokument med Aspose.Words.
//ExStepSummary:0: Följande kod initierar ett nytt dokument och DocumentBuilder.
//ExStepSummary:1: Ett Bubble diagram skapas och läggs till dokumentet med hjälp av DocumentBuilder.
//ExStepSummary:2: Dataserien för Bubble diagrammet läggs till med specifika X, Y, och storleksvärden.
//ExStepSummary:3: Slutligen sparas dokumentet med det infogade Bubble diagrammet.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Initialera ett nytt dokument och dokumentbyggare
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Infoga ett bubblar diagram i dokumentet
Shape shape = builder.InsertChart(ChartType.Bubble, 432, 252);

//ExStep:2-
// Lägg till dataserie till Bubble- diagrammet
Chart chart = shape.Chart;
chart.Series.Add("Aspose Series 1", new double[] { 0.7, 1.8, 2.6 }, // X values
    new double[] { 2.7, 3.2, 0.8 }, // Y values
    new double[] { 10, 4, 8 }); // Size values for the bubbles

//ExStep:3-
// Spara dokumentet med Bubble diagrammet
doc.Save("WorkingWithCharts.InsertBubbleChart.docx");
//ExEnd
