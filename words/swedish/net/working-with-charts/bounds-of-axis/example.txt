//ExSummary: Denna kod visar hur man skapar ett dokument med ett diagram med hjälp av Aspose.Words, ange gränserna för Y-axeln, och spara dokumentet som en DOCX- fil. 
// Varje steg bryts ned för att ge en tydlig förståelse för den process som är involverad i arbetet med diagram i Aspose.Words.
//ExStepSummary:0: Initiera ett nytt dokument och en instans för DocumentBuilder. 
//ExStepSummary:1: Infoga ett diagram i dokumentet och hämta diagramobjektet. 
//ExStepSummary:2: Lägg till en ny serie i diagrammet med angivna kategorier och värden. 
//ExStepSummary:3: Ställ in minsta och högsta gräns för y-axeln i diagrammet. 
//ExStepSummary:4: Spara dokumentet till angiven sökväg. 
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStepImage:3:images/3.png
//ExStep:0-
// Skapa en ny dokumentinstans.
Document doc = new Document();
// Initialisera DocumentBuilder för att bygga dokumentet.
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Infoga ett kolumndiagram i dokumentet.
Shape shape = builder.InsertChart(ChartType.Column, 432, 252);
// Hämta diagramobjektet från formen.
Chart chart = shape.Chart;
// Rensa alla befintliga serier från diagrammet för nya data.
chart.Series.Clear();

//ExStep:2-
// Definiera en ny serie med namnet "Aspose Serie 1".
chart.Series.Add("Aspose Series 1",
    new string[] { "Item 1", "Item 2", "Item 3", "Item 4", "Item 5" }, // Set categories for the series.
    new double[] { 1.2, 0.3, 2.1, 2.9, 4.2 });

//ExStep:3-
// Ställ in minsta gräns för Y- axeln.
chart.AxisY.Scaling.Minimum = new AxisBound(0);
chart.AxisY.Scaling.Maximum = new AxisBound(6);

//ExStep:4-
// Spara dokumentet med diagrammet till angiven sökväg.
doc.Save("WorkingWithCharts.BoundsOfAxis.docx");
//ExEnd
