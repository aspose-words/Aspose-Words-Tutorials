//ExStep:0-
// Skapa ett nytt dokumentobjekt
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Infoga ett linjediagram i dokumentet vid angivna mått
Shape shape = builder.InsertChart(ChartType.Line, 432, 252);

//ExStep:2-
// Få tillgång till diagramobjektet från formen
Chart chart = shape.Chart;
// Aktivera diagramrubrikens synlighet
chart.Title.Show = true;
// Ställ in titeltexten för diagrammet
chart.Title.Text = "Line Chart Title";
// Se till att titeln inte överlagrar diagrammet
chart.Title.Overlay = false;

//ExStep:3-
// Ställ in positionen för sjökortslegenden till vänster
chart.Legend.Position = LegendPosition.Left;
chart.Legend.Overlay = true;

//ExStep:4-
// Spara dokumentet med diagrammet till en fil
doc.Save("WorkingWithCharts.CreateChartUsingShape.docx");
