//ExStepSummary:2: Detta steg infogar en sida paus för att starta en ny sida.
//ExStepSummary:3: Detta steg lägger till text till andra sidan.
//ExStepSummary:4: Detta steg infogar en annan sida paus för en ny sida.

//ExStep:0-
// Skapa en ny dokumentinstans
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Lägg till text för första sidan
builder.Writeln("This is page 1.");

//ExStep:2-
// Infoga en sidbrytning för att starta en ny sida
builder.InsertBreak(BreakType.PageBreak);

//ExStep:3-
// Lägg till text för den andra sidan
builder.Writeln("This is page 2.");

//ExStep:4-
// Infoga en ny sidas paus för en ny sida
builder.InsertBreak(BreakType.PageBreak);

//ExStep:5-
// Lägg till text för tredje sidan
builder.Writeln("This is page 3.");

//ExStep:6-
// Spara dokumentet till angiven sökväg
doc.Save("AddContentUsingDocumentBuilder.InsertBreak.docx");
