//ExSummary: Denna kod visar hur man skapar ett Word-dokument med en innehållsförteckning med hjälp av DocumentBuilder. Den innehåller rubriker på olika nivåer och sparar dokumentet som en .docx-fil.
//ExStepSummary:0: Detta steg initierar ett nytt dokumentobjekt.
//ExStepSummary:1: Detta steg skapar ett DocumentBuilder-objekt för dokumentmanipulering.
//ExStepSummary:2: Detta steg infogar innehållsförteckningen i början av dokumentet.
//ExStepSummary:3: Detta steg introducerar en sidbrytning för att starta själva innehållet på en ny sida.
//ExStepSummary:4: Detta steg sätter den första rubriken och skriver "Heading 1" till dokumentet.
//ExStepSummary:5: Detta steg sätter den andra rubriknivån och skriver "Heading 1.1" och "Heading 1.2".
//ExStepSummary:6: Detta steg sätter återigen den första rubriknivån och skriver "Heading 2" och "Heading 3".
//ExStepSummary:7: Detta steg sätter den andra rubriknivån och skriver "Heading 3.1".
//ExStepSummary:8: Detta steg anger den tredje rubriknivån och skriver underrubrikerna "Heading 3.1.1", "Heading 3.1.2" och "Heading 3.1.3".
//ExStepSummary:9: Detta steg sätter återigen den andra rubriknivån och skriver "Heading 3.2" och "Heading 3.3".
//ExStepSummary:10: Slutligen, detta steg uppdaterar fälten i dokumentet och sparar det som en .docx-fil.

//ExStep:0-
// Skapa en ny dokumentinstans
Document doc = new Document();

//ExStep:1-
// Initialisera DocumentBuilder för dokumentmanipulering
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:2-
// Infoga en innehållsförteckning med angivna alternativ
builder.InsertTableOfContents("\\o \"1-3\" \\h \\z \\u");

//ExStep:3-
// Infoga en sidbrytning för att starta innehåll på en ny sida
builder.InsertBreak(BreakType.PageBreak);

//ExStep:4-
// Ställ in stil till rubrik 1
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading1;
builder.Writeln("Heading 1");

//ExStep:5-
// Ställ in stil till rubrik 2
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading2;
builder.Writeln("Heading 1.1");
builder.Writeln("Heading 1.2");

//ExStep:6-
// Ställ tillbaka stil till rubrik 1
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading1;
builder.Writeln("Heading 2");
builder.Writeln("Heading 3");

//ExStep:7-
// Ställ in stil till rubrik 2
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading2;
builder.Writeln("Heading 3.1");

//ExStep:8-
// Ställ in stil till rubrik 3
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading3;
builder.Writeln("Heading 3.1.1");
builder.Writeln("Heading 3.1.2");
builder.Writeln("Heading 3.1.3");

//ExStep:9-
// Ställ tillbaka stilen till rubrik 2
builder.ParagraphFormat.StyleIdentifier = StyleIdentifier.Heading2;
builder.Writeln("Heading 3.2");
builder.Writeln("Heading 3.3");

//ExStep:10-
// Uppdatera alla fält i dokumentet
doc.UpdateFields();
// Spara dokumentet
doc.Save("AddContentUsingDocumentBuilder.InsertTableOfContents.docx");
//ExEnd
