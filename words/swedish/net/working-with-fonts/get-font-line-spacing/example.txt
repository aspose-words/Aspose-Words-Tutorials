//ExStepSummary:0: Följande kod initierar ett nytt dokument och DocumentBuilder.
//ExStepSummary:1: Koden anger teckensnittet till "Calibri" och lägger till en textrad.
//ExStepSummary:2: Koden hämtar radavståndet för den första körningen i dokumentet.
//ExStepImage:1:images/1.png
//ExStepImage:2:images/2.png
//ExStep:0-
// Skapa en ny dokumentinstans
Document doc = new Document();
DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Ställ in teckensnittets namn till "Calibri"
builder.Font.Name = "Calibri";
builder.Writeln("qText");

//ExStep:2-
// Få tillgång till teckensnittet för den första körningen
Font font = builder.Document.FirstSection.Body.FirstParagraph.Runs[0].Font;
// Utmatning av radavståndet till konsolen
Console.WriteLine($"lineSpacing = {font.LineSpacing}");
