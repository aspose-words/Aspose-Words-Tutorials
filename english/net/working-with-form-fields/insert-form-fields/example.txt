// Initializes an empty Word document

DocumentBuilder builder = new DocumentBuilder(doc);

//ExStep:1-
// Define and insert a combo box form field
string[] items = { "One", "Two", "Three" };
builder.InsertComboBox("DropDown", items, 0);
